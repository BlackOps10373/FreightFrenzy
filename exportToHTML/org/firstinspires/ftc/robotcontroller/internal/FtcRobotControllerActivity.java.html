209    @Override
210    protected void onNewIntent(Intent intent) {
211      super.onNewIntent(intent);
212  
213      if (UsbManager.ACTION_USB_DEVICE_ATTACHED.equals(intent.getAction())) {
214        UsbDevice usbDevice = intent.getParcelableExtra(UsbManager.EXTRA_DEVICE);
215        RobotLog.vv(TAG, "ACTION_USB_DEVICE_ATTACHED: %s", usbDevice.getDeviceName());
216  
217        if (usbDevice != null) {  // paranoia
218          // We might get attachment notifications before the event loop is set up, so
219          // we hold on to them and pass them along only when we're good and ready.
220          if (receivedUsbAttachmentNotifications != null) { // *total* paranoia
221            receivedUsbAttachmentNotifications.add(usbDevice);
222            passReceivedUsbAttachmentsToEventLoop();
223          }
224        }
225      }
226    }
227  
228    protected void passReceivedUsbAttachmentsToEventLoop() {
229      if (this.eventLoop != null) {
230        for (;;) {
231          UsbDevice usbDevice = receivedUsbAttachmentNotifications.poll();
232          if (usbDevice == null)
233            break;
234          this.eventLoop.onUsbDeviceAttached(usbDevice);
235        }
236      }
237      else {
238        // Paranoia: we don't want the pending list to grow without bound when we don't
239        // (yet) have an event loop
240        while (receivedUsbAttachmentNotifications.size() > 100) {
241          receivedUsbAttachmentNotifications.poll();
242        }
243      }
244    }

246    /**
247     * There are cases where a permission may be revoked and the system restart will restart the
248     * FtcRobotControllerActivity, instead of the launch activity.  Detect when that happens, and throw
249     * the device back to the permission validator activity.
250     */
251    protected boolean enforcePermissionValidator() {
252      if (!permissionsValidated) {
253        RobotLog.vv(TAG, "Redirecting to permission validator");
254        Intent permissionValidatorIntent = new Intent(AppUtil.getDefContext(), PermissionValidatorWrapper.class);
255        startActivity(permissionValidatorIntent);
256        finish();
257        return true;
258      } else {
259        RobotLog.vv(TAG, "Permissions validated already");
260        return false;
261      }
262    }

515    protected void readNetworkType() {
516      // Control hubs are always running the access point model.  Everything else, for the time
517      // being always runs the Wi-Fi Direct model.
518      if (Device.isRevControlHub() == true) {
519        networkType = NetworkType.RCWIRELESSAP;
520      } else {
521        networkType = NetworkType.fromString(preferencesHelper.readString(context.getString(R.string.pref_pairing_kind), NetworkType.globalDefaultAsString()));
522      }
523  
524      // update the app_settings
525      preferencesHelper.writeStringPrefIfDifferent(context.getString(R.string.pref_pairing_kind), networkType.toString());
526    }

545    private boolean isRobotRunning() {
546      if (controllerService == null) {
547        return false;
548      }
549  
550      Robot robot = controllerService.getRobot();
551  
552      if ((robot == null) || (robot.eventLoopManager == null)) {
553        return false;
554      }
555  
556      RobotState robotState = robot.eventLoopManager.state;
557  
558      if (robotState != RobotState.RUNNING) {
559        return false;
560      } else {
561        return true;
562      }
563    }

565    @Override
566    public boolean onOptionsItemSelected(MenuItem item) {
567      int id = item.getItemId();
568  
569      if (id == R.id.action_program_and_manage) {
570        if (isRobotRunning()) {
571          Intent programmingModeIntent = new Intent(AppUtil.getDefContext(), ProgramAndManageActivity.class);
572          RobotControllerWebInfo webInfo = programmingModeManager.getWebServer().getConnectionInformation();
573          programmingModeIntent.putExtra(LaunchActivityConstantsList.RC_WEB_INFO, webInfo.toJson());
574          startActivity(programmingModeIntent);
575        } else {
576          AppUtil.getInstance().showToast(UILocation.ONLY_LOCAL, context.getString(R.string.toastWifiUpBeforeProgrammingMode));
577        }
578      } else if (id == R.id.action_inspection_mode) {
579        Intent inspectionModeIntent = new Intent(AppUtil.getDefContext(), RcInspectionActivity.class);
580        startActivity(inspectionModeIntent);
581        return true;
582      } else if (id == R.id.action_restart_robot) {
583        dimmer.handleDimTimer();
584        AppUtil.getInstance().showToast(UILocation.BOTH, context.getString(R.string.toastRestartingRobot));
585        requestRobotRestart();
586        return true;
587      }
588      else if (id == R.id.action_configure_robot) {
589        EditParameters parameters = new EditParameters();
590        Intent intentConfigure = new Intent(AppUtil.getDefContext(), FtcLoadFileActivity.class);
591        parameters.putIntent(intentConfigure);
592        startActivityForResult(intentConfigure, RequestCode.CONFIGURE_ROBOT_CONTROLLER.ordinal());
593      }
594      else if (id == R.id.action_settings) {
595  	  // historical: this once erroneously used FTC_CONFIGURE_REQUEST_CODE_ROBOT_CONTROLLER
596        Intent settingsIntent = new Intent(AppUtil.getDefContext(), FtcRobotControllerSettingsActivity.class);
597        startActivityForResult(settingsIntent, RequestCode.SETTINGS_ROBOT_CONTROLLER.ordinal());
598        return true;
599      }
600      else if (id == R.id.action_about) {
601        Intent intent = new Intent(AppUtil.getDefContext(), FtcAboutActivity.class);
602        startActivity(intent);
603        return true;
604      }
605      else if (id == R.id.action_exit_app) {
606  
607        //Clear backstack and everything to prevent edge case where VM might be
608        //restarted (after it was exited) if more than one activity was on the
609        //backstack for some reason.
610        finishAffinity();
611  
612        //For lollipop and up, we can clear ourselves from the recents list too
613        if (android.os.Build.VERSION.SDK_INT >= android.os.Build.VERSION_CODES.LOLLIPOP) {
614          ActivityManager manager = (ActivityManager) getSystemService(ACTIVITY_SERVICE);
615          List<ActivityManager.AppTask> tasks = manager.getAppTasks();
616  
617          for (ActivityManager.AppTask task : tasks) {
618            task.finishAndRemoveTask();
619          }
620        }
621  
622        // Allow the user to use the Control Hub operating system's UI, instead of relaunching the app
623        AppAliveNotifier.getInstance().disableAppWatchdogUntilNextAppStart();
624  
625        //Finally, nuke the VM from orbit
626        AppUtil.getInstance().exitApplication();
627  
628        return true;
629      }
630  
631     return super.onOptionsItemSelected(item);
632    }

641    /**
642     * Updates the orientation of monitorContainer (which contains cameraMonitorView and
643     * tfodMonitorView) based on the given configuration. Makes the children split the space.
644     */
645    private void updateMonitorLayout(Configuration configuration) {
646      LinearLayout monitorContainer = (LinearLayout) findViewById(R.id.monitorContainer);
647      if (configuration.orientation == Configuration.ORIENTATION_LANDSCAPE) {
648        // When the phone is landscape, lay out the monitor views horizontally.
649        monitorContainer.setOrientation(LinearLayout.HORIZONTAL);
650        for (int i = 0; i < monitorContainer.getChildCount(); i++) {
651          View view = monitorContainer.getChildAt(i);
652          view.setLayoutParams(new LayoutParams(0, LayoutParams.MATCH_PARENT, 1 /* weight */));
653        }
654      } else {
655        // When the phone is portrait, lay out the monitor views vertically.
656        monitorContainer.setOrientation(LinearLayout.VERTICAL);
657        for (int i = 0; i < monitorContainer.getChildCount(); i++) {
658          View view = monitorContainer.getChildAt(i);
659          view.setLayoutParams(new LayoutParams(LayoutParams.MATCH_PARENT, 0, 1 /* weight */));
660        }
661      }
662      monitorContainer.requestLayout();
663    }

712    private void updateUIAndRequestRobotSetup() {
713      if (controllerService != null) {
714        callback.networkConnectionUpdate(controllerService.getNetworkConnectionStatus());
715        callback.updateRobotStatus(controllerService.getRobotStatus());
716        // Only show this first-time toast on headless systems: what we have now on non-headless suffices
717        requestRobotSetup(LynxConstants.isRevControlHub()
718          ? new Runnable() {
719              @Override public void run() {
720                showRestartRobotCompleteToast(R.string.toastRobotSetupComplete);
721              }
722            }
723          : null);
724      }

727    private void requestRobotSetup(@Nullable Runnable runOnComplete) {
728      if (controllerService == null) return;
729  
730      RobotConfigFile file = cfgFileMgr.getActiveConfigAndUpdateUI();
731      HardwareFactory hardwareFactory = new HardwareFactory(context);
732      try {
733        hardwareFactory.setXmlPullParser(file.getXml());
734      } catch (FileNotFoundException | XmlPullParserException e) {
735        RobotLog.ww(TAG, e, "Unable to set configuration file %s. Falling back on noConfig.", file.getName());
736        file = RobotConfigFile.noConfig(cfgFileMgr);
737        try {
738          hardwareFactory.setXmlPullParser(file.getXml());
739          cfgFileMgr.setActiveConfigAndUpdateUI(false, file);
740        } catch (FileNotFoundException | XmlPullParserException e1) {
741          RobotLog.ee(TAG, e1, "Failed to fall back on noConfig");
742        }
743      }
744  
745      OpModeRegister userOpModeRegister = createOpModeRegister();
746      eventLoop = new FtcEventLoop(hardwareFactory, userOpModeRegister, callback, this);
747      FtcEventLoopIdle idleLoop = new FtcEventLoopIdle(hardwareFactory, userOpModeRegister, callback, this);
748  
749      controllerService.setCallback(callback);
750      controllerService.setupRobot(eventLoop, idleLoop, runOnComplete);
751  
752      passReceivedUsbAttachmentsToEventLoop();
753      AndroidBoard.showErrorIfUnknownControlHub();
754  
755      AnnotatedHooksClassFilter.getInstance().callOnCreateEventLoopMethods(this, eventLoop);
756    }

783    private void checkPreferredChannel() {
784      // For P2P network, check to see what preferred channel is.
785      if (networkType ==  NetworkType.WIFIDIRECT) {
786        int prefChannel = preferencesHelper.readInt(getString(com.qualcomm.ftccommon.R.string.pref_wifip2p_channel), -1);
787        if (prefChannel == -1) {
788          prefChannel = 0;
789          RobotLog.vv(TAG, "pref_wifip2p_channel: No preferred channel defined. Will use a default value of %d", prefChannel);
790        } else {
791          RobotLog.vv(TAG, "pref_wifip2p_channel: Found existing preferred channel (%d).", prefChannel);
792        }
793  
794        // attempt to set the preferred channel.
795        RobotLog.vv(TAG, "pref_wifip2p_channel: attempting to set preferred channel...");
796        wifiDirectChannelChanger = new WifiDirectChannelChanger();
797        wifiDirectChannelChanger.changeToChannel(prefChannel);
798      }
799    }

829    protected void initWifiMute(boolean enable) {
830      if (enable) {
831        wifiMuteStateMachine = new WifiMuteStateMachine();
832        wifiMuteStateMachine.initialize();
833        wifiMuteStateMachine.start();
834  
835        motionDetection = new MotionDetection(2.0, 10);
836        motionDetection.startListening();
837        motionDetection.registerListener(new MotionDetection.MotionDetectionListener() {
838          @Override
839          public void onMotionDetected(double vector)
840          {
841            wifiMuteStateMachine.consumeEvent(WifiMuteEvent.USER_ACTIVITY);
842          }
843        });
844      } else {
845        wifiMuteStateMachine.stop();
846        wifiMuteStateMachine = null;
847        motionDetection.stopListening();
848        motionDetection.purgeListeners();
849        motionDetection = null;
850      }
851    }
