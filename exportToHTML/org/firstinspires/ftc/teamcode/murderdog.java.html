201      private void turnPower(double amount){
202          // use a positive parameter to turn right (clockwise)
203          // use a negative parameter to turn left (counterclockwise)
204          leftPower += amount;
205          rightPower -= amount;
206          backLeftPower += amount;
207          backRightPower -= amount;
208      }

210      public int degreeCalc(int degree)
211      {
212          int returnDegree = degree;
213          if(returnDegree < 0){
214              returnDegree = returnDegree + 360;
215          }
216          if(returnDegree >= 360){
217              returnDegree = returnDegree - 360;
218          }
219          return returnDegree;

221      }
