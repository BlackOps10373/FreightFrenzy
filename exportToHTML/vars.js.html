58   function createNonEditableField(label) {
59     var field = new Blockly.FieldTextInput(label);
60     field.CURSOR = '';
61     field.showEditor_ = function(opt_quietInput) {};
62     return field;
63   }

65   function createFieldDropdown(choices) {
66     if (choices.length == 0) {
67       return createNonEditableField('');
68     }
69     // Disable validation. We'll show a warning if the newValue is not in the choices. This can
70     // happen if the hardware configuration has changed.
71     var field = new Blockly.FieldDropdown(choices);
72     field.doClassValidation_ = function(newValue) {
73       return newValue;
74     };
75     return field;
76   }

78   function isJavaIdentifierStart(c) {
79     return /[a-zA-Z$_]/.test(c);
80   }
81   
82   function isJavaIdentifierPart(c) {
83     return /[a-zA-Z0-9$_]/.test(c);
84   }
85   
86   function containsAtLeastOneAlphanumeric(s) {
87     return /[a-zA-Z0-9]+/.test(s);
88   }

90   function containsAmpersand(s) {
91     return /&+/.test(s);
92   }

94   function makeIdentifier(deviceName) {
95     var identifier = '';
96   
97     var c = deviceName.charAt(0);
98     if (isJavaIdentifierStart(c)) {
99       identifier += c;
100    } else if (isJavaIdentifierPart(c)) {
101      identifier += ('_' + c);
102    }
103  
104    for (var i = 1; i < deviceName.length; i++) {
105      c = deviceName.charAt(i);
106      if (isJavaIdentifierPart(c)) {
107        identifier += c;
108      }
109    }
110    return identifier;
111  }

113  function escapeHtml(text) {
114    var out = '';
115    for (var i = 0; i < text.length; i++) {
116      var c = text.charAt(i);
117      if (c == ' ') {
118        out += '&nbsp;';
119      } else if (c == '<') {
120        out += '&lt;'
121      } else if (c == '>') {
122        out += '&gt;';
123      } else if (c == '&') {
124        out += '&amp;';
125      } else if (c > 0x7E || c < ' ') {
126        out += ('&#' + text.charCodeAt(i) + ';');
127      } else {
128        out += c;
129      }
130    }
131    return out;
132  }

134  function formatExtraXml(flavor, group, autoTransition, enabled) {
135    return '<?xml version=\'1.0\' encoding=\'UTF-8\' standalone=\'yes\' ?>' +
136        '<Extra>' +
137        '<OpModeMeta flavor="' + flavor + '" group="' + group + '" autoTransition="' + autoTransition + '" />' +
138        '<Enabled value="' + enabled + '" />' +
139        '</Extra> ';
140  }

142  function parseExtraXml(blkFileContent) {
143    var extra = Object.create(null);
144    extra['flavor'] = 'TELEOP';
145    extra['group'] = '';
146    extra['autoTransition'] = '';
147    extra['enabled'] = true;
148  
149    // The blocks content is up to and including the first </xml>.
150    var i = blkFileContent.indexOf('</xml>');
151    // The extra xml content is after the first </xml>.
152    // Set OpModeMeta and Enabled UI components.
153    var extraXml = blkFileContent.substring(i + 6); // 6 is length of </xml>
154    if (extraXml.length > 0) {
155      var parser = new DOMParser();
156      var xmlDoc = parser.parseFromString(extraXml.trim(), 'text/xml');
157      var opModeMetaElements = xmlDoc.getElementsByTagName('OpModeMeta');
158      if (opModeMetaElements.length >= 1) {
159        extra['flavor'] = opModeMetaElements[0].getAttribute('flavor');
160        extra['group'] = opModeMetaElements[0].getAttribute('group');
161        extra['autoTransition'] = opModeMetaElements[0].getAttribute('autoTransition');
162      }
163      var enabledElements = xmlDoc.getElementsByTagName('Enabled');
164      if (enabledElements.length >= 1) {
165        var enabledString = enabledElements[0].getAttribute('value');
166        if (enabledString) {
167          extra['enabled'] = (enabledString == 'true');
168        }
169      }
170    }
171    return extra;
172  }
