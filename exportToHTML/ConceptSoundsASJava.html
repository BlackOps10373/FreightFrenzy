30   package org.firstinspires.ftc.robotcontroller.external.samples;
31   
32   import com.qualcomm.ftccommon.SoundPlayer;
33   import com.qualcomm.robotcore.eventloop.opmode.Disabled;
34   import com.qualcomm.robotcore.eventloop.opmode.LinearOpMode;
35   import com.qualcomm.robotcore.eventloop.opmode.TeleOp;
36   
37   import java.io.File;
38   
39   /**
40    * This file demonstrates how to play simple sounds on both the RC and DS phones.
41    * It illustrates how to build sounds into your application as a resource.
42    * This technique is best suited for use with Android Studio since it assumes you will be creating a new application
43    *
44    * If you are using OnBotJava, please see the ConceptSoundsOnBotJava sample
45    *
46    * Use Android Studios to Copy this Class, and Paste it into your team's code folder with a new name.
47    * Remove or comment out the @Disabled line to add this opmode to the Driver Station OpMode list
48    *
49    * Operation:
50    *
51    * Gamepad X & B buttons are used to trigger sounds in this example, but any event can be used.
52    * Note: Time should be allowed for sounds to complete before playing other sounds.
53    *
54    * For sound files to be used as a compiled-in resource, they need to be located in a folder called "raw" under your "res" (resources) folder.
55    * You can create your own "raw" folder from scratch, or you can copy the one from the FtcRobotController module.
56    *
57    *     Android Studio coders will ultimately need a folder in your path as follows:
58    *       <project root>/TeamCode/src/main/res/raw
59    *
60    *     Copy any .wav files you want to play into this folder.
61    *     Make sure that your files ONLY use lower-case characters, and have no spaces or special characters other than underscore.
62    *
63    *     The name you give your .wav files will become the resource ID for these sounds.
64    *     eg:  gold.wav becomes R.raw.gold
65    *
66    *     If you wish to use the sounds provided for this sample, they are located in:
67    *     <project root>/FtcRobotController/src/main/res/raw
68    *     You can copy and paste the entire 'raw' folder using Android Studio.
69    *
70    */
71   
72   @TeleOp(name="Concept: Sound Resources", group="Concept")
73   @Disabled
74   public class ConceptSoundsASJava extends LinearOpMode {
75   
76       // Declare OpMode members.
77       private boolean goldFound;      // Sound file present flags
78       private boolean silverFound;
79   
80       private boolean isX = false;    // Gamepad button state variables
81       private boolean isB = false;
82   
83       private boolean wasX = false;   // Gamepad button history variables
84       private boolean WasB = false;
85   
86       @Override
87       public void runOpMode() {
88   
89           // Determine Resource IDs for sounds built into the RC application.
90           int silverSoundID = hardwareMap.appContext.getResources().getIdentifier("silver", "raw", hardwareMap.appContext.getPackageName());
91           int goldSoundID   = hardwareMap.appContext.getResources().getIdentifier("gold",   "raw", hardwareMap.appContext.getPackageName());
92   
93           // Determine if sound resources are found.
94           // Note: Preloading is NOT required, but it's a good way to verify all your sounds are available before you run.
95           if (goldSoundID != 0)
96               goldFound   = SoundPlayer.getInstance().preload(hardwareMap.appContext, goldSoundID);
97   
98           if (silverSoundID != 0)
99               silverFound = SoundPlayer.getInstance().preload(hardwareMap.appContext, silverSoundID);
100  
101          // Display sound status
102          telemetry.addData("gold resource",   goldFound ?   "Found" : "NOT found\n Add gold.wav to /src/main/res/raw" );
103          telemetry.addData("silver resource", silverFound ? "Found" : "Not found\n Add silver.wav to /src/main/res/raw" );
104  
105          // Wait for the game to start (driver presses PLAY)
106          telemetry.addData(">", "Press Start to continue");
107          telemetry.update();
108          waitForStart();
109  
110          telemetry.addData(">", "Press X, B to play sounds.");
111          telemetry.update();
112  
113          // run until the end of the match (driver presses STOP)
114          while (opModeIsActive()) {
115  
116              // say Silver each time gamepad X is pressed (This sound is a resource)
117              if (silverFound && (isX = gamepad1.x) && !wasX) {
118                  SoundPlayer.getInstance().startPlaying(hardwareMap.appContext, silverSoundID);
119                  telemetry.addData("Playing", "Resource Silver");
120                  telemetry.update();
121              }
122  
123              // say Gold each time gamepad B is pressed  (This sound is a resource)
124              if (goldFound && (isB = gamepad1.b) && !WasB) {
125                  SoundPlayer.getInstance().startPlaying(hardwareMap.appContext, goldSoundID);
126                  telemetry.addData("Playing", "Resource Gold");
127                  telemetry.update();
128              }
129  
130              // Save last button states
131              wasX = isX;
132              WasB = isB;
133          }
134      }
135  }
