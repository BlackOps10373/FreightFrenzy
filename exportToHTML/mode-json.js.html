32   "use strict";
33   
34   var oop = require("../lib/oop");
35   var TextHighlightRules = require("./text_highlight_rules").TextHighlightRules;
36   
37   var JsonHighlightRules = function() {
38       this.$rules = {
39           "start" : [
40               {
41                   token : "variable", // single line
42                   regex : '["](?:(?:\\\\.)|(?:[^"\\\\]))*?["]\\s*(?=:)'
43               }, {
44                   token : "string", // single line
45                   regex : '"',
46                   next  : "string"
47               }, {
48                   token : "constant.numeric", // hex
49                   regex : "0[xX][0-9a-fA-F]+\\b"
50               }, {
51                   token : "constant.numeric", // float
52                   regex : "[+-]?\\d+(?:(?:\\.\\d*)?(?:[eE][+-]?\\d+)?)?\\b"
53               }, {
54                   token : "constant.language.boolean",
55                   regex : "(?:true|false)\\b"
56               }, {
57                   token : "text", // single quoted strings are not allowed
58                   regex : "['](?:(?:\\\\.)|(?:[^'\\\\]))*?[']"
59               }, {
60                   token : "comment", // comments are not allowed, but who cares?
61                   regex : "\\/\\/.*$"
62               }, {
63                   token : "comment.start", // comments are not allowed, but who cares?
64                   regex : "\\/\\*",
65                   next  : "comment"
66               }, {
67                   token : "paren.lparen",
68                   regex : "[[({]"
69               }, {
70                   token : "paren.rparen",
71                   regex : "[\\])}]"
72               }, {
73                   token : "text",
74                   regex : "\\s+"
75               }
76           ],
77           "string" : [
78               {
79                   token : "constant.language.escape",
80                   regex : /\\(?:x[0-9a-fA-F]{2}|u[0-9a-fA-F]{4}|["\\\/bfnrt])/
81               }, {
82                   token : "string",
83                   regex : '"|$',
84                   next  : "start"
85               }, {
86                   defaultToken : "string"
87               }
88           ],
89           "comment" : [
90               {
91                   token : "comment.end", // comments are not allowed, but who cares?
92                   regex : "\\*\\/",
93                   next  : "start"
94               }, {
95                   defaultToken: "comment"
96               }
97           ]
98       };
99       
100  };
101  
102  oop.inherits(JsonHighlightRules, TextHighlightRules);
103  
104  exports.JsonHighlightRules = JsonHighlightRules;

107  define("ace/mode/matching_brace_outdent",["require","exports","module","ace/range"], function(require, exports, module) {
108  "use strict";
109  
110  var Range = require("../range").Range;
111  
112  var MatchingBraceOutdent = function() {};
113  
114  (function() {
115  
116      this.checkOutdent = function(line, input) {
117          if (! /^\s+$/.test(line))
118              return false;
119  
120          return /^\s*\}/.test(input);
121      };
122  
123      this.autoOutdent = function(doc, row) {
124          var line = doc.getLine(row);
125          var match = line.match(/^(\s*\})/);
126  
127          if (!match) return 0;
128  
129          var column = match[1].length;
130          var openBracePos = doc.findMatchingBracket({row: row, column: column});
131  
132          if (!openBracePos || openBracePos.row == row) return 0;
133  
134          var indent = this.$getIndent(doc.getLine(openBracePos.row));
135          doc.replace(new Range(row, 0, row, column-1), indent);
136      };
137  
138      this.$getIndent = function(line) {
139          return line.match(/^\s*/)[0];
140      };
141  
142  }).call(MatchingBraceOutdent.prototype);
143  
144  exports.MatchingBraceOutdent = MatchingBraceOutdent;
145  });

147  define("ace/mode/folding/cstyle",["require","exports","module","ace/lib/oop","ace/range","ace/mode/folding/fold_mode"], function(require, exports, module) {
148  "use strict";
149  
150  var oop = require("../../lib/oop");
151  var Range = require("../../range").Range;
152  var BaseFoldMode = require("./fold_mode").FoldMode;
153  
154  var FoldMode = exports.FoldMode = function(commentRegex) {
155      if (commentRegex) {
156          this.foldingStartMarker = new RegExp(
157              this.foldingStartMarker.source.replace(/\|[^|]*?$/, "|" + commentRegex.start)
158          );
159          this.foldingStopMarker = new RegExp(
160              this.foldingStopMarker.source.replace(/\|[^|]*?$/, "|" + commentRegex.end)
161          );
162      }
163  };
164  oop.inherits(FoldMode, BaseFoldMode);
165  
166  (function() {
167      
168      this.foldingStartMarker = /([\{\[\(])[^\}\]\)]*$|^\s*(\/\*)/;
169      this.foldingStopMarker = /^[^\[\{\(]*([\}\]\)])|^[\s\*]*(\*\/)/;
170      this.singleLineBlockCommentRe= /^\s*(\/\*).*\*\/\s*$/;
171      this.tripleStarBlockCommentRe = /^\s*(\/\*\*\*).*\*\/\s*$/;
172      this.startRegionRe = /^\s*(\/\*|\/\/)#?region\b/;
173      this._getFoldWidgetBase = this.getFoldWidget;
174      this.getFoldWidget = function(session, foldStyle, row) {
175          var line = session.getLine(row);
176      
177          if (this.singleLineBlockCommentRe.test(line)) {
178              if (!this.startRegionRe.test(line) && !this.tripleStarBlockCommentRe.test(line))
179                  return "";
180          }
181      
182          var fw = this._getFoldWidgetBase(session, foldStyle, row);
183      
184          if (!fw && this.startRegionRe.test(line))
185              return "start"; // lineCommentRegionStart
186      
187          return fw;
188      };
189  
190      this.getFoldWidgetRange = function(session, foldStyle, row, forceMultiline) {
191          var line = session.getLine(row);
192          
193          if (this.startRegionRe.test(line))
194              return this.getCommentRegionBlock(session, line, row);
195          
196          var match = line.match(this.foldingStartMarker);
197          if (match) {
198              var i = match.index;
199  
200              if (match[1])
201                  return this.openingBracketBlock(session, match[1], row, i);
202                  
203              var range = session.getCommentFoldRange(row, i + match[0].length, 1);
204              
205              if (range && !range.isMultiLine()) {
206                  if (forceMultiline) {
207                      range = this.getSectionRange(session, row);
208                  } else if (foldStyle != "all")
209                      range = null;
210              }
211              
212              return range;
213          }
214  
215          if (foldStyle === "markbegin")
216              return;
217  
218          var match = line.match(this.foldingStopMarker);
219          if (match) {
220              var i = match.index + match[0].length;
221  
222              if (match[1])
223                  return this.closingBracketBlock(session, match[1], row, i);
224  
225              return session.getCommentFoldRange(row, i, -1);
226          }
227      };
228      
229      this.getSectionRange = function(session, row) {
230          var line = session.getLine(row);
231          var startIndent = line.search(/\S/);
232          var startRow = row;
233          var startColumn = line.length;
234          row = row + 1;
235          var endRow = row;
236          var maxRow = session.getLength();
237          while (++row < maxRow) {
238              line = session.getLine(row);
239              var indent = line.search(/\S/);
240              if (indent === -1)
241                  continue;
242              if  (startIndent > indent)
243                  break;
244              var subRange = this.getFoldWidgetRange(session, "all", row);
245              
246              if (subRange) {
247                  if (subRange.start.row <= startRow) {
248                      break;
249                  } else if (subRange.isMultiLine()) {
250                      row = subRange.end.row;
251                  } else if (startIndent == indent) {
252                      break;
253                  }
254              }
255              endRow = row;
256          }
257          
258          return new Range(startRow, startColumn, endRow, session.getLine(endRow).length);
259      };
260      this.getCommentRegionBlock = function(session, line, row) {
261          var startColumn = line.search(/\s*$/);
262          var maxRow = session.getLength();
263          var startRow = row;
264          
265          var re = /^\s*(?:\/\*|\/\/|--)#?(end)?region\b/;
266          var depth = 1;
267          while (++row < maxRow) {
268              line = session.getLine(row);
269              var m = re.exec(line);
270              if (!m) continue;
271              if (m[1]) depth--;
272              else depth++;
273  
274              if (!depth) break;
275          }
276  
277          var endRow = row;
278          if (endRow > startRow) {
279              return new Range(startRow, startColumn, endRow, line.length);
280          }
281      };
282  
283  }).call(FoldMode.prototype);
284  
285  });

287  define("ace/mode/json",["require","exports","module","ace/lib/oop","ace/mode/text","ace/mode/json_highlight_rules","ace/mode/matching_brace_outdent","ace/mode/behaviour/cstyle","ace/mode/folding/cstyle","ace/worker/worker_client"], function(require, exports, module) {
288  "use strict";
289  
290  var oop = require("../lib/oop");
291  var TextMode = require("./text").Mode;
292  var HighlightRules = require("./json_highlight_rules").JsonHighlightRules;
293  var MatchingBraceOutdent = require("./matching_brace_outdent").MatchingBraceOutdent;
294  var CstyleBehaviour = require("./behaviour/cstyle").CstyleBehaviour;
295  var CStyleFoldMode = require("./folding/cstyle").FoldMode;
296  var WorkerClient = require("../worker/worker_client").WorkerClient;
297  
298  var Mode = function() {
299      this.HighlightRules = HighlightRules;
300      this.$outdent = new MatchingBraceOutdent();
301      this.$behaviour = new CstyleBehaviour();
302      this.foldingRules = new CStyleFoldMode();
303  };
304  oop.inherits(Mode, TextMode);
305  
306  (function() {
307  
308      this.getNextLineIndent = function(state, line, tab) {
309          var indent = this.$getIndent(line);
310  
311          if (state == "start") {
312              var match = line.match(/^.*[\{\(\[]\s*$/);
313              if (match) {
314                  indent += tab;
315              }
316          }
317  
318          return indent;
319      };
320  
321      this.checkOutdent = function(state, line, input) {
322          return this.$outdent.checkOutdent(line, input);
323      };
324  
325      this.autoOutdent = function(state, doc, row) {
326          this.$outdent.autoOutdent(doc, row);
327      };
328  
329      this.createWorker = function(session) {
330          var worker = new WorkerClient(["ace"], "ace/mode/json_worker", "JsonWorker");
331          worker.attachToDocument(session.getDocument());
332  
333          worker.on("annotate", function(e) {
334              session.setAnnotations(e.data);
335          });
336  
337          worker.on("terminate", function() {
338              session.clearAnnotations();
339          });
340  
341          return worker;
342      };
343  
344  
345      this.$id = "ace/mode/json";
346  }).call(Mode.prototype);
347  
348  exports.Mode = Mode;
