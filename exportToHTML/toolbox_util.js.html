23   function addToolboxIcons(workspace) {
24     addToolboxIconsForChildren(workspace.toolbox_.tree_.getChildren());
25   }
26   
27   function addToolboxIconsForChildren(children) {
28     for (var i = 0, child; child = children[i]; i++) {
29       if (child.getChildCount() > 0) {
30         addToolboxIconsForChildren(child.getChildren());
31       } else {
32         var iconClass = getIconClass(child.getText());
33         if (iconClass) {
34           child.setIconClass('toolbox-node-icon ' + iconClass);
35         }
36       }
37     }
38   }
