28   'use strict';
29   
30   goog.provide('Blockly.FtcJava.variables');
31   
32   goog.require('Blockly.FtcJava');
33   
34   
35   Blockly.FtcJava['variables_get'] = function(block) {
36     // Variable getter.
37     var code = Blockly.FtcJava.variableDB_.getName(block.getFieldValue('VAR'),
38         Blockly.Variables.NAME_TYPE);
39     return [code, Blockly.FtcJava.ORDER_ATOMIC];
40   };
41   
42   Blockly.FtcJava['variables_set'] = function(block) {
43     // Variable setter.
44     var argument0 = Blockly.FtcJava.valueToCode(block, 'VALUE',
45         Blockly.FtcJava.ORDER_ASSIGNMENT) || '0';
46     var varName = Blockly.FtcJava.variableDB_.getName(
47         block.getFieldValue('VAR'), Blockly.Variables.NAME_TYPE);
48     return varName + ' = ' + argument0 + ';\n';
49   };
