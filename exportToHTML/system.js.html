28   Blockly.Blocks['system_nanoTime'] = {
29     init: function() {
30       this.setOutput(true, 'Number');
31       this.appendDummyInput()
32           .appendField('call')
33           .appendField(createNonEditableField('System'))
34           .appendField('.')
35           .appendField(createNonEditableField('nanoTime'));
36       this.setColour(functionColor);
37       this.setTooltip('Returns the current value of the running Java Virtual Machine\'s ' +
38           'high-resolution time source, in nanoseconds.');
39       this.getFtcJavaOutputType = function() {
40         return 'long';
41       };
42     }
43   };
44   
45   Blockly.JavaScript['system_nanoTime'] = function(block) {
46     var code = systemIdentifierForJavaScript + '.nanoTime()';
47     return [code, Blockly.JavaScript.ORDER_FUNCTION_CALL];
48   };
49   
50   Blockly.FtcJava['system_nanoTime'] = function(block) {
51     var code = 'System.nanoTime()';
52     return [code, Blockly.FtcJava.ORDER_FUNCTION_CALL];
53   };
