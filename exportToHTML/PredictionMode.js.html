29   //
30   // The SLL(*) prediction mode. This prediction mode ignores the current
31   // parser context when making predictions. This is the fastest prediction
32   // mode, and provides correct results for many grammars. This prediction
33   // mode is more powerful than the prediction mode provided by ANTLR 3, but
34   // may result in syntax errors for grammar and input combinations which are
35   // not SLL.
36   //
37   // <p>
38   // When using this prediction mode, the parser will either return a correct
39   // parse tree (i.e. the same parse tree that would be returned with the
40   // {@link //LL} prediction mode), or it will report a syntax error. If a
41   // syntax error is encountered when using the {@link //SLL} prediction mode,
42   // it may be due to either an actual syntax error in the input or indicate
43   // that the particular combination of grammar and input requires the more
44   // powerful {@link //LL} prediction abilities to complete successfully.</p>
45   //
46   // <p>
47   // This prediction mode does not provide any guarantees for prediction
48   // behavior for syntactically-incorrect inputs.</p>
49   //
50   PredictionMode.SLL = 0;
51   //
52   // The LL(*) prediction mode. This prediction mode allows the current parser
53   // context to be used for resolving SLL conflicts that occur during
54   // prediction. This is the fastest prediction mode that guarantees correct
55   // parse results for all combinations of grammars with syntactically correct
56   // inputs.
57   //
58   // <p>
59   // When using this prediction mode, the parser will make correct decisions
60   // for all syntactically-correct grammar and input combinations. However, in
61   // cases where the grammar is truly ambiguous this prediction mode might not
62   // report a precise answer for <em>exactly which</em> alternatives are
63   // ambiguous.</p>
64   //
65   // <p>
66   // This prediction mode does not provide any guarantees for prediction
67   // behavior for syntactically-incorrect inputs.</p>
68   //
69   PredictionMode.LL = 1;
70   //
71   // The LL(*) prediction mode with exact ambiguity detection. In addition to
72   // the correctness guarantees provided by the {@link //LL} prediction mode,
73   // this prediction mode instructs the prediction algorithm to determine the
74   // complete and exact set of ambiguous alternatives for every ambiguous
75   // decision encountered while parsing.
76   //
77   // <p>
78   // This prediction mode may be used for diagnosing ambiguities during
79   // grammar development. Due to the performance overhead of calculating sets
80   // of ambiguous alternatives, this prediction mode should be avoided when
81   // the exact results are not necessary.</p>
82   //
83   // <p>
84   // This prediction mode does not provide any guarantees for prediction
85   // behavior for syntactically-incorrect inputs.</p>
86   //
87   PredictionMode.LL_EXACT_AMBIG_DETECTION = 2;

90   //
91   // Computes the SLL prediction termination condition.
92   //
93   // <p>
94   // This method computes the SLL prediction termination condition for both of
95   // the following cases.</p>
96   //
97   // <ul>
98   // <li>The usual SLL+LL fallback upon SLL conflict</li>
99   // <li>Pure SLL without LL fallback</li>
100  // </ul>
101  //
102  // <p><strong>COMBINED SLL+LL PARSING</strong></p>
103  //
104  // <p>When LL-fallback is enabled upon SLL conflict, correct predictions are
105  // ensured regardless of how the termination condition is computed by this
106  // method. Due to the substantially higher cost of LL prediction, the
107  // prediction should only fall back to LL when the additional lookahead
108  // cannot lead to a unique SLL prediction.</p>
109  //
110  // <p>Assuming combined SLL+LL parsing, an SLL configuration set with only
111  // conflicting subsets should fall back to full LL, even if the
112  // configuration sets don't resolve to the same alternative (e.g.
113  // {@code {1,2}} and {@code {3,4}}. If there is at least one non-conflicting
114  // configuration, SLL could continue with the hopes that more lookahead will
115  // resolve via one of those non-conflicting configurations.</p>
116  //
117  // <p>Here's the prediction termination rule them: SLL (for SLL+LL parsing)
118  // stops when it sees only conflicting configuration subsets. In contrast,
119  // full LL keeps going when there is uncertainty.</p>
120  //
121  // <p><strong>HEURISTIC</strong></p>
122  //
123  // <p>As a heuristic, we stop prediction when we see any conflicting subset
124  // unless we see a state that only has one alternative associated with it.
125  // The single-alt-state thing lets prediction continue upon rules like
126  // (otherwise, it would admit defeat too soon):</p>
127  //
128  // <p>{@code [12|1|[], 6|2|[], 12|2|[]]. s : (ID | ID ID?) ';' ;}</p>
129  //
130  // <p>When the ATN simulation reaches the state before {@code ';'}, it has a
131  // DFA state that looks like: {@code [12|1|[], 6|2|[], 12|2|[]]}. Naturally
132  // {@code 12|1|[]} and {@code 12|2|[]} conflict, but we cannot stop
133  // processing this node because alternative to has another way to continue,
134  // via {@code [6|2|[]]}.</p>
135  //
136  // <p>It also let's us continue for this rule:</p>
137  //
138  // <p>{@code [1|1|[], 1|2|[], 8|3|[]] a : A | A | A B ;}</p>
139  //
140  // <p>After matching input A, we reach the stop state for rule A, state 1.
141  // State 8 is the state right before B. Clearly alternatives 1 and 2
142  // conflict and no amount of further lookahead will separate the two.
143  // However, alternative 3 will be able to continue and so we do not stop
144  // working on this state. In the previous example, we're concerned with
145  // states associated with the conflicting alternatives. Here alt 3 is not
146  // associated with the conflicting configs, but since we can continue
147  // looking for input reasonably, don't declare the state done.</p>
148  //
149  // <p><strong>PURE SLL PARSING</strong></p>
150  //
151  // <p>To handle pure SLL parsing, all we have to do is make sure that we
152  // combine stack contexts for configurations that differ only by semantic
153  // predicate. From there, we can do the usual SLL termination heuristic.</p>
154  //
155  // <p><strong>PREDICATES IN SLL+LL PARSING</strong></p>
156  //
157  // <p>SLL decisions don't evaluate predicates until after they reach DFA stop
158  // states because they need to create the DFA cache that works in all
159  // semantic situations. In contrast, full LL evaluates predicates collected
160  // during start state computation so it can ignore predicates thereafter.
161  // This means that SLL termination detection can totally ignore semantic
162  // predicates.</p>
163  //
164  // <p>Implementation-wise, {@link ATNConfigSet} combines stack contexts but not
165  // semantic predicate contexts so we might see two configurations like the
166  // following.</p>
167  //
168  // <p>{@code (s, 1, x, {}), (s, 1, x', {p})}</p>
169  //
170  // <p>Before testing these configurations against others, we have to merge
171  // {@code x} and {@code x'} (without modifying the existing configurations).
172  // For example, we test {@code (x+x')==x''} when looking for conflicts in
173  // the following configurations.</p>
174  //
175  // <p>{@code (s, 1, x, {}), (s, 1, x', {p}), (s, 2, x'', {})}</p>
176  //
177  // <p>If the configuration set has predicates (as indicated by
178  // {@link ATNConfigSet//hasSemanticContext}), this algorithm makes a copy of
179  // the configurations to strip out all of the predicates so that a standard
180  // {@link ATNConfigSet} will merge everything ignoring predicates.</p>
181  //
182  PredictionMode.hasSLLConflictTerminatingPrediction = function( mode, configs) {
183      // Configs in rule stop states indicate reaching the end of the decision
184      // rule (local context) or end of start rule (full context). If all
185      // configs meet this condition, then none of the configurations is able
186      // to match additional input so we terminate prediction.
187      //
188      if (PredictionMode.allConfigsInRuleStopStates(configs)) {
189          return true;
190      }
191      // pure SLL mode parsing
192      if (mode === PredictionMode.SLL) {
193          // Don't bother with combining configs from different semantic
194          // contexts if we can fail over to full LL; costs more time
195          // since we'll often fail over anyway.
196          if (configs.hasSemanticContext) {
197              // dup configs, tossing out semantic predicates
198              var dup = new ATNConfigSet();
199              for(var i=0;i<configs.items.length;i++) {
200              	var c = configs.items[i];
201                  c = new ATNConfig({semanticContext:SemanticContext.NONE}, c);
202                  dup.add(c);
203              }
204              configs = dup;
205          }
206          // now we have combined contexts for configs with dissimilar preds
207      }
208      // pure SLL or combined SLL+LL mode parsing
209      var altsets = PredictionMode.getConflictingAltSubsets(configs);
210      return PredictionMode.hasConflictingAltSet(altsets) && !PredictionMode.hasStateAssociatedWithOneAlt(configs);
211  };

213  // Checks if any configuration in {@code configs} is in a
214  // {@link RuleStopState}. Configurations meeting this condition have reached
215  // the end of the decision rule (local context) or end of start rule (full
216  // context).
217  //
218  // @param configs the configuration set to test
219  // @return {@code true} if any configuration in {@code configs} is in a
220  // {@link RuleStopState}, otherwise {@code false}
221  PredictionMode.hasConfigInRuleStopState = function(configs) {
222  	for(var i=0;i<configs.items.length;i++) {
223  		var c = configs.items[i];
224          if (c.state instanceof RuleStopState) {
225              return true;
226          }
227  	}
228      return false;
229  };

231  // Checks if all configurations in {@code configs} are in a
232  // {@link RuleStopState}. Configurations meeting this condition have reached
233  // the end of the decision rule (local context) or end of start rule (full
234  // context).
235  //
236  // @param configs the configuration set to test
237  // @return {@code true} if all configurations in {@code configs} are in a
238  // {@link RuleStopState}, otherwise {@code false}
239  PredictionMode.allConfigsInRuleStopStates = function(configs) {
240  	for(var i=0;i<configs.items.length;i++) {
241  		var c = configs.items[i];
242          if (!(c.state instanceof RuleStopState)) {
243              return false;
244          }
245  	}
246      return true;
247  };

249  //
250  // Full LL prediction termination.
251  //
252  // <p>Can we stop looking ahead during ATN simulation or is there some
253  // uncertainty as to which alternative we will ultimately pick, after
254  // consuming more input? Even if there are partial conflicts, we might know
255  // that everything is going to resolve to the same minimum alternative. That
256  // means we can stop since no more lookahead will change that fact. On the
257  // other hand, there might be multiple conflicts that resolve to different
258  // minimums. That means we need more look ahead to decide which of those
259  // alternatives we should predict.</p>
260  //
261  // <p>The basic idea is to split the set of configurations {@code C}, into
262  // conflicting subsets {@code (s, _, ctx, _)} and singleton subsets with
263  // non-conflicting configurations. Two configurations conflict if they have
264  // identical {@link ATNConfig//state} and {@link ATNConfig//context} values
265  // but different {@link ATNConfig//alt} value, e.g. {@code (s, i, ctx, _)}
266  // and {@code (s, j, ctx, _)} for {@code i!=j}.</p>
267  //
268  // <p>Reduce these configuration subsets to the set of possible alternatives.
269  // You can compute the alternative subsets in one pass as follows:</p>
270  //
271  // <p>{@code A_s,ctx = {i | (s, i, ctx, _)}} for each configuration in
272  // {@code C} holding {@code s} and {@code ctx} fixed.</p>
273  //
274  // <p>Or in pseudo-code, for each configuration {@code c} in {@code C}:</p>
275  //
276  // <pre>
277  // map[c] U= c.{@link ATNConfig//alt alt} // map hash/equals uses s and x, not
278  // alt and not pred
279  // </pre>
280  //
281  // <p>The values in {@code map} are the set of {@code A_s,ctx} sets.</p>
282  //
283  // <p>If {@code |A_s,ctx|=1} then there is no conflict associated with
284  // {@code s} and {@code ctx}.</p>
285  //
286  // <p>Reduce the subsets to singletons by choosing a minimum of each subset. If
287  // the union of these alternative subsets is a singleton, then no amount of
288  // more lookahead will help us. We will always pick that alternative. If,
289  // however, there is more than one alternative, then we are uncertain which
290  // alternative to predict and must continue looking for resolution. We may
291  // or may not discover an ambiguity in the future, even if there are no
292  // conflicting subsets this round.</p>
293  //
294  // <p>The biggest sin is to terminate early because it means we've made a
295  // decision but were uncertain as to the eventual outcome. We haven't used
296  // enough lookahead. On the other hand, announcing a conflict too late is no
297  // big deal; you will still have the conflict. It's just inefficient. It
298  // might even look until the end of file.</p>
299  //
300  // <p>No special consideration for semantic predicates is required because
301  // predicates are evaluated on-the-fly for full LL prediction, ensuring that
302  // no configuration contains a semantic context during the termination
303  // check.</p>
304  //
305  // <p><strong>CONFLICTING CONFIGS</strong></p>
306  //
307  // <p>Two configurations {@code (s, i, x)} and {@code (s, j, x')}, conflict
308  // when {@code i!=j} but {@code x=x'}. Because we merge all
309  // {@code (s, i, _)} configurations together, that means that there are at
310  // most {@code n} configurations associated with state {@code s} for
311  // {@code n} possible alternatives in the decision. The merged stacks
312  // complicate the comparison of configuration contexts {@code x} and
313  // {@code x'}. Sam checks to see if one is a subset of the other by calling
314  // merge and checking to see if the merged result is either {@code x} or
315  // {@code x'}. If the {@code x} associated with lowest alternative {@code i}
316  // is the superset, then {@code i} is the only possible prediction since the
317  // others resolve to {@code min(i)} as well. However, if {@code x} is
318  // associated with {@code j>i} then at least one stack configuration for
319  // {@code j} is not in conflict with alternative {@code i}. The algorithm
320  // should keep going, looking for more lookahead due to the uncertainty.</p>
321  //
322  // <p>For simplicity, I'm doing a equality check between {@code x} and
323  // {@code x'} that lets the algorithm continue to consume lookahead longer
324  // than necessary. The reason I like the equality is of course the
325  // simplicity but also because that is the test you need to detect the
326  // alternatives that are actually in conflict.</p>
327  //
328  // <p><strong>CONTINUE/STOP RULE</strong></p>
329  //
330  // <p>Continue if union of resolved alternative sets from non-conflicting and
331  // conflicting alternative subsets has more than one alternative. We are
332  // uncertain about which alternative to predict.</p>
333  //
334  // <p>The complete set of alternatives, {@code [i for (_,i,_)]}, tells us which
335  // alternatives are still in the running for the amount of input we've
336  // consumed at this point. The conflicting sets let us to strip away
337  // configurations that won't lead to more states because we resolve
338  // conflicts to the configuration with a minimum alternate for the
339  // conflicting set.</p>
340  //
341  // <p><strong>CASES</strong></p>
342  //
343  // <ul>
344  //
345  // <li>no conflicts and more than 1 alternative in set =&gt; continue</li>
346  //
347  // <li> {@code (s, 1, x)}, {@code (s, 2, x)}, {@code (s, 3, z)},
348  // {@code (s', 1, y)}, {@code (s', 2, y)} yields non-conflicting set
349  // {@code {3}} U conflicting sets {@code min({1,2})} U {@code min({1,2})} =
350  // {@code {1,3}} =&gt; continue
351  // </li>
352  //
353  // <li>{@code (s, 1, x)}, {@code (s, 2, x)}, {@code (s', 1, y)},
354  // {@code (s', 2, y)}, {@code (s'', 1, z)} yields non-conflicting set
355  // {@code {1}} U conflicting sets {@code min({1,2})} U {@code min({1,2})} =
356  // {@code {1}} =&gt; stop and predict 1</li>
357  //
358  // <li>{@code (s, 1, x)}, {@code (s, 2, x)}, {@code (s', 1, y)},
359  // {@code (s', 2, y)} yields conflicting, reduced sets {@code {1}} U
360  // {@code {1}} = {@code {1}} =&gt; stop and predict 1, can announce
361  // ambiguity {@code {1,2}}</li>
362  //
363  // <li>{@code (s, 1, x)}, {@code (s, 2, x)}, {@code (s', 2, y)},
364  // {@code (s', 3, y)} yields conflicting, reduced sets {@code {1}} U
365  // {@code {2}} = {@code {1,2}} =&gt; continue</li>
366  //
367  // <li>{@code (s, 1, x)}, {@code (s, 2, x)}, {@code (s', 3, y)},
368  // {@code (s', 4, y)} yields conflicting, reduced sets {@code {1}} U
369  // {@code {3}} = {@code {1,3}} =&gt; continue</li>
370  //
371  // </ul>
372  //
373  // <p><strong>EXACT AMBIGUITY DETECTION</strong></p>
374  //
375  // <p>If all states report the same conflicting set of alternatives, then we
376  // know we have the exact ambiguity set.</p>
377  //
378  // <p><code>|A_<em>i</em>|&gt;1</code> and
379  // <code>A_<em>i</em> = A_<em>j</em></code> for all <em>i</em>, <em>j</em>.</p>
380  //
381  // <p>In other words, we continue examining lookahead until all {@code A_i}
382  // have more than one alternative and all {@code A_i} are the same. If
383  // {@code A={{1,2}, {1,3}}}, then regular LL prediction would terminate
384  // because the resolved set is {@code {1}}. To determine what the real
385  // ambiguity is, we have to know whether the ambiguity is between one and
386  // two or one and three so we keep going. We can only stop prediction when
387  // we need exact ambiguity detection when the sets look like
388  // {@code A={{1,2}}} or {@code {{1,2},{1,2}}}, etc...</p>
389  //
390  PredictionMode.resolvesToJustOneViableAlt = function(altsets) {
391      return PredictionMode.getSingleViableAlt(altsets);
392  };

394  //
395  // Determines if every alternative subset in {@code altsets} contains more
396  // than one alternative.
397  //
398  // @param altsets a collection of alternative subsets
399  // @return {@code true} if every {@link BitSet} in {@code altsets} has
400  // {@link BitSet//cardinality cardinality} &gt; 1, otherwise {@code false}
401  //
402  PredictionMode.allSubsetsConflict = function(altsets) {
403      return ! PredictionMode.hasNonConflictingAltSet(altsets);
404  };
405  //
406  // Determines if any single alternative subset in {@code altsets} contains
407  // exactly one alternative.
408  //
409  // @param altsets a collection of alternative subsets
410  // @return {@code true} if {@code altsets} contains a {@link BitSet} with
411  // {@link BitSet//cardinality cardinality} 1, otherwise {@code false}
412  //
413  PredictionMode.hasNonConflictingAltSet = function(altsets) {
414  	for(var i=0;i<altsets.length;i++) {
415  		var alts = altsets[i];
416          if (alts.length===1) {
417              return true;
418          }
419  	}
420      return false;
421  };

423  //
424  // Determines if any single alternative subset in {@code altsets} contains
425  // more than one alternative.
426  //
427  // @param altsets a collection of alternative subsets
428  // @return {@code true} if {@code altsets} contains a {@link BitSet} with
429  // {@link BitSet//cardinality cardinality} &gt; 1, otherwise {@code false}
430  //
431  PredictionMode.hasConflictingAltSet = function(altsets) {
432  	for(var i=0;i<altsets.length;i++) {
433  		var alts = altsets[i];
434          if (alts.length>1) {
435              return true;
436          }
437  	}
438      return false;
439  };

441  //
442  // Determines if every alternative subset in {@code altsets} is equivalent.
443  //
444  // @param altsets a collection of alternative subsets
445  // @return {@code true} if every member of {@code altsets} is equal to the
446  // others, otherwise {@code false}
447  //
448  PredictionMode.allSubsetsEqual = function(altsets) {
449      var first = null;
450  	for(var i=0;i<altsets.length;i++) {
451  		var alts = altsets[i];
452          if (first === null) {
453              first = alts;
454          } else if (alts!==first) {
455              return false;
456          }
457  	}
458      return true;
459  };

461  //
462  // Returns the unique alternative predicted by all alternative subsets in
463  // {@code altsets}. If no such alternative exists, this method returns
464  // {@link ATN//INVALID_ALT_NUMBER}.
465  //
466  // @param altsets a collection of alternative subsets
467  //
468  PredictionMode.getUniqueAlt = function(altsets) {
469      var all = PredictionMode.getAlts(altsets);
470      if (all.length===1) {
471          return all.minValue();
472      } else {
473          return ATN.INVALID_ALT_NUMBER;
474      }
475  };
476  
477  // Gets the complete set of represented alternatives for a collection of
478  // alternative subsets. This method returns the union of each {@link BitSet}
479  // in {@code altsets}.
480  //
481  // @param altsets a collection of alternative subsets
482  // @return the set of represented alternatives in {@code altsets}
483  //
484  PredictionMode.getAlts = function(altsets) {
485      var all = new BitSet();
486      altsets.map( function(alts) { all.or(alts); });
487      return all;
488  };

490  //
491  // This function gets the conflicting alt subsets from a configuration set.
492  // For each configuration {@code c} in {@code configs}:
493  //
494  // <pre>
495  // map[c] U= c.{@link ATNConfig//alt alt} // map hash/equals uses s and x, not
496  // alt and not pred
497  // </pre>
498  
499  PredictionMode.getConflictingAltSubsets = function(configs) {
500      var configToAlts = new Map();
501      configToAlts.hashFunction = function(cfg) { hashStuff(cfg.state.stateNumber, cfg.context); };
502      configToAlts.equalsFunction = function(c1, c2) { return c1.state.stateNumber==c2.state.stateNumber && c1.context.equals(c2.context);}
503      configs.items.map(function(cfg) {
504          var alts = configToAlts.get(cfg);
505          if (alts === null) {
506              alts = new BitSet();
507              configToAlts.put(cfg, alts);
508          }
509          alts.add(cfg.alt);
510  	});
511      return configToAlts.getValues();
512  };
513  
514  //
515  // Get a map from state to alt subset from a configuration set. For each
516  // configuration {@code c} in {@code configs}:
517  //
518  // <pre>
519  // map[c.{@link ATNConfig//state state}] U= c.{@link ATNConfig//alt alt}
520  // </pre>
521  //
522  PredictionMode.getStateToAltMap = function(configs) {
523      var m = new AltDict();
524      configs.items.map(function(c) {
525          var alts = m.get(c.state);
526          if (alts === null) {
527              alts = new BitSet();
528              m.put(c.state, alts);
529          }
530          alts.add(c.alt);
531      });
532      return m;
533  };
534  
535  PredictionMode.hasStateAssociatedWithOneAlt = function(configs) {
536      var values = PredictionMode.getStateToAltMap(configs).values();
537      for(var i=0;i<values.length;i++) {
538          if (values[i].length===1) {
539              return true;
540          }
541      }
542      return false;
543  };
544  
545  PredictionMode.getSingleViableAlt = function(altsets) {
546      var result = null;
547  	for(var i=0;i<altsets.length;i++) {
548  		var alts = altsets[i];
549          var minAlt = alts.minValue();
550          if(result===null) {
551              result = minAlt;
552          } else if(result!==minAlt) { // more than 1 viable alt
553              return ATN.INVALID_ALT_NUMBER;
554          }
555  	}
556      return result;
557  };
