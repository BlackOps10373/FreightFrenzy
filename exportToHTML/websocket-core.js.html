34   'use strict';
35   
36   // This file should ONLY be loaded from frame.html, and is not used directly by webpages.
37   // iframes that need WebSocket support should load websocket-iframe.js.
38   
39   // iframes can create new messages by calling new WebSocketMessage()
40   
41   // we define WEBSOCKET_CORE as a var, because consts cannot be accessed by iframes
42   var WEBSOCKET_CORE = function () {
43       // If WEBSOCKET_CORE is already defined, we don't want to overwrite it.
44       if (WEBSOCKET_CORE) return WEBSOCKET_CORE;
45   
46       if (window != window.top) {
47           throw new Error('websocket-core.js must only be included in the main window, not an iframe');
48       }
49   
50       var globalState = {
51           webSocket: null, // If this is null, then the WebSocket is CLOSED. If this is not null, it may be CONNECTING, OPEN, or CLOSING.
52           connectionListeners: new Set(),
53           disconnectionListeners: new Set(),
54           namespaceMap: new Map(), // All namespaces in this map are subscribed to by at least one instance of WebSocketManager
55           logMessages: false, // Log all incoming and outgoing messages by running WEBSOCKET_CORE.enableLogging('messages') from the parent console (not an iframe)
56           logDebug: false // Enable debug logging by running WEBSOCKET_CORE.enableLogging('debug') from the parent console (not an iframe)
57       };
58   
59       const SYSTEM_NAMESPACE = 'system';
60       const SUBSCRIBE_TO_NAMESPACE_MESSAGE_TYPE = 'subscribeToNamespace';
61       const UNSUBSCRIBE_FROM_NAMESPACE_MESSAGE_TYPE = 'unsubscribeFromNamespace';
62       const REQUEST_CURRENT_TIME_MESSAGE_TYPE = "requestCurrentTime";
63       const NOTIFY_CURRENT_TIME_MESSAGE_TYPE = "notifyCurrentTime";
64   
65       function WebSocketMessage(namespace, type, payload) {
66           this.namespace = namespace;
67           this.type = type;
68           if (payload && typeof payload !== 'string') {
69               throw new Error('If a payload is provided, it must be a string')
70           }
71           this.payload = payload ? payload : "";
72       }
73   
74   
75       //----------------------------------------------------------------------------------------------
76       // WebSocketManager constructor
77       //----------------------------------------------------------------------------------------------
78   
79       // There is typically one instance per iframe, created and used through websocket-iframe.js
80       // There should be no reason that instances of WebSocketManager have to be tied to iframes.
81       // You should be able to create a new instance for any self-contained piece of UI, not just an iframe.
82   
83       // Rules for manually-created WebSocketManager instances:
84       //      1. Be careful. This use-case was considered, but not tested. Contact Noah for assistance.
85       //      2. Call the finish() method when the UI section has been removed from the DOM.
86       //      3. Make SURE that you don't register any of the same handler instances on two different
87       //         instances of WebSocketManager. Doing this will break things when finish() is called
88       //         on one of the instances.
89       function WebSocketManager() {
90   
91           //----------------------------------------------------------------------------------------------
92           // Private state variables
93           //----------------------------------------------------------------------------------------------
94           var localNamespaceMap = new Map(); // All namespaces in this map are subscribed to by this WebSocketManager instance
95           var connectionListenersToDispose = new Set();
96           var disconnectionListenersToDispose = new Set();
97   
98           /** 
99            * Send an instance of WebSocketMessage. 
100           * 
101           * Will throw an Error if the WebSocket is not open, the namespace has not been subscribed to, 
102           * or the message is addressed to the system namespace. 
103           * 
104           * params: 
105           *      message: an instance of WebSocketMessage to be sent 
106           */
107          this.sendMessage = function (message) {
108              if (!localNamespaceMap.has(message.namespace)) {
109                  throw new Error('Subscribe to ' + message.namespace + ' before sending a message to it');
110              }
111  
112              if (message.namespace === SYSTEM_NAMESPACE) {
113                  throw new Error('You cannot send messages to the system namespace');
114              }
115  
116              internalSendMessage(message);
117          };
118  
119          /** 
120           * Register a connection listener and a disconnection listener. 
121           * 
122           * The connection listener will be called (with no parameters) to inform you that the 
123           * WebSocket connection is established and it is safe to start sending messages. 
124           * 
125           * The disconnection listener will be called (with no parameters) to notify you that the 
126           * WebSocket has been disconnected and that you should stop sending messages. You should not 
127           * assume that this will be called in the event that a connection is not established in the 
128           * first place, so assume that you're disconnected until the connection listener has been 
129           * called. 
130           * 
131           * The listeners will be unregistered automatically when the iframe is unloaded or the 
132           * finish() method is called. 
133           * 
134           * Be careful with what you call from the connection listener. If you're sending messages, be 
135           * sure that it makes sense to send them every time the connection is re-established. 
136           * 
137           * This method is idempotent. 
138           * 
139           * params: 
140           *      connectionListener:     A function that takes no parameters 
141           *      disconnectionListener:  A function that takes no parameters 
142           */
143          this.registerConnectionStateListeners = function (connectionListener, disconnectionListener) {
144              if (isWebSocketConnected()) {
145                  connectionListener();
146              }
147              connectionListenersToDispose.add(connectionListener);
148              disconnectionListenersToDispose.add(disconnectionListener);
149              globalState.connectionListeners.add(connectionListener);
150              globalState.disconnectionListeners.add(disconnectionListener);
151          };
152  
153          /** 
154           * Subscribe to a namespace so that we can send and receive messages within it. 
155           * 
156           * This method is idempotent. 
157           * 
158           * params: 
159           *      namespace: A string specify the namespace to subscribe to 
160           */
161          this.subscribeToNamespace = function (namespace) {
162              if (localNamespaceMap.has(namespace)) return; // If the namespace is in the local map, we've already subscribed.
163  
164              localNamespaceMap.set(namespace, new Namespace());
165  
166              if (globalState.namespaceMap.has(namespace)) {
167                  var globalNamespaceObject = globalState.namespaceMap.get(namespace);
168                  globalNamespaceObject.subscriberCount++;
169              } else {
170                  globalState.namespaceMap.set(namespace, new Namespace());
171                  if (isWebSocketConnected()) { // In the alternative case (we're not connected), a subscription request will be sent in onopen instead of here
172                      var subscriptionRequest = new WebSocketMessage(SYSTEM_NAMESPACE, SUBSCRIBE_TO_NAMESPACE_MESSAGE_TYPE, namespace);
173                      internalSendMessage(subscriptionRequest);
174                  }
175              }
176          };
177  
178          /** 
179           * Register a handler for all messages received from a particular namespace. 
180           * 
181           * 
182           * The handler will be unregistered automatically when the iframe is unloaded or the 
183           * finish() method is called. 
184           * 
185           * This method is idempotent. 
186           * 
187           * params: 
188           *      namespace:  A string specifying the namespace to register a handler for 
189           *      handler:    A function accepting one parameter, which will be called for every 
190           *                  WebSocketMessage that is received from the specified namespace 
191           */
192          this.registerNamespaceHandler = function (namespace, handler) {
193              if (!localNamespaceMap.has(namespace)) {
194                  throw new Error('Subscribe to ' + namespace + ' before registering a handler for it');
195              }
196              localNamespaceMap.get(namespace).handlers.add(handler);
197              globalState.namespaceMap.get(namespace).handlers.add(handler);
198          }
199  
200          /** 
201           * Register a handler for all messages from a specific namespace with a specific type. 
202           * 
203           * The handler will be unregistered automatically when the iframe is unloaded or the 
204           * finish() method is called. 
205           * 
206           * This method is idempotent. 
207           * 
208           * params: 
209           *      namespace:  A string specifying the namespace to process messages from 
210           *      type:       A string specifying the message type to be handled 
211           *      handler:    A function accepting one parameter, which will be called for every 
212           *                  WebSocketMessage that matches the specified namespace and type 
213           */
214          this.registerTypeHandler = function (namespace, type, handler) {
215              if (!localNamespaceMap.has(namespace)) {
216                  throw new Error('Subscribe to ' + namespace + ' before registering a handler for one of its types');
217              }
218              var localNamespaceObject = localNamespaceMap.get(namespace);
219              var globalNamespaceObject = globalState.namespaceMap.get(namespace);
220  
221              // Make sure that both the local and global `typeHandlersMap`s have an entry corresponding to our type
222              if (!localNamespaceObject.typeHandlersMap.has(type)) {
223                  localNamespaceObject.typeHandlersMap.set(type, new Set());
224                  if (!globalNamespaceObject.typeHandlersMap.has(type)) {
225                      globalNamespaceObject.typeHandlersMap.set(type, new Set());
226                  }
227              }
228  
229              localNamespaceObject.typeHandlersMap.get(type).add(handler);
230              globalNamespaceObject.typeHandlersMap.get(type).add(handler);
231          }
232  
233          // We're done with this instance of WebSocketManager, so "subtract" the local state from the global state
234          this.finish = function () {
235              localNamespaceMap.forEach(function (localNamespaceObject, namespaceName, map) {
236                  let globalNamespaceObject = globalState.namespaceMap.get(namespaceName);
237  
238                  localNamespaceObject.handlers.forEach( function (handler){
239                      globalNamespaceObject.handlers.delete(handler);
240                  });
241  
242                  localNamespaceObject.typeHandlersMap.forEach( function (handlers, type) {
243                      handlers.forEach( function (handler) {
244                          globalNamespaceObject.typeHandlersMap.get(type).delete(handler);
245                      });
246                  });
247  
248                  if (--globalNamespaceObject.subscriberCount < 1) {
249                      internalSendMessage(new WebSocketMessage(SYSTEM_NAMESPACE, UNSUBSCRIBE_FROM_NAMESPACE_MESSAGE_TYPE, namespaceName));
250                      globalState.namespaceMap.delete(namespaceName);
251                  }
252              });
253  
254              connectionListenersToDispose.forEach( function (connectionListener) {
255                  globalState.connectionListeners.delete(connectionListener);
256              });
257  
258  
259              disconnectionListenersToDispose.forEach( function (disconnectionListener) {
260                  globalState.disconnectionListeners.delete(disconnectionListener);
261              });
262  
263              localNamespaceMap = new Map();
264              connectionListenersToDispose = new Set();
265              disconnectionListenersToDispose = new Set();
266  
267              if (globalState.logDebug) {
268                  console.log("globalState:");
269                  console.log(globalState);
270              }
271          }
272      }
273  
274      //----------------------------------------------------------------------------------------------
275      // End of WebSocketManager
276      //----------------------------------------------------------------------------------------------
277  
278  
279      function Namespace() {
280          this.handlers = new Set();
281          this.typeHandlersMap = new Map();
282          this.subscriberCount = 1;
283      }
284  
285      function isWebSocketConnected() {
286          if (!globalState.webSocket) return false;
287          return globalState.webSocket.readyState === globalState.webSocket.OPEN;
288      }
289  
290      function enableLogging(val) {
291          if (val === 'messages') globalState.logMessages = true;
292          else if (val === 'debug') globalState.logDebug = true;
293      }
294  
295      function disableLogging(val) {
296          if (val === 'messages') globalState.logMessages = false;
297          else if (val === 'debug') globalState.logDebug = false;
298      }
299  
300      function onFramePingSuccess() {
301          var namespaceIterator = globalState.namespaceMap.values();
302          var result = namespaceIterator.next();
303          while (!result.done) {
304              if (result.value.subscriberCount > 0) {
305                  openWebSocket();
306                  break;
307              }
308              result = namespaceIterator.next();
309          }
310      };
311  
312      function onClose(closeEvent) {
313          console.log("WebSocket has closed");
314          globalState.webSocket = null;
315          globalState.disconnectionListeners.forEach( function (listener) {
316              listener();
317          });
318      }
319  
320      function onFramePingFailure() {
321          // If webSocket is not null, then we haven't yet realized that the WebSocket is closed.
322          if (globalState.webSocket) {
323              onClose();
324          }
325      }
326  
327      function internalSendMessage(message) {
328          if (! message instanceof WebSocketMessage) {
329              throw new Error('Only instances of WebSocketMessage can be sent');
330          }
331          if (!isWebSocketConnected()) {
332              throw new Error('Cannot send message, the WebSocket connection is not open');
333          }
334  
335          if (globalState.logMessages) {
336              console.log("Sending message:");
337              console.log(message);
338          }
339  
340          globalState.webSocket.send(JSON.stringify(message));
341          return true;
342      }
343  
344      // Attempt to open the WebSocket. This method is idempotent.
345      function openWebSocket() {
346          // If globalState.webSocket exists and is either connecting or open, exit now to preserve idempotency.
347          if (globalState.webSocket && (globalState.webSocket.readyState == 0 || globalState.webSocket.readyState == 1 )) return;
348  
349          globalState.webSocket = new WebSocket("ws://" + window.location.hostname + ':' + (parseInt(window.location.port) + 1));
350  
351          globalState.webSocket.onopen = function () {
352              globalState.namespaceMap.forEach( function (value, key, map) {
353                  var subscriptionRequest = new WebSocketMessage(SYSTEM_NAMESPACE, SUBSCRIBE_TO_NAMESPACE_MESSAGE_TYPE, key);
354                  internalSendMessage(subscriptionRequest);
355              });
356  
357              globalState.connectionListeners.forEach( function (listener) {
358                  listener();
359              });
360          };
361  
362          globalState.webSocket.onmessage = function (messageEvent) {
363              var json = messageEvent.data;
364              var rawMessage = JSON.parse(json);
365              var message = new WebSocketMessage(rawMessage.namespace, rawMessage.type, rawMessage.payload);
366  
367              if (globalState.logMessages) {
368                  console.log("Received message:");
369                  console.log(message);
370              }
371  
372              if (message.namespace === SYSTEM_NAMESPACE) {
373                  handleSystemMessage(message);
374              } else if (globalState.namespaceMap.has(message.namespace)) {
375                  let namespaceObject = globalState.namespaceMap.get(message.namespace);
376                  namespaceObject.handlers.forEach( function (handler) {
377                      handler(message);
378                  });
379  
380                  if (namespaceObject.typeHandlersMap.has(message.type)) {
381                      namespaceObject.typeHandlersMap.get(message.type).forEach( function (handler){
382                          handler(message);
383                      });
384                  }
385              }
386          };
387  
388          globalState.webSocket.onclose = onClose;
389  
390          globalState.webSocket.onerror = function (errorEvent) {
391              console.error("Websocket error:", errorEvent);
392          };
393      }
394  
395      function handleSystemMessage(message) {
396          if (message.type === REQUEST_CURRENT_TIME_MESSAGE_TYPE) {
397              var payload = {
398                  timeMs: Date.now(),
399                  timezone: Intl.DateTimeFormat().resolvedOptions().timeZone
400              };
401              var timeNotificationMessage = new WebSocketMessage(SYSTEM_NAMESPACE, NOTIFY_CURRENT_TIME_MESSAGE_TYPE, JSON.stringify(payload));
402              internalSendMessage(timeNotificationMessage);
403          }
404      }
405  
406      console.log("Log all incoming and outgoing WebSocket messages by running WEBSOCKET_CORE.enableLogging('messages') from the console");
407      console.log("Enable WebSocket debug logging by running WEBSOCKET_CORE.enableLogging('debug') from the console");
408      return {
409          WebSocketMessage: WebSocketMessage,
410          WebSocketManager: WebSocketManager,
411          isWebSocketConnected: isWebSocketConnected,
412          enableLogging: enableLogging,
413          disableLogging: disableLogging,
414          openWebSocket: openWebSocket,
415          onFramePingSuccess: onFramePingSuccess,
416          onFramePingFailure: onFramePingFailure
417      }
418  }();
