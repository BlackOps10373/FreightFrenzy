15   // Represents {@code $} in local context prediction, which means wildcard.
16   // {@code//+x =//}.

18   PredictionContext.EMPTY = null;

20   // Represents {@code $} in an array in full context mode, when {@code $}
21   // doesn't mean wildcard: {@code $ + x = [$,x]}. Here,
22   // {@code $} = {@link //EMPTY_RETURN_STATE}.
23   // /
24   PredictionContext.EMPTY_RETURN_STATE = 0x7FFFFFFF;

29   // Stores the computed hash code of this {@link PredictionContext}. The hash
30   // code is computed in parts to match the following reference algorithm.
31   //
32   // <pre>
33   // private int referenceHashCode() {
34   // int hash = {@link MurmurHash//initialize MurmurHash.initialize}({@link
35   // //INITIAL_HASH});
36   //
37   // for (int i = 0; i &lt; {@link //size()}; i++) {
38   // hash = {@link MurmurHash//update MurmurHash.update}(hash, {@link //getParent
39   // getParent}(i));
40   // }
41   //
42   // for (int i = 0; i &lt; {@link //size()}; i++) {
43   // hash = {@link MurmurHash//update MurmurHash.update}(hash, {@link
44   // //getReturnState getReturnState}(i));
45   // }
46   //
47   // hash = {@link MurmurHash//finish MurmurHash.finish}(hash, 2// {@link
48   // //size()});
49   // return hash;
50   // }

77   // Used to cache {@link PredictionContext} objects. Its used for the shared
78   // context cash associated with contexts in DFA states. This cache
79   // can be used for both lexers and parsers.
80   
81   function PredictionContextCache() {
82   	this.cache = {};
83   	return this;
84   }
85   
86   // Add a context to the cache and return it. If the context already exists,
87   // return that one instead and do not add a new context to the cache.
88   // Protect shared cache from unsafe thread access.
89   //
90   PredictionContextCache.prototype.add = function(ctx) {
91   	if (ctx === PredictionContext.EMPTY) {
92   		return PredictionContext.EMPTY;
93   	}
94   	var existing = this.cache[ctx] || null;
95   	if (existing !== null) {
96   		return existing;
97   	}
98   	this.cache[ctx] = ctx;
99   	return ctx;
100  };
101  
102  PredictionContextCache.prototype.get = function(ctx) {
103  	return this.cache[ctx] || null;
104  };

107  	get : function() {
108  		return this.cache.length;
109  	}

112  function SingletonPredictionContext(parent, returnState) {
113  	var hashCode = 0;
114  	if(parent !== null) {
115  		var hash = new Hash();
116  		hash.update(parent, returnState);
117          hashCode = hash.finish();
118  	}
119  	PredictionContext.call(this, hashCode);
120  	this.parentCtx = parent;
121  	this.returnState = returnState;
122  }

127  SingletonPredictionContext.create = function(parent, returnState) {
128  	if (returnState === PredictionContext.EMPTY_RETURN_STATE && parent === null) {
129  		// someone can pass in the bits of an array ctx that mean $
130  		return PredictionContext.EMPTY;
131  	} else {
132  		return new SingletonPredictionContext(parent, returnState);
133  	}
134  };

150  SingletonPredictionContext.prototype.equals = function(other) {
151  	if (this === other) {
152  		return true;
153  	} else if (!(other instanceof SingletonPredictionContext)) {
154  		return false;
155  	} else if (this.hashCode() !== other.hashCode()) {
156  		return false; // can't be same if hash is different
157  	} else {
158  		if(this.returnState !== other.returnState)
159              return false;
160          else if(this.parentCtx==null)
161              return other.parentCtx==null
162  		else
163              return this.parentCtx.equals(other.parentCtx);
164  	}
165  };

167  SingletonPredictionContext.prototype.toString = function() {
168  	var up = this.parentCtx === null ? "" : this.parentCtx.toString();
169  	if (up.length === 0) {
170  		if (this.returnState === PredictionContext.EMPTY_RETURN_STATE) {
171  			return "$";
172  		} else {
173  			return "" + this.returnState;
174  		}
175  	} else {
176  		return "" + this.returnState + " " + up;
177  	}
178  };

210  function ArrayPredictionContext(parents, returnStates) {
211  	// Parent can be null only if full ctx mode and we make an array
212  	// from {@link //EMPTY} and non-empty. We merge {@link //EMPTY} by using
213  	// null parent and
214  	// returnState == {@link //EMPTY_RETURN_STATE}.
215  	var h = new Hash();
216  	h.update(parents, returnStates);
217  	var hashCode = h.finish();
218  	PredictionContext.call(this, hashCode);
219  	this.parents = parents;
220  	this.returnStates = returnStates;
221  	return this;
222  }

225  ArrayPredictionContext.prototype.constructor = ArrayPredictionContext;
226  
227  ArrayPredictionContext.prototype.isEmpty = function() {
228  	// since EMPTY_RETURN_STATE can only appear in the last position, we
229  	// don't need to verify that size==1
230  	return this.returnStates[0] === PredictionContext.EMPTY_RETURN_STATE;
231  };

239  ArrayPredictionContext.prototype.getParent = function(index) {
240  	return this.parents[index];

242  
243  ArrayPredictionContext.prototype.getReturnState = function(index) {
244  	return this.returnStates[index];
245  };
246  
247  ArrayPredictionContext.prototype.equals = function(other) {
248  	if (this === other) {
249  		return true;
250  	} else if (!(other instanceof ArrayPredictionContext)) {
251  		return false;
252  	} else if (this.hashCode() !== other.hashCode()) {
253  		return false; // can't be same if hash is different
254  	} else {
255  		return this.returnStates === other.returnStates &&
256  				this.parents === other.parents;
257  	}
258  };
259  
260  ArrayPredictionContext.prototype.toString = function() {
261  	if (this.isEmpty()) {
262  		return "[]";
263  	} else {
264  		var s = "[";
265  		for (var i = 0; i < this.returnStates.length; i++) {
266  			if (i > 0) {
267  				s = s + ", ";
268  			}
269  			if (this.returnStates[i] === PredictionContext.EMPTY_RETURN_STATE) {
270  				s = s + "$";
271  				continue;
272  			}
273  			s = s + this.returnStates[i];
274  			if (this.parents[i] !== null) {
275  				s = s + " " + this.parents[i];
276  			} else {
277  				s = s + "null";
278  			}
279  		}
280  		return s + "]";
281  	}
282  };

284  // Convert a {@link RuleContext} tree to a {@link PredictionContext} graph.
285  // Return {@link //EMPTY} if {@code outerContext} is empty or null.
286  // /
287  function predictionContextFromRuleContext(atn, outerContext) {
288  	if (outerContext === undefined || outerContext === null) {
289  		outerContext = RuleContext.EMPTY;
290  	}
291  	// if we are in RuleContext of start rule, s, then PredictionContext
292  	// is EMPTY. Nobody called us. (if we are empty, return empty)
293  	if (outerContext.parentCtx === null || outerContext === RuleContext.EMPTY) {
294  		return PredictionContext.EMPTY;
295  	}
296  	// If we have a parent, convert it to a PredictionContext graph
297  	var parent = predictionContextFromRuleContext(atn, outerContext.parentCtx);
298  	var state = atn.states[outerContext.invokingState];
299  	var transition = state.transitions[0];
300  	return SingletonPredictionContext.create(parent, transition.followState.stateNumber);
301  }

314  function merge(a, b, rootIsWildcard, mergeCache) {
315  	// share same graph if both same
316  	if (a === b) {
317  		return a;
318  	}
319  	if (a instanceof SingletonPredictionContext && b instanceof SingletonPredictionContext) {
320  		return mergeSingletons(a, b, rootIsWildcard, mergeCache);
321  	}
322  	// At least one of a or b is array
323  	// If one is $ and rootIsWildcard, return $ as// wildcard
324  	if (rootIsWildcard) {
325  		if (a instanceof EmptyPredictionContext) {
326  			return a;
327  		}
328  		if (b instanceof EmptyPredictionContext) {
329  			return b;
330  		}
331  	}
332  	// convert singleton so both are arrays to normalize
333  	if (a instanceof SingletonPredictionContext) {
334  		a = new ArrayPredictionContext([a.getParent()], [a.returnState]);
335  	}
336  	if (b instanceof SingletonPredictionContext) {
337  		b = new ArrayPredictionContext([b.getParent()], [b.returnState]);
338  	}
339  	return mergeArrays(a, b, rootIsWildcard, mergeCache);
340  }

342  //
343  // Merge two {@link SingletonPredictionContext} instances.
344  //
345  // <p>Stack tops equal, parents merge is same; return left graph.<br>
346  // <embed src="images/SingletonMerge_SameRootSamePar.svg"
347  // type="image/svg+xml"/></p>
348  //
349  // <p>Same stack top, parents differ; merge parents giving array node, then
350  // remainders of those graphs. A new root node is created to point to the
351  // merged parents.<br>
352  // <embed src="images/SingletonMerge_SameRootDiffPar.svg"
353  // type="image/svg+xml"/></p>
354  //
355  // <p>Different stack tops pointing to same parent. Make array node for the
356  // root where both element in the root point to the same (original)
357  // parent.<br>
358  // <embed src="images/SingletonMerge_DiffRootSamePar.svg"
359  // type="image/svg+xml"/></p>
360  //
361  // <p>Different stack tops pointing to different parents. Make array node for
362  // the root where each element points to the corresponding original
363  // parent.<br>
364  // <embed src="images/SingletonMerge_DiffRootDiffPar.svg"
365  // type="image/svg+xml"/></p>
366  //
367  // @param a the first {@link SingletonPredictionContext}
368  // @param b the second {@link SingletonPredictionContext}
369  // @param rootIsWildcard {@code true} if this is a local-context merge,
370  // otherwise false to indicate a full-context merge
371  // @param mergeCache
372  // /
373  function mergeSingletons(a, b, rootIsWildcard, mergeCache) {
374  	if (mergeCache !== null) {
375  		var previous = mergeCache.get(a, b);
376  		if (previous !== null) {
377  			return previous;
378  		}
379  		previous = mergeCache.get(b, a);
380  		if (previous !== null) {
381  			return previous;
382  		}
383  	}
384  
385  	var rootMerge = mergeRoot(a, b, rootIsWildcard);
386  	if (rootMerge !== null) {
387  		if (mergeCache !== null) {
388  			mergeCache.set(a, b, rootMerge);
389  		}
390  		return rootMerge;
391  	}
392  	if (a.returnState === b.returnState) {
393  		var parent = merge(a.parentCtx, b.parentCtx, rootIsWildcard, mergeCache);
394  		// if parent is same as existing a or b parent or reduced to a parent,
395  		// return it
396  		if (parent === a.parentCtx) {
397  			return a; // ax + bx = ax, if a=b
398  		}
399  		if (parent === b.parentCtx) {
400  			return b; // ax + bx = bx, if a=b
401  		}
402  		// else: ax + ay = a'[x,y]
403  		// merge parents x and y, giving array node with x,y then remainders
404  		// of those graphs. dup a, a' points at merged array
405  		// new joined parent so create new singleton pointing to it, a'
406  		var spc = SingletonPredictionContext.create(parent, a.returnState);
407  		if (mergeCache !== null) {
408  			mergeCache.set(a, b, spc);
409  		}
410  		return spc;
411  	} else { // a != b payloads differ
412  		// see if we can collapse parents due to $+x parents if local ctx
413  		var singleParent = null;
414  		if (a === b || (a.parentCtx !== null && a.parentCtx === b.parentCtx)) { // ax +
415  																				// bx =
416  																				// [a,b]x
417  			singleParent = a.parentCtx;
418  		}
419  		if (singleParent !== null) { // parents are same
420  			// sort payloads and use same parent
421  			var payloads = [ a.returnState, b.returnState ];
422  			if (a.returnState > b.returnState) {
423  				payloads[0] = b.returnState;
424  				payloads[1] = a.returnState;
425  			}
426  			var parents = [ singleParent, singleParent ];
427  			var apc = new ArrayPredictionContext(parents, payloads);
428  			if (mergeCache !== null) {
429  				mergeCache.set(a, b, apc);
430  			}
431  			return apc;
432  		}
433  		// parents differ and can't merge them. Just pack together
434  		// into array; can't merge.
435  		// ax + by = [ax,by]
436  		var payloads = [ a.returnState, b.returnState ];
437  		var parents = [ a.parentCtx, b.parentCtx ];
438  		if (a.returnState > b.returnState) { // sort by payload
439  			payloads[0] = b.returnState;
440  			payloads[1] = a.returnState;
441  			parents = [ b.parentCtx, a.parentCtx ];
442  		}
443  		var a_ = new ArrayPredictionContext(parents, payloads);
444  		if (mergeCache !== null) {
445  			mergeCache.set(a, b, a_);
446  		}
447  		return a_;
448  	}
449  }

451  //
452  // Handle case where at least one of {@code a} or {@code b} is
453  // {@link //EMPTY}. In the following diagrams, the symbol {@code $} is used
454  // to represent {@link //EMPTY}.
455  //
456  // <h2>Local-Context Merges</h2>
457  //
458  // <p>These local-context merge operations are used when {@code rootIsWildcard}
459  // is true.</p>
460  //
461  // <p>{@link //EMPTY} is superset of any graph; return {@link //EMPTY}.<br>
462  // <embed src="images/LocalMerge_EmptyRoot.svg" type="image/svg+xml"/></p>
463  //
464  // <p>{@link //EMPTY} and anything is {@code //EMPTY}, so merged parent is
465  // {@code //EMPTY}; return left graph.<br>
466  // <embed src="images/LocalMerge_EmptyParent.svg" type="image/svg+xml"/></p>
467  //
468  // <p>Special case of last merge if local context.<br>
469  // <embed src="images/LocalMerge_DiffRoots.svg" type="image/svg+xml"/></p>
470  //
471  // <h2>Full-Context Merges</h2>
472  //
473  // <p>These full-context merge operations are used when {@code rootIsWildcard}
474  // is false.</p>
475  //
476  // <p><embed src="images/FullMerge_EmptyRoots.svg" type="image/svg+xml"/></p>
477  //
478  // <p>Must keep all contexts; {@link //EMPTY} in array is a special value (and
479  // null parent).<br>
480  // <embed src="images/FullMerge_EmptyRoot.svg" type="image/svg+xml"/></p>
481  //
482  // <p><embed src="images/FullMerge_SameRoot.svg" type="image/svg+xml"/></p>
483  //
484  // @param a the first {@link SingletonPredictionContext}
485  // @param b the second {@link SingletonPredictionContext}
486  // @param rootIsWildcard {@code true} if this is a local-context merge,
487  // otherwise false to indicate a full-context merge
488  // /
489  function mergeRoot(a, b, rootIsWildcard) {
490  	if (rootIsWildcard) {
491  		if (a === PredictionContext.EMPTY) {
492  			return PredictionContext.EMPTY; // // + b =//
493  		}
494  		if (b === PredictionContext.EMPTY) {
495  			return PredictionContext.EMPTY; // a +// =//
496  		}
497  	} else {
498  		if (a === PredictionContext.EMPTY && b === PredictionContext.EMPTY) {
499  			return PredictionContext.EMPTY; // $ + $ = $
500  		} else if (a === PredictionContext.EMPTY) { // $ + x = [$,x]
501  			var payloads = [ b.returnState,
502  					PredictionContext.EMPTY_RETURN_STATE ];
503  			var parents = [ b.parentCtx, null ];
504  			return new ArrayPredictionContext(parents, payloads);
505  		} else if (b === PredictionContext.EMPTY) { // x + $ = [$,x] ($ is always first if present)
506  			var payloads = [ a.returnState, PredictionContext.EMPTY_RETURN_STATE ];
507  			var parents = [ a.parentCtx, null ];
508  			return new ArrayPredictionContext(parents, payloads);
509  		}
510  	}
511  	return null;
512  }

514  //
515  // Merge two {@link ArrayPredictionContext} instances.
516  //
517  // <p>Different tops, different parents.<br>
518  // <embed src="images/ArrayMerge_DiffTopDiffPar.svg" type="image/svg+xml"/></p>
519  //
520  // <p>Shared top, same parents.<br>
521  // <embed src="images/ArrayMerge_ShareTopSamePar.svg" type="image/svg+xml"/></p>
522  //
523  // <p>Shared top, different parents.<br>
524  // <embed src="images/ArrayMerge_ShareTopDiffPar.svg" type="image/svg+xml"/></p>
525  //
526  // <p>Shared top, all shared parents.<br>
527  // <embed src="images/ArrayMerge_ShareTopSharePar.svg"
528  // type="image/svg+xml"/></p>
529  //
530  // <p>Equal tops, merge parents and reduce top to
531  // {@link SingletonPredictionContext}.<br>
532  // <embed src="images/ArrayMerge_EqualTop.svg" type="image/svg+xml"/></p>
533  // /
534  function mergeArrays(a, b, rootIsWildcard, mergeCache) {
535  	if (mergeCache !== null) {
536  		var previous = mergeCache.get(a, b);
537  		if (previous !== null) {
538  			return previous;
539  		}
540  		previous = mergeCache.get(b, a);
541  		if (previous !== null) {
542  			return previous;
543  		}
544  	}
545  	// merge sorted payloads a + b => M
546  	var i = 0; // walks a
547  	var j = 0; // walks b
548  	var k = 0; // walks target M array
549  
550  	var mergedReturnStates = [];
551  	var mergedParents = [];
552  	// walk and merge to yield mergedParents, mergedReturnStates
553  	while (i < a.returnStates.length && j < b.returnStates.length) {
554  		var a_parent = a.parents[i];
555  		var b_parent = b.parents[j];
556  		if (a.returnStates[i] === b.returnStates[j]) {
557  			// same payload (stack tops are equal), must yield merged singleton
558  			var payload = a.returnStates[i];
559  			// $+$ = $
560  			var bothDollars = payload === PredictionContext.EMPTY_RETURN_STATE &&
561  					a_parent === null && b_parent === null;
562  			var ax_ax = (a_parent !== null && b_parent !== null && a_parent === b_parent); // ax+ax
563  																							// ->
564  																							// ax
565  			if (bothDollars || ax_ax) {
566  				mergedParents[k] = a_parent; // choose left
567  				mergedReturnStates[k] = payload;
568  			} else { // ax+ay -> a'[x,y]
569  				var mergedParent = merge(a_parent, b_parent, rootIsWildcard, mergeCache);
570  				mergedParents[k] = mergedParent;
571  				mergedReturnStates[k] = payload;
572  			}
573  			i += 1; // hop over left one as usual
574  			j += 1; // but also skip one in right side since we merge
575  		} else if (a.returnStates[i] < b.returnStates[j]) { // copy a[i] to M
576  			mergedParents[k] = a_parent;
577  			mergedReturnStates[k] = a.returnStates[i];
578  			i += 1;
579  		} else { // b > a, copy b[j] to M
580  			mergedParents[k] = b_parent;
581  			mergedReturnStates[k] = b.returnStates[j];
582  			j += 1;
583  		}
584  		k += 1;
585  	}
586  	// copy over any payloads remaining in either array
587  	if (i < a.returnStates.length) {
588  		for (var p = i; p < a.returnStates.length; p++) {
589  			mergedParents[k] = a.parents[p];
590  			mergedReturnStates[k] = a.returnStates[p];
591  			k += 1;
592  		}
593  	} else {
594  		for (var p = j; p < b.returnStates.length; p++) {
595  			mergedParents[k] = b.parents[p];
596  			mergedReturnStates[k] = b.returnStates[p];
597  			k += 1;
598  		}
599  	}
600  	// trim merged if we combined a few that had same stack tops
601  	if (k < mergedParents.length) { // write index < last position; trim
602  		if (k === 1) { // for just one merged element, return singleton top
603  			var a_ = SingletonPredictionContext.create(mergedParents[0],
604  					mergedReturnStates[0]);
605  			if (mergeCache !== null) {
606  				mergeCache.set(a, b, a_);
607  			}
608  			return a_;
609  		}
610  		mergedParents = mergedParents.slice(0, k);
611  		mergedReturnStates = mergedReturnStates.slice(0, k);
612  	}
613  
614  	var M = new ArrayPredictionContext(mergedParents, mergedReturnStates);
615  
616  	// if we created same array as a or b, return that instead
617  	// TODO: track whether this is possible above during merge sort for speed
618  	if (M === a) {
619  		if (mergeCache !== null) {
620  			mergeCache.set(a, b, a);
621  		}
622  		return a;
623  	}
624  	if (M === b) {
625  		if (mergeCache !== null) {
626  			mergeCache.set(a, b, b);
627  		}
628  		return b;
629  	}
630  	combineCommonParents(mergedParents);
631  
632  	if (mergeCache !== null) {
633  		mergeCache.set(a, b, M);
634  	}
635  	return M;
636  }

638  //
639  // Make pass over all <em>M</em> {@code parents}; merge any {@code equals()}
640  // ones.
641  // /
642  function combineCommonParents(parents) {
643  	var uniqueParents = {};
644  
645  	for (var p = 0; p < parents.length; p++) {
646  		var parent = parents[p];
647  		if (!(parent in uniqueParents)) {
648  			uniqueParents[parent] = parent;
649  		}
650  	}
651  	for (var q = 0; q < parents.length; q++) {
652  		parents[q] = uniqueParents[parents[q]];
653  	}
654  }

656  function getCachedPredictionContext(context, contextCache, visited) {
657  	if (context.isEmpty()) {
658  		return context;
659  	}
660  	var existing = visited[context] || null;
661  	if (existing !== null) {
662  		return existing;
663  	}
664  	existing = contextCache.get(context);
665  	if (existing !== null) {
666  		visited[context] = existing;
667  		return existing;
668  	}
669  	var changed = false;
670  	var parents = [];
671  	for (var i = 0; i < parents.length; i++) {
672  		var parent = getCachedPredictionContext(context.getParent(i), contextCache, visited);
673  		if (changed || parent !== context.getParent(i)) {
674  			if (!changed) {
675  				parents = [];
676  				for (var j = 0; j < context.length; j++) {
677  					parents[j] = context.getParent(j);
678  				}
679  				changed = true;
680  			}
681  			parents[i] = parent;
682  		}
683  	}
684  	if (!changed) {
685  		contextCache.add(context);
686  		visited[context] = context;
687  		return context;
688  	}
689  	var updated = null;
690  	if (parents.length === 0) {
691  		updated = PredictionContext.EMPTY;
692  	} else if (parents.length === 1) {
693  		updated = SingletonPredictionContext.create(parents[0], context
694  				.getReturnState(0));
695  	} else {
696  		updated = new ArrayPredictionContext(parents, context.returnStates);
697  	}
698  	contextCache.add(updated);
699  	visited[updated] = updated;
700  	visited[context] = updated;
701  
702  	return updated;
703  }

705  // ter's recursive version of Sam's getAllNodes()
706  function getAllContextNodes(context, nodes, visited) {
707  	if (nodes === null) {
708  		nodes = [];
709  		return getAllContextNodes(context, nodes, visited);
710  	} else if (visited === null) {
711  		visited = {};
712  		return getAllContextNodes(context, nodes, visited);
713  	} else {
714  		if (context === null || visited[context] !== null) {
715  			return nodes;
716  		}
717  		visited[context] = context;
718  		nodes.push(context);
719  		for (var i = 0; i < context.length; i++) {
720  			getAllContextNodes(context.getParent(i), nodes, visited);
721  		}
722  		return nodes;
723  	}
