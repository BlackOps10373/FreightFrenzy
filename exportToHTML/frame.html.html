19       function getParameterByName(name, url) {
20           if (!url) url = window.location.href;
21           name = name.replace(/[\[\]]/g, "\\$&");
22           var regex = new RegExp("[?&]" + name + "(=([^&#]*)|&|#|$)");
23           var results = regex.exec(url);
24           if (!results) return null;
25           if (!results[2]) return '';
26           return decodeURIComponent(results[2].replace(/\+/g, " "));
27       }

30       // Fires every time the iframe navigates
31       window.iframeChanged = function iframeChanged() {
32           var theIframe = document.getElementById('theIFrame');
33           var currentUri = theIframe.contentWindow.document.URL;
34           var id = "";
35           if (currentUri.indexOf("/") !== -1) {
36               id = currentUri.replace(/http(?:s|):\/\/[.\d:]+\//, '');
37           }
38   
39           if (id !== "") {
40               var newDocLocation;
41               if (document.URL.indexOf('?') === -1) {
42                   newDocLocation = document.URL;
43               } else {
44                   newDocLocation = document.URL.substring(0, document.URL.indexOf('?') - 1);
45               }
46   
47               newDocLocation += "?page=";
48               newDocLocation += id;
49               if (document.URL !== newDocLocation) {
50                   const hostUrl = window.location.protocol + '//' + window.location.host;
51                   // test if the history already knows about this state or we are coming from the host url, if so update accordingly
52                   if (getParameterByName('pop') === 'true' || window.location.origin === hostUrl || window.location.origin === hostUrl + '/') {
53                       window.history.replaceState('', theIframe.contentWindow.document.title, newDocLocation + '&pop=true');
54                   } else {
55                       window.history.pushState('', theIframe.contentWindow.document.title, newDocLocation + '&pop=true');
56                   }
57               }
58           }
59   
60           document.title = theIframe.contentWindow.document.title;
61       }

66       // The native prompt dialog is ugly and not supported by Electron, which means it doesn't work in the REV Hardware Client.
67       var showPrompt = function(message, defaultValue, callback) {
68           const promptModal = $('#promptModal');
69           promptModal.off();
70   
71           const finishShowingPrompt = function() {
72               promptModal.off();
73               promptModal.on('hide.bs.modal', function (e) {
74                   console.log("Modal hidden");
75                   if (promptCallback) {
76                       promptCallback(null);
77                       promptCallback = null;
78                   }
79               });
80   
81               console.log("Setting promptCallback");
82               promptCallback = callback;
83               document.getElementById('promptMessage').textContent = message;
84               if (!defaultValue) {
85                   defaultValue = "";
86               }
87               document.getElementById('promptResponse').value = '';
88               document.getElementById('promptResponse').placeholder = defaultValue;
89               console.log("Showing modal");
90               promptModal.modal('show').on('shown.bs.modal', function() {
91                   console.log("Focusing promptResponse");
92                   document.getElementById('promptResponse').focus();
93               });
94           }
95   
96           if (promptCallback) {
97               promptCallback(null);
98               promptCallback = null;
99               console.log("Hiding existing modal");
100              promptModal.modal('hide').on('hidden.bs.modal', finishShowingPrompt);
101          } else {
102              finishShowingPrompt();
103          }
104  
105      };

114              }

116              promptCallback = null;
117              const promptModal = $('#promptModal');
118              promptModal.off();
119              promptModal.modal('hide');

310      function getRelativeURI(loc) {
311          var there = new URI(getUrlParts(loc).href);
312          var here = new URI(location.href);
313          return there.relativeTo(here).toString();
314      }

323              port: a.port,
324              pathname: a.pathname,

326              hash: a.hash,

348      function sendPingRequest() {
349          sendPing(getRelativeURI(theIFrame.src), function (pingResponse) {
350              var hiddenClass = "hidden";
351              if (pingResponse) {
352                  ensureClass(pingFailure, hiddenClass);
353                  removeClass(theNavBarItems, hiddenClass);
354                  removeClass(theNavBarItemsRight, hiddenClass);
355                  WEBSOCKET_CORE.onFramePingSuccess();
356              } else {
357                  WEBSOCKET_CORE.onFramePingFailure();
358                  removeClass(pingFailure, hiddenClass);
359                  ensureClass(theNavBarItems, hiddenClass);
360                  ensureClass(theNavBarItemsRight, hiddenClass);
361              }
362  
363              var child = theIFrame.contentDocument;
364              if (child && child.onPingResult) {
365                  child.onPingResult(pingResponse);
366              }
367  
368          });
369      }

371      function sendPing(name, callback) {
372          var xhr = new XMLHttpRequest();
373          xhr.open('POST', URI_PING, true);
374          xhr.setRequestHeader('Content-type', 'application/x-www-form-urlencoded');
375          xhr.onreadystatechange = function () {
376              if (xhr.readyState === 4) {
377                  if (xhr.status === 200) {
378                      var pingResponse = JSON.parse(xhr.responseText);
379                      callback(pingResponse);
380                  } else {
381                      callback(null);
382                  }
383              }
384          };
385          var params = PARAM_NAME + '=' + encodeURIComponent(name);
386          xhr.send(params);
387      }
