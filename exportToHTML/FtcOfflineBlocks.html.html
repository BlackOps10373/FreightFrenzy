201    function exportJavaButtonClicked() {
202      var javaCode = generateJavaCode();
203      if (javaCode) {
204        // Download the java file.
205        var a = document.getElementById('download_link');
206        a.href = 'data:text/plain;charset=utf-8,' + encodeURIComponent(javaCode);
207        a.download = currentClassName + '.java';
208        a.target = '_blank';
209        a.click();
210      } else {
211        alert('Unable to download Java. Sorry about that!');
212      }
213    }
