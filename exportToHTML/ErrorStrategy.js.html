42   // This is the default implementation of {@link ANTLRErrorStrategy} used for
43   // error reporting and recovery in ANTLR parsers.
44   //
45   function DefaultErrorStrategy() {
46   	ErrorStrategy.call(this);
47       // Indicates whether the error strategy is currently "recovering from an
48       // error". This is used to suppress reporting multiple error messages while
49       // attempting to recover from a detected syntax error.
50       //
51       // @see //inErrorRecoveryMode
52       //
53       this.errorRecoveryMode = false;
54   
55       // The index into the input stream where the last error occurred.
56       // This is used to prevent infinite loops where an error is found
57       // but no token is consumed during recovery...another error is found,
58       // ad nauseum. This is a failsafe mechanism to guarantee that at least
59       // one token/tree node is consumed for two errors.
60       //
61       this.lastErrorIndex = -1;
62       this.lastErrorStates = null;
63       return this;
64   }

110  //
111  // {@inheritDoc}
112  //
113  // <p>The default implementation returns immediately if the handler is already
114  // in error recovery mode. Otherwise, it calls {@link //beginErrorCondition}
115  // and dispatches the reporting task based on the runtime type of {@code e}
116  // according to the following table.</p>
117  //
118  // <ul>
119  // <li>{@link NoViableAltException}: Dispatches the call to
120  // {@link //reportNoViableAlternative}</li>
121  // <li>{@link InputMismatchException}: Dispatches the call to
122  // {@link //reportInputMismatch}</li>
123  // <li>{@link FailedPredicateException}: Dispatches the call to
124  // {@link //reportFailedPredicate}</li>
125  // <li>All other types: calls {@link Parser//notifyErrorListeners} to report
126  // the exception</li>
127  // </ul>
128  //
129  DefaultErrorStrategy.prototype.reportError = function(recognizer, e) {
130     // if we've already reported an error and have not matched a token
131     // yet successfully, don't report any errors.
132      if(this.inErrorRecoveryMode(recognizer)) {
133          return; // don't report spurious errors
134      }
135      this.beginErrorCondition(recognizer);
136      if ( e instanceof NoViableAltException ) {
137          this.reportNoViableAlternative(recognizer, e);
138      } else if ( e instanceof InputMismatchException ) {
139          this.reportInputMismatch(recognizer, e);
140      } else if ( e instanceof FailedPredicateException ) {
141          this.reportFailedPredicate(recognizer, e);
142      } else {
143          console.log("unknown recognition error type: " + e.constructor.name);
144          console.log(e.stack);
145          recognizer.notifyErrorListeners(e.getOffendingToken(), e.getMessage(), e);
146      }
147  };
148  //
149  // {@inheritDoc}
150  //
151  // <p>The default implementation resynchronizes the parser by consuming tokens
152  // until we find one in the resynchronization set--loosely the set of tokens
153  // that can follow the current rule.</p>
154  //
155  DefaultErrorStrategy.prototype.recover = function(recognizer, e) {
156      if (this.lastErrorIndex===recognizer.getInputStream().index &&
157          this.lastErrorStates !== null && this.lastErrorStates.indexOf(recognizer.state)>=0) {
158  		// uh oh, another error at same token index and previously-visited
159  		// state in ATN; must be a case where LT(1) is in the recovery
160  		// token set so nothing got consumed. Consume a single token
161  		// at least to prevent an infinite loop; this is a failsafe.
162  		recognizer.consume();
163      }
164      this.lastErrorIndex = recognizer._input.index;
165      if (this.lastErrorStates === null) {
166          this.lastErrorStates = [];
167      }
168      this.lastErrorStates.push(recognizer.state);
169      var followSet = this.getErrorRecoverySet(recognizer);
170      this.consumeUntil(recognizer, followSet);
171  };

173  // The default implementation of {@link ANTLRErrorStrategy//sync} makes sure
174  // that the current lookahead symbol is consistent with what were expecting
175  // at this point in the ATN. You can call this anytime but ANTLR only
176  // generates code to check before subrules/loops and each iteration.
177  //
178  // <p>Implements Jim Idle's magic sync mechanism in closures and optional
179  // subrules. E.g.,</p>
180  //
181  // <pre>
182  // a : sync ( stuff sync )* ;
183  // sync : {consume to what can follow sync} ;
184  // </pre>
185  //
186  // At the start of a sub rule upon error, {@link //sync} performs single
187  // token deletion, if possible. If it can't do that, it bails on the current
188  // rule and uses the default error recovery, which consumes until the
189  // resynchronization set of the current rule.
190  //
191  // <p>If the sub rule is optional ({@code (...)?}, {@code (...)*}, or block
192  // with an empty alternative), then the expected set includes what follows
193  // the subrule.</p>
194  //
195  // <p>During loop iteration, it consumes until it sees a token that can start a
196  // sub rule or what follows loop. Yes, that is pretty aggressive. We opt to
197  // stay in the loop as long as possible.</p>
198  //
199  // <p><strong>ORIGINS</strong></p>
200  //
201  // <p>Previous versions of ANTLR did a poor job of their recovery within loops.
202  // A single mismatch token or missing token would force the parser to bail
203  // out of the entire rules surrounding the loop. So, for rule</p>
204  //
205  // <pre>
206  // classDef : 'class' ID '{' member* '}'
207  // </pre>
208  //
209  // input with an extra token between members would force the parser to
210  // consume until it found the next class definition rather than the next
211  // member definition of the current class.
212  //
213  // <p>This functionality cost a little bit of effort because the parser has to
214  // compare token set at the start of the loop and at each iteration. If for
215  // some reason speed is suffering for you, you can turn off this
216  // functionality by simply overriding this method as a blank { }.</p>
217  //
218  DefaultErrorStrategy.prototype.sync = function(recognizer) {
219      // If already recovering, don't try to sync
220      if (this.inErrorRecoveryMode(recognizer)) {
221          return;
222      }
223      var s = recognizer._interp.atn.states[recognizer.state];
224      var la = recognizer.getTokenStream().LA(1);
225      // try cheaper subset first; might get lucky. seems to shave a wee bit off
226      var nextTokens = recognizer.atn.nextTokens(s);
227      if (nextTokens.contains(Token.EPSILON) || nextTokens.contains(la)) {
228          return;
229      }
230      switch (s.stateType) {
231      case ATNState.BLOCK_START:
232      case ATNState.STAR_BLOCK_START:
233      case ATNState.PLUS_BLOCK_START:
234      case ATNState.STAR_LOOP_ENTRY:
235         // report error and recover if possible
236          if( this.singleTokenDeletion(recognizer) !== null) {
237              return;
238          } else {
239              throw new InputMismatchException(recognizer);
240          }
241          break;
242      case ATNState.PLUS_LOOP_BACK:
243      case ATNState.STAR_LOOP_BACK:
244          this.reportUnwantedToken(recognizer);
245          var expecting = new IntervalSet();
246          expecting.addSet(recognizer.getExpectedTokens());
247          var whatFollowsLoopIterationOrRule = expecting.addSet(this.getErrorRecoverySet(recognizer));
248          this.consumeUntil(recognizer, whatFollowsLoopIterationOrRule);
249          break;
250      default:
251          // do nothing if we can't identify the exact kind of ATN state
252      }
253  };

255  // This is called by {@link //reportError} when the exception is a
256  // {@link NoViableAltException}.
257  //
258  // @see //reportError
259  //
260  // @param recognizer the parser instance
261  // @param e the recognition exception
262  //
263  DefaultErrorStrategy.prototype.reportNoViableAlternative = function(recognizer, e) {
264      var tokens = recognizer.getTokenStream();
265      var input;
266      if(tokens !== null) {
267          if (e.startToken.type===Token.EOF) {
268              input = "<EOF>";
269          } else {
270              input = tokens.getText(new Interval(e.startToken, e.offendingToken));
271          }
272      } else {
273          input = "<unknown input>";
274      }
275      var msg = "no viable alternative at input " + this.escapeWSAndQuote(input);
276      recognizer.notifyErrorListeners(msg, e.offendingToken, e);
277  };

309  // This method is called to report a syntax error which requires the removal
310  // of a token from the input stream. At the time this method is called, the
311  // erroneous symbol is current {@code LT(1)} symbol and has not yet been
312  // removed from the input stream. When this method returns,
313  // {@code recognizer} is in error recovery mode.
314  //
315  // <p>This method is called when {@link //singleTokenDeletion} identifies
316  // single-token deletion as a viable recovery strategy for a mismatched
317  // input error.</p>
318  //
319  // <p>The default implementation simply returns if the handler is already in
320  // error recovery mode. Otherwise, it calls {@link //beginErrorCondition} to
321  // enter error recovery mode, followed by calling
322  // {@link Parser//notifyErrorListeners}.</p>
323  //
324  // @param recognizer the parser instance
325  //
326  DefaultErrorStrategy.prototype.reportUnwantedToken = function(recognizer) {
327      if (this.inErrorRecoveryMode(recognizer)) {
328          return;
329      }
330      this.beginErrorCondition(recognizer);
331      var t = recognizer.getCurrentToken();
332      var tokenName = this.getTokenErrorDisplay(t);
333      var expecting = this.getExpectedTokens(recognizer);
334      var msg = "extraneous input " + tokenName + " expecting " +
335          expecting.toString(recognizer.literalNames, recognizer.symbolicNames);
336      recognizer.notifyErrorListeners(msg, t, null);
337  };
338  // This method is called to report a syntax error which requires the
339  // insertion of a missing token into the input stream. At the time this
340  // method is called, the missing token has not yet been inserted. When this
341  // method returns, {@code recognizer} is in error recovery mode.
342  //
343  // <p>This method is called when {@link //singleTokenInsertion} identifies
344  // single-token insertion as a viable recovery strategy for a mismatched
345  // input error.</p>
346  //
347  // <p>The default implementation simply returns if the handler is already in
348  // error recovery mode. Otherwise, it calls {@link //beginErrorCondition} to
349  // enter error recovery mode, followed by calling
350  // {@link Parser//notifyErrorListeners}.</p>
351  //
352  // @param recognizer the parser instance
353  //
354  DefaultErrorStrategy.prototype.reportMissingToken = function(recognizer) {
355      if ( this.inErrorRecoveryMode(recognizer)) {
356          return;
357      }
358      this.beginErrorCondition(recognizer);
359      var t = recognizer.getCurrentToken();
360      var expecting = this.getExpectedTokens(recognizer);
361      var msg = "missing " + expecting.toString(recognizer.literalNames, recognizer.symbolicNames) +
362            " at " + this.getTokenErrorDisplay(t);
363      recognizer.notifyErrorListeners(msg, t, null);
364  };

366  // <p>The default implementation attempts to recover from the mismatched input
367  // by using single token insertion and deletion as described below. If the
368  // recovery attempt fails, this method throws an
369  // {@link InputMismatchException}.</p>
370  //
371  // <p><strong>EXTRA TOKEN</strong> (single token deletion)</p>
372  //
373  // <p>{@code LA(1)} is not what we are looking for. If {@code LA(2)} has the
374  // right token, however, then assume {@code LA(1)} is some extra spurious
375  // token and delete it. Then consume and return the next token (which was
376  // the {@code LA(2)} token) as the successful result of the match operation.</p>
377  //
378  // <p>This recovery strategy is implemented by {@link
379  // //singleTokenDeletion}.</p>
380  //
381  // <p><strong>MISSING TOKEN</strong> (single token insertion)</p>
382  //
383  // <p>If current token (at {@code LA(1)}) is consistent with what could come
384  // after the expected {@code LA(1)} token, then assume the token is missing
385  // and use the parser's {@link TokenFactory} to create it on the fly. The
386  // "insertion" is performed by returning the created token as the successful
387  // result of the match operation.</p>
388  //
389  // <p>This recovery strategy is implemented by {@link
390  // //singleTokenInsertion}.</p>
391  //
392  // <p><strong>EXAMPLE</strong></p>
393  //
394  // <p>For example, Input {@code i=(3;} is clearly missing the {@code ')'}. When
395  // the parser returns from the nested call to {@code expr}, it will have
396  // call chain:</p>
397  //
398  // <pre>
399  // stat &rarr; expr &rarr; atom
400  // </pre>
401  //
402  // and it will be trying to match the {@code ')'} at this point in the
403  // derivation:
404  //
405  // <pre>
406  // =&gt; ID '=' '(' INT ')' ('+' atom)* ';'
407  // ^
408  // </pre>
409  //
410  // The attempt to match {@code ')'} will fail when it sees {@code ';'} and
411  // call {@link //recoverInline}. To recover, it sees that {@code LA(1)==';'}
412  // is in the set of tokens that can follow the {@code ')'} token reference
413  // in rule {@code atom}. It can assume that you forgot the {@code ')'}.
414  //
415  DefaultErrorStrategy.prototype.recoverInline = function(recognizer) {
416      // SINGLE TOKEN DELETION
417      var matchedSymbol = this.singleTokenDeletion(recognizer);
418      if (matchedSymbol !== null) {
419          // we have deleted the extra token.
420          // now, move past ttype token as if all were ok
421          recognizer.consume();
422          return matchedSymbol;
423      }
424      // SINGLE TOKEN INSERTION
425      if (this.singleTokenInsertion(recognizer)) {
426          return this.getMissingSymbol(recognizer);
427      }
428      // even that didn't work; must throw the exception
429      throw new InputMismatchException(recognizer);
430  };

432  //
433  // This method implements the single-token insertion inline error recovery
434  // strategy. It is called by {@link //recoverInline} if the single-token
435  // deletion strategy fails to recover from the mismatched input. If this
436  // method returns {@code true}, {@code recognizer} will be in error recovery
437  // mode.
438  //
439  // <p>This method determines whether or not single-token insertion is viable by
440  // checking if the {@code LA(1)} input symbol could be successfully matched
441  // if it were instead the {@code LA(2)} symbol. If this method returns
442  // {@code true}, the caller is responsible for creating and inserting a
443  // token with the correct type to produce this behavior.</p>
444  //
445  // @param recognizer the parser instance
446  // @return {@code true} if single-token insertion is a viable recovery
447  // strategy for the current mismatched input, otherwise {@code false}
448  //
449  DefaultErrorStrategy.prototype.singleTokenInsertion = function(recognizer) {
450      var currentSymbolType = recognizer.getTokenStream().LA(1);
451      // if current token is consistent with what could come after current
452      // ATN state, then we know we're missing a token; error recovery
453      // is free to conjure up and insert the missing token
454      var atn = recognizer._interp.atn;
455      var currentState = atn.states[recognizer.state];
456      var next = currentState.transitions[0].target;
457      var expectingAtLL2 = atn.nextTokens(next, recognizer._ctx);
458      if (expectingAtLL2.contains(currentSymbolType) ){
459          this.reportMissingToken(recognizer);
460          return true;
461      } else {
462          return false;
463      }
464  };

466  // This method implements the single-token deletion inline error recovery
467  // strategy. It is called by {@link //recoverInline} to attempt to recover
468  // from mismatched input. If this method returns null, the parser and error
469  // handler state will not have changed. If this method returns non-null,
470  // {@code recognizer} will <em>not</em> be in error recovery mode since the
471  // returned token was a successful match.
472  //
473  // <p>If the single-token deletion is successful, this method calls
474  // {@link //reportUnwantedToken} to report the error, followed by
475  // {@link Parser//consume} to actually "delete" the extraneous token. Then,
476  // before returning {@link //reportMatch} is called to signal a successful
477  // match.</p>
478  //
479  // @param recognizer the parser instance
480  // @return the successfully matched {@link Token} instance if single-token
481  // deletion successfully recovers from the mismatched input, otherwise
482  // {@code null}
483  //
484  DefaultErrorStrategy.prototype.singleTokenDeletion = function(recognizer) {
485      var nextTokenType = recognizer.getTokenStream().LA(2);
486      var expecting = this.getExpectedTokens(recognizer);
487      if (expecting.contains(nextTokenType)) {
488          this.reportUnwantedToken(recognizer);
489          // print("recoverFromMismatchedToken deleting " \
490          // + str(recognizer.getTokenStream().LT(1)) \
491          // + " since " + str(recognizer.getTokenStream().LT(2)) \
492          // + " is what we want", file=sys.stderr)
493          recognizer.consume(); // simply delete extra token
494          // we want to return the token we're actually matching
495          var matchedSymbol = recognizer.getCurrentToken();
496          this.reportMatch(recognizer); // we know current token is correct
497          return matchedSymbol;
498      } else {
499          return null;
500      }
501  };

503  // Conjure up a missing token during error recovery.
504  //
505  // The recognizer attempts to recover from single missing
506  // symbols. But, actions might refer to that missing symbol.
507  // For example, x=ID {f($x);}. The action clearly assumes
508  // that there has been an identifier matched previously and that
509  // $x points at that token. If that token is missing, but
510  // the next token in the stream is what we want we assume that
511  // this token is missing and we keep going. Because we
512  // have to return some token to replace the missing token,
513  // we have to conjure one up. This method gives the user control
514  // over the tokens returned for missing tokens. Mostly,
515  // you will want to create something special for identifier
516  // tokens. For literals such as '{' and ',', the default
517  // action in the parser or tree parser works. It simply creates
518  // a CommonToken of the appropriate type. The text will be the token.
519  // If you change what tokens must be created by the lexer,
520  // override this method to create the appropriate tokens.
521  //
522  DefaultErrorStrategy.prototype.getMissingSymbol = function(recognizer) {
523      var currentSymbol = recognizer.getCurrentToken();
524      var expecting = this.getExpectedTokens(recognizer);
525      var expectedTokenType = expecting.first(); // get any element
526      var tokenText;
527      if (expectedTokenType===Token.EOF) {
528          tokenText = "<missing EOF>";
529      } else {
530          tokenText = "<missing " + recognizer.literalNames[expectedTokenType] + ">";
531      }
532      var current = currentSymbol;
533      var lookback = recognizer.getTokenStream().LT(-1);
534      if (current.type===Token.EOF && lookback !== null) {
535          current = lookback;
536      }
537      return recognizer.getTokenFactory().create(current.source,
538          expectedTokenType, tokenText, Token.DEFAULT_CHANNEL,
539          -1, -1, current.line, current.column);
540  };

542  DefaultErrorStrategy.prototype.getExpectedTokens = function(recognizer) {
543      return recognizer.getExpectedTokens();

546  // How should a token be displayed in an error message? The default
547  // is to display just the text, but during development you might
548  // want to have a lot of information spit out. Override in that case
549  // to use t.toString() (which, for CommonToken, dumps everything about
550  // the token). This is better than forcing you to override a method in
551  // your token objects because you don't have to go modify your lexer
552  // so that it creates a new Java type.
553  //
554  DefaultErrorStrategy.prototype.getTokenErrorDisplay = function(t) {
555      if (t === null) {
556          return "<no token>";
557      }
558      var s = t.text;
559      if (s === null) {
560          if (t.type===Token.EOF) {
561              s = "<EOF>";
562          } else {
563              s = "<" + t.type + ">";
564          }
565      }
566      return this.escapeWSAndQuote(s);
567  };

569  DefaultErrorStrategy.prototype.escapeWSAndQuote = function(s) {
570      s = s.replace(/\n/g,"\\n");
571      s = s.replace(/\r/g,"\\r");
572      s = s.replace(/\t/g,"\\t");
573      return "'" + s + "'";
574  };

576  // Compute the error recovery set for the current rule. During
577  // rule invocation, the parser pushes the set of tokens that can
578  // follow that rule reference on the stack; this amounts to
579  // computing FIRST of what follows the rule reference in the
580  // enclosing rule. See LinearApproximator.FIRST().
581  // This local follow set only includes tokens
582  // from within the rule; i.e., the FIRST computation done by
583  // ANTLR stops at the end of a rule.
584  //
585  // EXAMPLE
586  //
587  // When you find a "no viable alt exception", the input is not
588  // consistent with any of the alternatives for rule r. The best
589  // thing to do is to consume tokens until you see something that
590  // can legally follow a call to r//or* any rule that called r.
591  // You don't want the exact set of viable next tokens because the
592  // input might just be missing a token--you might consume the
593  // rest of the input looking for one of the missing tokens.
594  //
595  // Consider grammar:
596  //
597  // a : '[' b ']'
598  // | '(' b ')'
599  // ;
600  // b : c '^' INT ;
601  // c : ID
602  // | INT
603  // ;
604  //
605  // At each rule invocation, the set of tokens that could follow
606  // that rule is pushed on a stack. Here are the various
607  // context-sensitive follow sets:
608  //
609  // FOLLOW(b1_in_a) = FIRST(']') = ']'
610  // FOLLOW(b2_in_a) = FIRST(')') = ')'
611  // FOLLOW(c_in_b) = FIRST('^') = '^'
612  //
613  // Upon erroneous input "[]", the call chain is
614  //
615  // a -> b -> c
616  //
617  // and, hence, the follow context stack is:
618  //
619  // depth follow set start of rule execution
620  // 0 <EOF> a (from main())
621  // 1 ']' b
622  // 2 '^' c
623  //
624  // Notice that ')' is not included, because b would have to have
625  // been called from a different context in rule a for ')' to be
626  // included.
627  //
628  // For error recovery, we cannot consider FOLLOW(c)
629  // (context-sensitive or otherwise). We need the combined set of
630  // all context-sensitive FOLLOW sets--the set of all tokens that
631  // could follow any reference in the call chain. We need to
632  // resync to one of those tokens. Note that FOLLOW(c)='^' and if
633  // we resync'd to that token, we'd consume until EOF. We need to
634  // sync to context-sensitive FOLLOWs for a, b, and c: {']','^'}.
635  // In this case, for input "[]", LA(1) is ']' and in the set, so we would
636  // not consume anything. After printing an error, rule c would
637  // return normally. Rule b would not find the required '^' though.
638  // At this point, it gets a mismatched token error and throws an
639  // exception (since LA(1) is not in the viable following token
640  // set). The rule exception handler tries to recover, but finds
641  // the same recovery set and doesn't consume anything. Rule b
642  // exits normally returning to rule a. Now it finds the ']' (and
643  // with the successful match exits errorRecovery mode).
644  //
645  // So, you can see that the parser walks up the call chain looking
646  // for the token that was a member of the recovery set.
647  //
648  // Errors are not generated in errorRecovery mode.
649  //
650  // ANTLR's error recovery mechanism is based upon original ideas:
651  //
652  // "Algorithms + Data Structures = Programs" by Niklaus Wirth
653  //
654  // and
655  //
656  // "A note on error recovery in recursive descent parsers":
657  // http://portal.acm.org/citation.cfm?id=947902.947905
658  //
659  // Later, Josef Grosch had some good ideas:
660  //
661  // "Efficient and Comfortable Error Recovery in Recursive Descent
662  // Parsers":
663  // ftp://www.cocolab.com/products/cocktail/doca4.ps/ell.ps.zip
664  //
665  // Like Grosch I implement context-sensitive FOLLOW sets that are combined
666  // at run-time upon error to avoid overhead during parsing.
667  //
668  DefaultErrorStrategy.prototype.getErrorRecoverySet = function(recognizer) {
669      var atn = recognizer._interp.atn;
670      var ctx = recognizer._ctx;
671      var recoverSet = new IntervalSet();
672      while (ctx !== null && ctx.invokingState>=0) {
673          // compute what follows who invoked us
674          var invokingState = atn.states[ctx.invokingState];
675          var rt = invokingState.transitions[0];
676          var follow = atn.nextTokens(rt.followState);
677          recoverSet.addSet(follow);
678          ctx = ctx.parentCtx;
679      }
680      recoverSet.removeOne(Token.EPSILON);
681      return recoverSet;
682  };
683  
684  // Consume tokens until one matches the given token set.//
685  DefaultErrorStrategy.prototype.consumeUntil = function(recognizer, set) {
686      var ttype = recognizer.getTokenStream().LA(1);
687      while( ttype !== Token.EOF && !set.contains(ttype)) {
688          recognizer.consume();
689          ttype = recognizer.getTokenStream().LA(1);
690      }
691  };

693  //
694  // This implementation of {@link ANTLRErrorStrategy} responds to syntax errors
695  // by immediately canceling the parse operation with a
696  // {@link ParseCancellationException}. The implementation ensures that the
697  // {@link ParserRuleContext//exception} field is set for all parse tree nodes
698  // that were not completed prior to encountering the error.
699  //
700  // <p>
701  // This error strategy is useful in the following scenarios.</p>
702  //
703  // <ul>
704  // <li><strong>Two-stage parsing:</strong> This error strategy allows the first
705  // stage of two-stage parsing to immediately terminate if an error is
706  // encountered, and immediately fall back to the second stage. In addition to
707  // avoiding wasted work by attempting to recover from errors here, the empty
708  // implementation of {@link BailErrorStrategy//sync} improves the performance of
709  // the first stage.</li>
710  // <li><strong>Silent validation:</strong> When syntax errors are not being
711  // reported or logged, and the parse result is simply ignored if errors occur,
712  // the {@link BailErrorStrategy} avoids wasting work on recovering from errors
713  // when the result will be ignored either way.</li>
714  // </ul>
715  //
716  // <p>
717  // {@code myparser.setErrorHandler(new BailErrorStrategy());}</p>
718  //
719  // @see Parser//setErrorHandler(ANTLRErrorStrategy)
720  //
721  function BailErrorStrategy() {
722  	DefaultErrorStrategy.call(this);
723  	return this;
724  }
725  
726  BailErrorStrategy.prototype = Object.create(DefaultErrorStrategy.prototype);
727  BailErrorStrategy.prototype.constructor = BailErrorStrategy;
728  
729  // Instead of recovering from exception {@code e}, re-throw it wrapped
730  // in a {@link ParseCancellationException} so it is not caught by the
731  // rule function catches. Use {@link Exception//getCause()} to get the
732  // original {@link RecognitionException}.
733  //
734  BailErrorStrategy.prototype.recover = function(recognizer, e) {
735      var context = recognizer._ctx;
736      while (context !== null) {
737          context.exception = e;
738          context = context.parentCtx;
739      }
740      throw new ParseCancellationException(e);
741  };
742  
743  // Make sure we don't attempt to recover inline; if the parser
744  // successfully recovers, it won't throw an exception.
745  //
746  BailErrorStrategy.prototype.recoverInline = function(recognizer) {
747      this.recover(recognizer, new InputMismatchException(recognizer));
748  };
749  
750  // Make sure we don't attempt to recover from problems in subrules.//
751  BailErrorStrategy.prototype.sync = function(recognizer) {
752      // pass
753  };

755  exports.BailErrorStrategy = BailErrorStrategy;
756  exports.DefaultErrorStrategy = DefaultErrorStrategy;
