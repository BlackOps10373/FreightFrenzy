61   function initializeBlocks() {
62     var projectName = getParameterByName('project');
63     if (isValidProjectName(projectName) && !containsAmpersand(projectName) && containsAtLeastOneAlphanumeric(projectName)) {
64       currentProjectName = projectName;
65       getBlocksJavaClassName(currentProjectName, function(className, errorMessage) {
66         if (className) {
67           currentClassName = className;
68           Blockly.FtcJava.setClassNameForFtcJava_(currentClassName);
69         } else {
70           if (errorMessage) {
71             alert(errorMessage);
72           } else {
73             alert('Error: The specified project name is not valid. Please rename it in the Blocks projects page.');
74           }
75         }
76       });
77       fetchBlkFileContent(currentProjectName, function(blkFileContent, errorMessage) {
78         if (blkFileContent) {
79           var blocksLoadedCallback = function() {
80             showJava();
81           };
82           loadBlocks(blkFileContent, blocksLoadedCallback);
83         } else {
84           alert(errorMessage);
85         }
86       });
87     } else {
88       alert('Error: The specified project name is not valid. Please rename it on the Blocks projects page.');
89     }
90   }

105  /** 
106   * Get a URL parameter by name. 
107   * From http://stackoverflow.com/a/901144 
108   */
109  function getParameterByName(name) {
110    url = window.location.href;
111    name = name.replace(/[\[\]]/g, '\\$&');
112    var regex = new RegExp('[?&]' + name + '(=([^&#]*)|&|#|$)'),
113        results = regex.exec(url);
114    if (!results) return null;
115    if (!results[2]) return '';
116    return decodeURIComponent(results[2].replace(/\+/g, ' '));
117  }

195  function saveProjectNow(opt_success_callback) {
196    if (currentProjectName) {
197      var allBlocks = workspace.getAllBlocks();
198      for (var iBlock = 0, block; block = allBlocks[iBlock]; iBlock++) {
199        saveBlockWarningHidden(block);
200      }
201      // Get the blocks as xml (text).
202      var blocksContent = Blockly.Xml.domToText(Blockly.Xml.workspaceToDom(workspace));
203      // Don't bother saving if there are no blocks.
204      if (blocksContent.indexOf('<block') > -1) {
205        var disabled = disableOrphans();
206        // Generate JavaScript code.
207        var jsFileContent = Blockly.JavaScript.workspaceToCode(workspace);
208        reenableOrphans(disabled);
209        var flavorSelect = document.getElementById('project_flavor');
210        var flavor = flavorSelect.options[flavorSelect.selectedIndex].value;
211        var group = document.getElementById('project_group').value;
212        var autoTransitionSelect = document.getElementById('project_autoTransition');
213        var autoTransition = autoTransitionSelect.options[autoTransitionSelect.selectedIndex].value;
214        var blkFileContent = blocksContent + formatExtraXml(flavor, group, autoTransition, projectEnabled);
215        saveProject(currentProjectName, blkFileContent, jsFileContent,
216            function(success, errorMessage) {
217          if (success) {
218            isDirty = false;
219            document.getElementById('saveSuccess').style.display = 'inline-block';
220            document.getElementById('saveFailure').style.display = 'none';
221            window.setTimeout(function() {
222              document.getElementById('saveSuccess').style.display = 'none';
223            }, 3000);
224            if (opt_success_callback) {
225              opt_success_callback();
226            }
227          } else {
228            document.getElementById('saveSuccess').style.display = 'none';
229            document.getElementById('saveFailure').innerHTML = errorMessage;
230            document.getElementById('saveFailure').style.display = 'inline-block';
231          }
232        });
233      }
234    } else {
235      alert('The specified project name is not valid');
236    }
237  }

239  function disableOrphans() {
240    Blockly.Events.disable();
241    var disabled = [];
242    var blocks = workspace.getTopBlocks(true);
243    for (var x = 0, block; block = blocks[x]; x++) {
244      if (block.type != 'procedures_defnoreturn' &&
245          block.type != 'procedures_defreturn' &&
246          block.isEnabled()) {
247        do {
248          block.setEnabled(false);
249          disabled.push(block);
250          block = block.getNextBlock();
251        } while (block);
252      }
253    }
254    Blockly.Events.enable();
255    return disabled;
256  }
257  
258  function reenableOrphans(disabled) {
259    Blockly.Events.disable();
260    for (var x = 0, block; block = disabled[x]; x++) {
261      block.setEnabled(true);
262    }
263    Blockly.Events.enable();
264  }

293  // Initialize global variables & blockly itself
294  function initializeBlockly() {
295    addReservedWordsForJavaScript();
296    addReservedWordsForFtcJava();
297  
298    document.addEventListener('mousemove', onMouseMove);
299    document.addEventListener('keydown', onKeyDown);
300  
301    // Blockly's text_quotes extension (which uses images for the quotes) causes the Download
302    // Image feature fail. Here, we replace it with one that uses quote characters.
303    Blockly.Extensions.ALL_['text_quotes'] = function() {
304      for (var i = 0, input; input = this.inputList[i]; i++) {
305        for (var j = 0, field; field = input.fieldRow[j]; j++) {
306          if ('TEXT' == field.name) {
307            var before = workspace.RTL ? '\u201D' : '\u201C';
308            var after = workspace.RTL ? '\u201C' : '\u201D';
309            input.insertFieldAt(j, new Blockly.FieldLabel(before));
310            input.insertFieldAt(j + 2, new Blockly.FieldLabel(after));
311            return;
312          }
313        }
314      }
315    };
316  
317    isDirty = false;
318    showJavaCheckbox = document.getElementById('show_java');
319    javaArea = document.getElementById('javaArea');
320    javaContent = document.getElementById('javaContent');
321    parentArea = document.getElementById('parentArea');
322    blocksAndBannerArea = document.getElementById('blocksAndBannerArea');
323    blocklyArea = document.getElementById('blocklyArea');
324    blocklyDiv = document.getElementById('blocklyDiv');
325    banner = document.getElementById('banner');
326    bannerText = document.getElementById('bannerText');
327    bannerButton = document.getElementById('bannerBtn');
328    workspace = Blockly.inject(blocklyDiv, {
329      media: 'blockly/media/',
330      zoom: {
331        controls: true,
332        wheel: true,
333        startScale: 1.2,
334        maxScale: 5,
335        minScale: 0.3,
336        scaleSpeed: 1.2},
337      trashcan: false,
338      toolbox: document.getElementById('toolbox')
339    });
340  
341    if (parentArea.clientWidth >= 800) {
342      showJavaCheckbox.checked = true;
343    }
344    showJavaChanged();
345  
346    parentArea.style.visibility = 'visible'; // Unhide parentArea
347  
348    window.addEventListener('resize', resizeBlocklyArea, false);
349    resizeBlocklyArea();
350    window.addEventListener('beforeunload', function(e) {
351      if (!isDirty) {
352        return undefined;
353      }
354      // It doesn't matter what string we return here. The browser will always use a standard message
355      // for security reasons.
356      (e || window.event).returnValue = ''; // Gecko + IE
357      return ''; // Gecko + Webkit, Safari, Chrome etc.
358    });
359  
360    workspace.addChangeListener(function(event) {
361      isDirty = true;
362  
363      // Check blocks.
364      var blockIds = [];
365      switch (event.type) {
366        case Blockly.Events.FINISHED_LOADING:
367          blocksFinishedLoading = true;
368          break;
369        case Blockly.Events.BLOCK_CREATE:
370          Array.prototype.push.apply(blockIds, event.ids);
371          break;
372        case Blockly.Events.BLOCK_CHANGE:
373          if (event.blockId) {
374            blockIds.push(event.blockId);
375          }
376          break;
377        case Blockly.Events.BLOCK_DELETE:
378          // Remove deleted blocks from blockIdsWithMissingHardware.
379          for (var iId = 0, blockId; blockId = event.ids[iId]; iId++) {
380            if (blockIdsWithMissingHardware.includes(blockId)) {
381              var index = blockIdsWithMissingHardware.indexOf(blockId);
382              blockIdsWithMissingHardware.splice(index, 1);
383            }
384          }
385          break;
386      }
387      for (var i = 0; i < blockIds.length; i++) {
388        var block = workspace.getBlockById(blockIds[i]);
389        if (block) {
390          var hasWarningBits = checkBlock(block, missingHardware);
391          if (hasWarningBits & WarningBits.MISSING_HARDWARE) {
392            if (!blockIdsWithMissingHardware.includes(blockIds[i])) {
393              blockIdsWithMissingHardware.push(blockIds[i]);
394            }
395          } else {
396            if (blockIdsWithMissingHardware.includes(blockIds[i])) {
397              var index = blockIdsWithMissingHardware.indexOf(blockIds[i]);
398              blockIdsWithMissingHardware.splice(index, 1);
399            }
400            saveVisibleIdentifiers(block);
401          }
402        }
403      }
404      if (blocksFinishedLoading) {
405        showJava();
406      }
407    });
408  }

410  function resizeBlocklyArea() {
411    // Compute the absolute coordinates and dimensions of blocklyArea.
412    var x = 0;
413    var y = 0;
414    var w = blocklyArea.offsetWidth;
415    var h = blocklyArea.offsetHeight;
416    var element = blocklyArea;
417    do {
418      x += element.offsetLeft;
419      y += element.offsetTop;
420      element = element.offsetParent;
421    } while (element);
422    // Position blocklyDiv over blocklyArea.
423    blocklyDiv.style.left = x + 'px';
424    blocklyDiv.style.top = y + 'px';
425    blocklyDiv.style.width = w + 'px';
426    blocklyDiv.style.height = h + 'px';
427    Blockly.svgResize(workspace);
428  }

435  function loadBlocks(blkFileContent, opt_blocksLoaded_callback) {
436    // The blocks content is up to and including the first </xml>.
437    var i = blkFileContent.indexOf('</xml>');
438    var blocksContent = blkFileContent.substring(0, i + 6);
439  
440    var extra = parseExtraXml(blkFileContent);
441    var flavorSelect = document.getElementById('project_flavor');
442    for (var i = 0; i < flavorSelect.options.length; i++) {
443      if (flavorSelect.options[i].value == extra['flavor']) {
444        flavorSelect.selectedIndex = i;
445        break;
446      }
447    }
448    document.getElementById('project_group').value = extra['group'];
449    var foundAutoTransition = false;
450    if (extra['autoTransition']) {
451      var autoTransitionSelect = document.getElementById('project_autoTransition');
452      for (var i = 0; i < autoTransitionSelect.options.length; i++) {
453        if (autoTransitionSelect.options[i].value == extra['autoTransition']) {
454          foundAutoTransition = true;
455          autoTransitionSelect.selectedIndex = i;
456          break;
457        }
458      }
459    } else {
460      foundAutoTransition = true;
461    }
462    setAutoTransitionDisplay();
463    document.getElementById('project_enabled').checked = extra['enabled'];
464  
465    if (!foundAutoTransition) {
466      alert('The Op Mode named "' + extra['autoTransition'] + '" was previously chosen as the ' +
467          'Preselect TeleOp. There is currently no TeleOp Op Mode with that name.');
468    }
469  
470    loadBlocksIntoWorkspace(blocksContent, opt_blocksLoaded_callback);
471    checkDownloadImageFeature();
472  }

474  /** 
475   * Loads the given blocksContent into the workspace. 
476   */
477  function loadBlocksIntoWorkspace(blocksContent, opt_blocksLoaded_callback) {
478    document.title = titlePrefix + ' - ' + currentProjectName;
479    var escapedProjectName = currentProjectName.replace(/&/g, '&amp;');
480    document.getElementById('project_name').innerHTML = escapedProjectName;
481    missingHardware = [];
482    blockIdsWithMissingHardware = [];
483    workspace.clear();
484    Blockly.Xml.domToWorkspace(Blockly.Xml.textToDom(blocksContent), workspace);
485  
486    // Use a timeout to allow the workspace change event to come through. Then, show an alert
487    // if any blocks have warnings. Then clear isDirty.
488    setTimeout(function() {
489      if (blockIdsWithMissingHardware.length > 0) {
490        var message = (blockIdsWithMissingHardware.length == 1)
491            ? 'An error occured when trying to find the hardware devices for 1 block.\n\n'
492            : 'Errors occured when trying to find the hardware devices for ' + blockIdsWithMissingHardware.length +
493                ' blocks.\n\n';
494        if (missingHardware.length > 0) {
495          message += 'The following hardware devices were not found:\n';
496          for (var i = 0; i < missingHardware.length; i++) {
497            message += '    "' + missingHardware[i] + '"\n';
498          }
499        }
500        message += '\nIf the current configuration is not the appropriate configuration for this ' +
501            'blocks project, please activate the appropriate configuration and reload this page.';
502        alert(message);
503      }
504  
505      isDirty = false;
506  
507      if (opt_blocksLoaded_callback) {
508        opt_blocksLoaded_callback();
509      }
510    }, 50);
511  }

513  /** 
514   * Add/remove the block warning if the given block's identifier(s) are not in the active 
515   * configuration. Return true if the block now has a warning, false otherwise. 
516   */
517  function checkBlock(block, missingHardware) {
518    var warningBits = 0;
519    try {
520      var warningText = null; // null will remove any previous warning.
521      for (var iFieldName = 0; iFieldName < identifierFieldNames.length; iFieldName++) {
522        var identifierFieldName = identifierFieldNames[iFieldName];
523        var field = block.getField(identifierFieldName);
524        if (field) {
525          var identifierFieldValue = field.getValue();
526          var identifierFound = true;
527          var fieldHasOptions = false;
528  
529          if (typeof field.getOptions === 'function') {
530            // The identifier field is a dropdown field with options.
531            // Check if identifierFieldValue is the value of one of the options.
532            var options = field.getOptions();
533            fieldHasOptions = true;
534            identifierFound = false;
535            for (var iOption = 0; iOption < options.length; iOption++) {
536              if (options[iOption][1] == identifierFieldValue) {
537                identifierFound = true;
538                break;
539              }
540            }
541            if (!identifierFound) {
542              // Check if identifierFieldValue is the visible name of one of the options.
543              for (var iOption = 0; iOption < options.length; iOption++) {
544                if (options[iOption][0] == identifierFieldValue) {
545                  identifierFieldValue = options[iOption][1];
546                  field.setValue(identifierFieldValue);
547                  identifierFound = true;
548                  break;
549                }
550              }
551            }
552          } else {
553            // The identifier field is a noneditable field instead of a dropdown field.
554            // Since blockly doesn't allow a dropdown field to have zero options, we need to use a
555            // noneditable field (which looks similar) when there are no hardware items for this
556            // type of block.
557            identifierFound = false;
558          }
559  
560          if (!identifierFound) {
561            // identifierFieldValue is the name of the identifier that will be used in generated
562            // javascript, but it is not necessarily the same as the name that is used in the
563            // hardware configuration and should be displayed on the block. For example, the
564            // visible identifier name may contain characters that are not allowed in a javascript
565            // identifier. Also, the identifier may have a suffix that allows us to distinguish
566            // between hardware devices with the same configuration name or hardware devices that
567            // support multiple hardware interfaces.
568            // We now store the visible identifier names using block.data.
569            // Here, we try to use block.data to retrieve the visible identifier name.
570            var visibleIdentifierName;
571            if (block.data) {
572              if (block.data.startsWith('{')) {
573                var visibleIdentifierNames = parseBlockDataJSON(block);
574                visibleIdentifierName = visibleIdentifierNames[identifierFieldName];
575              } else {
576                // Some older versions save as plain text instead of JSON.
577                visibleIdentifierName = block.data;
578              }
579            } else {
580              // If the blocks file is even older, we don't know what the visible name actually is.
581              // The best we can do is to remove the hardware identifier suffix if there is one.
582              visibleIdentifierName = removeHardwareIdentifierSuffix(identifierFieldValue);
583            }
584            if (typeof field.setText === 'function') {
585              field.setText(visibleIdentifierName);
586            }
587            if (!missingHardware.includes(visibleIdentifierName)) {
588              missingHardware.push(visibleIdentifierName);
589            }
590            warningBits |= WarningBits.MISSING_HARDWARE;
591            if (fieldHasOptions) {
592              warningText = addWarning(warningText,
593                  '"' + visibleIdentifierName + '" is not in the current robot configuration.\n\n' +
594                  'Please activate the configuration that contains the hardware device named "' +
595                  visibleIdentifierName + '",\nor select a device that is in the current robot configuration.');
596            } else {
597              warningText = addWarning(warningText,
598                  '"' + visibleIdentifierName + '" is not in the current robot configuration.\n\n' +
599                  'Please activate the configuration that contains the hardware device named "' +
600                  visibleIdentifierName + '".');
601            }
602          }
603        }
604      }
605      if (block.type == 'vuforia_initialize' ||
606          block.type == 'vuforia_initializeExtended' ||
607          block.type == 'vuforia_initializeExtendedNoKey' ||
608          block.type == 'vuforia_initialize_withWebcam' ||
609          block.type == 'vuforia_activate' ||
610          block.type == 'vuforia_deactivate' ||
611          block.type == 'vuforia_track' ||
612          block.type == 'vuforia_trackPose' ||
613          block.type == 'vuforia_typedEnum_trackableName' ||
614          block.type == 'vuforiaTrackingResults_getProperty_RelicRecoveryVuMark' ||
615          block.type == 'vuMarks_typedEnum_relicRecoveryVuMark') {
616        warningBits |= WarningBits.RELIC_RECOVERY;
617        warningText = addWarning(warningText,
618            'This block is optimized for Relic Recovery (2017-2018) and will not work correctly ' +
619            'for other FTC games.\n\n' +
620            'Please replace this block with the corresponding one from the ' +
621            'Optimized for ' + vuforiaCurrentGameName + ' toolbox category.');
622      } else if (block.type == 'vuforiaRoverRuckus_initialize_withCameraDirection' ||
623          block.type == 'vuforiaRoverRuckus_initialize_withWebcam' ||
624          block.type == 'vuforiaRoverRuckus_activate' ||
625          block.type == 'vuforiaRoverRuckus_deactivate' ||
626          block.type == 'vuforiaRoverRuckus_track' ||
627          block.type == 'vuforiaRoverRuckus_trackPose' ||
628          block.type == 'vuforiaRoverRuckus_typedEnum_trackableName') {
629        warningBits |= WarningBits.ROVER_RUCKUS;
630        warningText = addWarning(warningText,
631            'This block is optimized for Rover Ruckus (2018-2019) and will not work correctly ' +
632            'for other FTC games.\n\n' +
633            'Please replace this block with the corresponding one from the ' +
634            'Optimized for ' + vuforiaCurrentGameName + ' toolbox category.');
635      } else if (block.type == 'tfodRoverRuckus_initialize' ||
636          block.type == 'tfodRoverRuckus_activate' ||
637          block.type == 'tfodRoverRuckus_deactivate' ||
638          block.type == 'tfodRoverRuckus_setClippingMargins' ||
639          block.type == 'tfodRoverRuckus_getRecognitions' ||
640          block.type == 'tfodRoverRuckus_typedEnum_label') {
641        warningBits |= WarningBits.ROVER_RUCKUS;
642        warningText = addWarning(warningText,
643            'This block is optimized for Rover Ruckus (2018-2019) and will not work correctly ' +
644            'for other FTC games.\n\n' +
645            'Please replace this block with the corresponding one from the ' +
646            'Optimized for ' + tfodCurrentGameName + ' toolbox category.');
647      } else if (block.type == 'misc_callJava_return' ||
648          block.type == 'misc_callJava_noReturn' ||
649          block.type == 'misc_callHardware_return' ||
650          block.type == 'misc_callHardware_noReturn') {
651        if (!methodLookupStrings.includes(block.ftcAttributes_.methodLookupString)) {
652          warningBits |= WarningBits.MISSING_METHOD;
653          warningText = addWarning(warningText,
654              'This block refers to a Java method that has been changed or removed. It will not ' +
655              'work correctly.\n\n' +
656              'Please replace or remove this block, or restore the Java method it refers to.');
657        }
658      }
659  
660      // If warningText is null, the following will clear a previous warning.
661      var previousWarningText = block.warning ? block.warning.getText() : null;
662      if (previousWarningText != warningText) {
663        block.setWarningText(warningText);
664        if (warningText && block.warning && !readBlockWarningHidden(block)) {
665          block.warning.setVisible(true);
666        }
667      }
668    } catch (e) {
669      console.log('Unable to verify the following block due to exception.');
670      console.log(block);
671      console.log(e);
672    }
673    return warningBits;
674  }

738  function saveVisibleIdentifiers(block) {
739    var data = parseBlockDataJSON(block);
740  
741    for (var iFieldName = 0; iFieldName < identifierFieldNames.length; iFieldName++) {
742      var identifierFieldName = identifierFieldNames[iFieldName];
743      var field = block.getField(identifierFieldName);
744      if (field) {
745        if (typeof field.getOptions === 'function') {
746          // The identifier field is a dropdown field with options.
747          // Save the user visible identifiers using block.data, so we can use it in the future if
748          // the hardware device is not found in the configuration.
749          if (!data) {
750            data = Object.create(null);
751          }
752          // field.getText() returns the text with spaces changed to non-breakable spaces \u00A0.
753          // Here we change them back to normal spaces. Without this, the character becomes &nbsp; in
754          // the blk file when saved with Safari.
755          data[identifierFieldName] = field.getText().replace(/\u00A0/g, ' ');
756        }
757      }
758    }
759  
760    block.data = data ? JSON.stringify(data) : null;
761  }

768  function onKeyDown(e) {
769    if (Blockly.mainWorkspace.options.readOnly || Blockly.utils.isTargetInput(e)) {
770      // No key actions on readonly workspaces.
771      // When focused on an HTML text input widget, don't trap any keys.
772      return;
773    }
774    if (e.altKey || e.ctrlKey || e.metaKey) {
775      if (Blockly.selected &&
776          Blockly.selected.isDeletable() && Blockly.selected.isMovable()) {
777        if (e.keyCode == 67 || // 'c' for copy.
778            e.keyCode == 88) { // 'x' for cut.
779          // Use a timeout so we can capture the blockly clipboard.
780          setTimeout(function() {
781            captureBlocklyClipboard();
782          }, 1);
783        }
784      }
785  
786      if (e.keyCode == 86) { // 'v' for paste.
787        // Override blockly's default paste behavior.
788        paste();
789        e.stopImmediatePropagation();
790      }
791    }
792  }

811  function pasteContent(clipboardContent) {
812    if (!clipboardContent) {
813      return;
814    }
815    var parser = new DOMParser();
816    var xmlDoc = parser.parseFromString(clipboardContent, 'text/xml');
817    var blocks = xmlDoc.getElementsByTagName('block');
818    if (blocks.length > 0) {
819      var block = blocks[0];
820      // Place the pasted block near mouse.
821      var svg = workspace.getParentSvg();
822      var point = svg.createSVGPoint();
823      point.x = mouseX;
824      point.y = mouseY;
825      point = point.matrixTransform(svg.getScreenCTM().inverse());
826      point = point.matrixTransform(workspace.getCanvas().getCTM().inverse());
827      block.setAttribute('x', point.x);
828      block.setAttribute('y', point.y);
829      workspace.paste(block);
830    }
831  }

854  function downloadImageButtonClicked() {
855    // Clone the workspace' svg canvas.
856    var svgCanvas = workspace.svgBlockCanvas_;
857    var clone = svgCanvas.cloneNode(true);
858    var box;
859    if (svgCanvas.tagName == 'svg') {
860      box = svgCanvas.getBoundingClientRect();
861      var metrics = workspace.getMetrics();
862      var left = (parseFloat(metrics.contentLeft) - parseFloat(metrics.viewLeft)).toString();
863      var top = (parseFloat(metrics.contentTop) - parseFloat(metrics.viewTop)).toString();
864      var right = (parseFloat(metrics.contentWidth)).toString();
865      var bottom = (parseFloat(metrics.contentHeight)).toString();
866      clone.setAttribute('viewBox', left + ' ' + top + ' ' + right + ' ' + bottom);
867    } else {
868      clone.setAttribute('transform',
869          clone.getAttribute('transform')
870              .replace(/translate\(.*?\)/, '')
871              .replace(/scale\(.*?\)/, '')
872              .trim());
873      var svg = document.createElementNS('http://www.w3.org/2000/svg','svg')
874      svg.setAttribute('xmlns', 'http://www.w3.org/2000/svg');
875      svg.appendChild(clone)
876      clone = svg;
877      box = svgCanvas.getBBox();
878      clone.setAttribute('viewBox', box.x + ' ' + box.y + ' ' + box.width + ' ' + box.height);
879    }
880    clone.setAttribute('version', '1.1');
881    clone.setAttribute('width', box.width);
882    clone.setAttribute('height', box.height);
883    clone.setAttribute('style', 'background-color: #FFFFFF');
884    var divElement = document.createElement('div');
885    divElement.appendChild(clone);
886  
887    // Collect style sheets.
888    var css = '';
889    var sheets = document.styleSheets;
890    for (var i = 0; i < sheets.length; i++) {
891      if (isExternal(sheets[i].href)) {
892        continue;
893      }
894      var rules = null;
895      try {
896        rules = sheets[i].cssRules;
897      } catch (err) {
898        // Cannot access cssRules from external css files in offline blocks editor.
899      }
900      if (rules != null) {
901        for (var j = 0; j < rules.length; j++) {
902          var rule = rules[j];
903          if (typeof(rule.style) != 'undefined') {
904            var match = null;
905            try {
906              match = svgCanvas.querySelector(rule.selectorText);
907            } catch (err) {
908            }
909            if (match && rule.selectorText.indexOf('blocklySelected') == -1) {
910              css += rule.selectorText + ' { ' + rule.style.cssText + ' }\n';
911            } else if(rule.cssText.match(/^@font-face/)) {
912              css += rule.cssText + '\n';
913            }
914          }
915        }
916      }
917    }
918    var styleElement = document.createElement('style');
919    styleElement.setAttribute('type', 'text/css');
920    styleElement.innerHTML = '<![CDATA[\n' + css + '\n]]>';
921    var defsElement = document.createElement('defs');
922    defsElement.appendChild(styleElement);
923    clone.insertBefore(defsElement, clone.firstChild);
924  
925    // TODO(lizlooney): hide all blocklyScrollbarHandle, blocklyScrollbarBackground, image,
926    // .blocklyMainBackground, rectCorner, indicatorWarning?
927  
928    var doctype = '<?xml version="1.0" standalone="no"?>' +
929        '<!DOCTYPE svg PUBLIC "-//W3C//DTD SVG 1.1//EN" ' +
930        '"http://www.w3.org/Graphics/SVG/1.1/DTD/svg11.dtd">';
931    var svg = doctype + divElement.innerHTML;
932    svg = svg.replace(/&nbsp/g, '&#160');
933    svg = svg.replace(/sans-serif/g, 'Arial, Verdana, "Nimbus Sans L", Helvetica');
934    var uri = 'data:image/svg+xml;base64,' + window.btoa(unescape(encodeURIComponent(svg)));
935  
936    var image = new Image();
937    image.onload = function() {
938      var canvasElement = document.createElement('canvas');
939      canvasElement.width = image.width;
940      canvasElement.height = image.height;
941      canvasElement.getContext('2d').drawImage(image, 0, 0);
942      canvasElement.toBlob(function(blob) {
943        var a = document.getElementById('download_link');
944        a.href = URL.createObjectURL(blob);
945        a.download = currentProjectName + '.png';
946        a.target = '_blank';
947        a.click();
948      });
949    };
950    image.onerror = function (e) {
951      alert('Unable to download blocks image. Sorry about that!');
952    };
953    image.src = uri;
954  }
