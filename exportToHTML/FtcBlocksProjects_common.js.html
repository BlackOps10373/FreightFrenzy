56   function resize() {
57     // Compute the height of projectsTableScroll.
58     var projectsTableScroll = document.getElementById('projectsTableScroll');
59     var element = projectsTableScroll;
60     var y = 0;
61     do {
62       y += element.offsetTop;
63       element = element.offsetParent;
64     } while (element);
65     projectsTableScroll.style.height = (window.innerHeight - y) + 'px';
66   }

88   function initializeSamples() {
89     var select = document.getElementById('newProjectSamplesSelect');
90     select.options.length = 0; // Clear previous values just in case.
91   
92     // Insert the first option, which is for the default op mode.
93     var option = document.createElement('option')
94     option.innerHTML = 'BasicOpMode';
95     option.value = ''; // Indicates the default op mode.
96     select.appendChild(option);
97   
98     fetchSamples(function(jsonSamples, errorMessage) {
99       if (jsonSamples) {
100        var samples = JSON.parse(jsonSamples);
101        for (var i = 0; i < samples.length; i++) {
102          var sample = samples[i];
103          var option = document.createElement('option');
104          option.innerHTML = sample.escapedName;
105          option.value = JSON.stringify(sample);
106          select.appendChild(option);
107        }
108        // Now we can enable the Create New Op Mode  button.
109        document.getElementById('newProjectButton').disabled = false;
110      } else {
111        console.log(errorMessage);
112      }
113    });

137    if (sortByName) {
138      sortAscending = !sortAscending;
139    } else {

141      // When sorting by name, sorting ascending makes more sense than descending.
142      sortAscending = true;

148  function toggleSortByDateModified() {
149    if (sortByDateModified) {
150      sortAscending = !sortAscending;
151    } else {
152      sortByDateModified = true;
153      // When sorting by date, sorting descending makes more sense than ascending.
154      sortAscending = false;
155      sortByName = false;
156    }
157    sortProjectsAndFillTable();

160  function sortProjectsAndFillTable() {
161    if (sortByName) {
162      projects.sort(function(project1, project2) {
163        return project1.name.localeCompare(project2.name);
164      });
165    } else if (sortByDateModified) {
166      projects.sort(function(project1, project2) {
167        return project1.dateModifiedMillis - project2.dateModifiedMillis;
168      });
169    }
170    if (!sortAscending) {
171      projects.reverse();
172    }
173  
174    var table = document.getElementById('projectsTable');
175    // Remove all rows except the first one, which contains the column headers.
176    for (var i = table.rows.length - 1; i >= 1; i--) {
177      table.deleteRow(i);
178    }
179    for (var i = 0; i < projects.length; i++) {
180      var tr = table.insertRow(-1);
181      tr.className = 'project_tr';
182  
183      var tdTrash = tr.insertCell(-1);
184      tdTrash.innerHTML = '<input type="checkbox" id="checkbox_' + i + '" onclick="projectCheckChanged(' + i + ')">';
185  
186      var tdName = tr.insertCell(-1);
187      if (isValidProjectName(projects[i].name) &&
188          containsAtLeastOneAlphanumeric(projects[i].name) &&
189          !containsAmpersand(projects[i].name)) {
190        tdName.innerHTML = '<div class="project_name" onclick="openProject(' + i + ')">' +
191            projects[i].escapedName + '</div>';
192      } else {
193        tdName.innerHTML = '<div class="invalid_project_name">' +
194            projects[i].escapedName + '&nbsp;&nbsp;&nbsp;&nbsp;<span class="tooltiptext">This Op Mode must be renamed.</span></div>';
195      }
196  
197      var tdDateModified = tr.insertCell(-1);
198      tdDateModified.innerHTML = formatTimestamp(projects[i].dateModifiedMillis);
199  
200      var tdEnabled = tr.insertCell(-1);
201      tdEnabled.innerHTML = '<input type="checkbox" id="project_enabled_' + i + '" onclick="projectEnabledChanged(' + i + ')"' +
202          (projects[i].enabled ? ' checked' : '') + '>';
203    }
204  
205    var upTriangle = '&#x25B2;';
206    var downTriangle = '&#x25BC;';
207    var nameSortIndicator = document.getElementById('sortByNameIndicator');
208    if (sortByName) {
209      nameSortIndicator.innerHTML = (sortAscending ? upTriangle : downTriangle);
210    } else {
211      nameSortIndicator.innerHTML = '';
212    }
213    var dateSortIndicator = document.getElementById('sortByDateModifiedIndicator');
214    if (sortByDateModified) {
215      dateSortIndicator.innerHTML = (sortAscending ? upTriangle : downTriangle);
216    } else {
217      dateSortIndicator.innerHTML = '';
218    }
219  
220    checkedProjects = [];
221    updateButtons();
222  }

224  function formatTimestamp(timestampMillis) {
225    var date = new Date(timestampMillis);
226    var monthNames = ['January', 'February', 'March', 'April', 'May', 'June',
227        'July', 'August', 'September', 'October', 'November', 'December'];
228    var formatted = monthNames[date.getMonth()] + '&nbsp;' + date.getDate() +
229        ',&nbsp;' + date.getFullYear() + ',&nbsp;';
230    if (date.getHours() == 0) {
231      formatted += '12';
232    } else if (date.getHours() > 12) {
233      formatted += (date.getHours() - 12);
234    } else {
235      formatted += date.getHours();
236    }
237    formatted += ':';
238    if (date.getMinutes() < 10) {
239      formatted += '0';
240    }
241    formatted += date.getMinutes() + ':'
242    if (date.getSeconds() < 10) {
243      formatted += '0';
244    }
245    formatted += date.getSeconds() + '&nbsp;';
246    if (date.getHours() < 12) {
247      formatted += 'AM';
248    } else {
249      formatted += 'PM';
250    }
251    return formatted;
252  }

272  function okNewProjectNameDialog() {
273    // Validate name for legal characters and for existing project names.
274    var newProjectName = document.getElementById('newProjectName').value;
275    if (!isValidProjectName(newProjectName) || containsAmpersand(newProjectName)) {
276      document.getElementById('newProjectNameError').innerHTML =
277          'The project name must only contain alphanumeric<br>characters and !$%\'()+,-.;=@[]^_{}~.';
278      return;
279    }
280    if (!containsAtLeastOneAlphanumeric(newProjectName)) {
281      document.getElementById('newProjectNameError').innerHTML =
282          'The project name must contain at least one alphanumeric character.';
283      return;
284    }
285    var nameClash = false;
286    for (var i = 0; i < projects.length; i++) {
287      if (newProjectName == projects[i].name) {
288        nameClash = true;
289        break;
290      }
291    }
292    if (nameClash) {
293      document.getElementById('newProjectNameError').innerHTML =
294          'There is already a project with this name.';
295      return;
296    }
297  
298    switch (newProjectNameDialogMode) {
299      case NEW_PROJECT_NAME_DIALOG_MODE_NEW_PROJECT:
300        newProjectOk(newProjectName);
301        break;
302      case NEW_PROJECT_NAME_DIALOG_MODE_RENAME_PROJECT:
303        renameProjectOk(projects[checkedProjects[0]].name, newProjectName);
304        break;
305      case NEW_PROJECT_NAME_DIALOG_MODE_COPY_PROJECT:
306        copyProjectOk(projects[checkedProjects[0]].name, newProjectName);
307        break;
308    }
309  }

349  function okUploadProjectDialog() {
350    var file = document.getElementById('uploadProjectFileInput').files[0];
351    var uploadProjectName = makeUploadProjectName(file.name);
352    var reader = new FileReader();
353    reader.onload = function(event) {
354      // Get the blocks from the uploaded file.
355      var blkFileContent = event.target.result;

358          document.getElementById('uploadProjectDialog'));
359    };

363  function makeUploadProjectName(uploadFileName) {
364    var lastDot = uploadFileName.lastIndexOf('.');
365    var preferredName = (lastDot == -1)
366        ? uploadFileName
367        : uploadFileName.substr(0, lastDot);
368    var name = preferredName; // No suffix.
369    var suffix = 0;
370    while (true) {
371      var nameClash = false;
372      for (var i = 0; i < projects.length; i++) {
373        if (name == projects[i].name) {
374          nameClash = true;
375          break;
376        }
377      }
378      if (!nameClash) {
379        return name;
380      }
381      suffix++;
382      name = preferredName + suffix;
383    }
384  }

386  function finishNewOrUploadProject(projectName, blkFileContent, errorElement, dialogElement) {
387    // The blocks content is up to and including the first </xml>.
388    var i = blkFileContent.indexOf('</xml>');
389    var blocksContent = blkFileContent.substring(0, i + 6);
390    var dom = '';
391    try {
392      dom = Blockly.Xml.textToDom(blocksContent);
393    } catch (e) {
394      errorElement.innerHTML = 'Error: Invalid blocks. ' + e;
395      return;
396    }
397    // Create a headless workspace to generate the JavaScript.
398    var jsFileContent = '';
399    try {
400      var workspace = new Blockly.Workspace();
401      // For consistency with previous versions, we explicitly set oneBasedIndex to true.
402      workspace.options.oneBasedIndex = true;
403      Blockly.Xml.domToWorkspace(dom, workspace);
404      jsFileContent = Blockly.JavaScript.workspaceToCode(workspace);
405    } catch (e) {
406      errorElement.innerHTML = 'Error: Could not generate code for blocks. ' + e;
407      return;
408    }
409    if (!jsFileContent) {
410      errorElement.innerHTML = 'Error: No code generated for blocks.';
411      return;
412    }
413  
414    saveProject(projectName, blkFileContent, jsFileContent, function(success, errorMessage) {
415      if (success) {
416        // Close the dialog.
417        dialogElement.style.display = 'none';
418        openProjectBlocks(projectName);
419      } else {
420        errorElement.innerHTML = errorMessage;
421      }
422    });
423  }

500  function deleteProjectsButtonClicked() {
501    starDelimitedDeleteProjectNames = '';
502    var table = document.getElementById('deleteProjectsTable');
503    // Remove all rows.
504    for (var i = table.rows.length - 1; i >= 0; i--) {
505      table.deleteRow(i);
506    }
507    // Gather the checked projects.
508    var delimiter = '';
509    for (var i = 0; i < projects.length; i++) {
510      var checkbox = document.getElementById('checkbox_' + i);
511      if (checkbox.checked) {
512        starDelimitedDeleteProjectNames += delimiter + projects[i].name;
513        delimiter = '*';
514        // Insert project name into the table.
515        table.insertRow(-1).insertCell(-1).innerHTML = projects[i].escapedName;
516      }
517    }
518    if (table.rows.length > 0) {
519      // Show modal dialog confirming that the user wants to delete the projects.
520      document.getElementById('deleteProjectsDialog').style.display = 'block';
521    }
522  }

542  function projectCheckAllChanged() {
543    var checkboxAll = document.getElementById('checkbox_all');
544    if (checkedProjects.length == 0) {
545      // Check all.
546      checkedProjects = [];
547      for (var i = 0; i < projects.length; i++) {
548        var checkbox = document.getElementById('checkbox_' + i);
549        checkbox.checked = true;
550        checkedProjects.push(i);
551      }
552      checkboxAll.checked = true;

554      // Check none.
555      checkedProjects = [];
556      for (var i = 0; i < projects.length; i++) {
557        var checkbox = document.getElementById('checkbox_' + i);
558        checkbox.checked = false;
559      }
560      checkboxAll.checked = false;
561    }
562    updateButtons();

565  function projectCheckChanged(i) {
566    var checkbox = document.getElementById('checkbox_' + i);
567    if (checkbox.checked) {
568      if (checkedProjects.indexOf(i) == -1) {
569        checkedProjects.push(i);
570      }
571    } else {
572      var index = checkedProjects.indexOf(i);
573      checkedProjects.splice(index, 1);
574    }
575    updateButtons();
576  }
