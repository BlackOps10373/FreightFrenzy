35   Blockly.FtcJava.getOutputType_text_ = function(block) {
36     switch (block.type) {
37       case 'text_length':
38       case 'text_indexOf':
39       case 'text_count':
40         return 'int';
41     }
42     return false;
43   };

45   Blockly.FtcJava['text'] = function(block) {
46     // Text value.
47     var code = Blockly.FtcJava.quote_(block.getFieldValue('TEXT'));
48     return [code, Blockly.FtcJava.ORDER_ATOMIC];
49   };

51   Blockly.FtcJava['text_join'] = function(block) {
52     // Create a string made up of any number of elements of any type.
53     if (block.itemCount_ == 0) {
54       return ['""', Blockly.FtcJava.ORDER_ATOMIC];
55     }
56   
57     var order = Blockly.FtcJava.ORDER_ATOMIC;
58     var code = Blockly.FtcJava.valueToCode(block, 'ADD0',
59         Blockly.FtcJava.ORDER_ADDITION) || '""';
60     var zeroOrOneIsString = code.startsWith('"');
61     for (var i = 1; i < block.itemCount_; i++) {
62       var element = Blockly.FtcJava.valueToCode(block, 'ADD' + i,
63           Blockly.FtcJava.ORDER_ADDITION) || '""';
64       if (i == 1 && element.startsWith('"')) {
65         zeroOrOneIsString = true;
66       }
67       code += ' + ' + element;
68       order = Blockly.FtcJava.ORDER_ADDITION;
69     }
70     if (!zeroOrOneIsString) {
71       code = '"" + ' + code;
72       order = Blockly.FtcJava.ORDER_ADDITION;
73     }
74     return [code, order];
75   };

77   Blockly.FtcJava['text_append'] = function(block) {
78     // Append to a variable in place.
79     var varName = Blockly.FtcJava.variableDB_.getName(
80         block.getFieldValue('VAR'), Blockly.Variables.NAME_TYPE);
81     var value = Blockly.FtcJava.valueToCode(block, 'TEXT',
82         Blockly.FtcJava.ORDER_ASSIGNMENT) || '""';
83     // TODO(lizlooney): What if the var is not a string?
84     return varName + ' += ' + value + ';\n';
85   };

87   Blockly.FtcJava['text_length'] = function(block) {
88     // String length.
89     var text = Blockly.FtcJava.valueToCode(block, 'VALUE',
90         Blockly.FtcJava.ORDER_MEMBER) || '""';
91     return [text + '.length()', Blockly.FtcJava.ORDER_FUNCTION_CALL];

101  Blockly.FtcJava['text_indexOf'] = function(block) {
102    // Search the text for a substring.
103    var operator = block.getFieldValue('END') == 'FIRST' ?
104        'indexOf' : 'lastIndexOf';
105    var substring = Blockly.FtcJava.valueToCode(block, 'FIND',
106        Blockly.FtcJava.ORDER_NONE) || '""';
107    var text = Blockly.FtcJava.valueToCode(block, 'VALUE',
108        Blockly.FtcJava.ORDER_MEMBER) || '""';
109    var code = text + '.' + operator + '(' + substring + ')';
110    // Adjust index if using one-based indices.
111    if (block.workspace.options.oneBasedIndex) {
112      return [code + ' + 1', Blockly.FtcJava.ORDER_ADDITION];
113    }
114    return [code, Blockly.FtcJava.ORDER_FUNCTION_CALL];
115  };

117  Blockly.FtcJava['text_charAt'] = function(block) {
118    // Get letter at index.
119    // Note: Until January 2013 this block did not have the WHERE input.
120    var where = block.getFieldValue('WHERE') || 'FROM_START';
121    var text = Blockly.FtcJava.valueToCode(block, 'VALUE',
122        Blockly.FtcJava.ORDER_COMMA) || '""';
123    var at;
124    switch (where) {
125      case 'FROM_START':
126      case 'FROM_END':
127        at = Blockly.FtcJava.getAdjusted(block, 'AT', Blockly.FtcJava.ORDER_COMMA);
128        break;
129      default:
130        at = '0';
131        break;
132    }
133    Blockly.FtcJava.generateImport_('JavaUtil');
134    var code = 'JavaUtil.inTextGetLetter(' + text + ', JavaUtil.AtMode.' + where + ', ' + at + ')';
135    return [code, Blockly.FtcJava.ORDER_FUNCTION_CALL];
136  };

138  Blockly.FtcJava['text_getSubstring'] = function(block) {
139    // Get substring.
140    var text = Blockly.FtcJava.valueToCode(block, 'STRING',
141        Blockly.FtcJava.ORDER_COMMA) || '""';
142    var where1 = block.getFieldValue('WHERE1');
143    var at1;
144    switch (where1) {
145      case 'FROM_START':
146      case 'FROM_END':
147        at1 = Blockly.FtcJava.getAdjusted(block, 'AT1', Blockly.FtcJava.ORDER_COMMA);
148        break;
149      default:
150        at1 = '0';
151        break;
152    }
153    var where2 = block.getFieldValue('WHERE2');
154    var at2;
155    switch (where2) {
156      case 'FROM_START':
157      case 'FROM_END':
158        at2 = Blockly.FtcJava.getAdjusted(block, 'AT2', Blockly.FtcJava.ORDER_COMMA);
159        break;
160      default:
161        at2 = '0';
162        break;
163    }
164    Blockly.FtcJava.generateImport_('JavaUtil');
165    var code = 'JavaUtil.inTextGetSubstring(' + text +
166        ', JavaUtil.AtMode.' + where1 + ', ' + at1 +
167        ', JavaUtil.AtMode.' + where2 + ', ' + at2 + ')';
168    return [code, Blockly.FtcJava.ORDER_FUNCTION_CALL];
169  };

171  Blockly.FtcJava['text_changeCase'] = function(block) {
172    // Change capitalization.
173    var OPERATORS = {
174      'UPPERCASE': '.toUpperCase()',
175      'LOWERCASE': '.toLowerCase()',
176      'TITLECASE': null
177    };
178    var operator = OPERATORS[block.getFieldValue('CASE')];
179    var textOrder = operator ? Blockly.FtcJava.ORDER_MEMBER :
180        Blockly.FtcJava.ORDER_NONE;
181    var text = Blockly.FtcJava.valueToCode(block, 'TEXT',
182        textOrder) || '""';
183    var code;
184    if (operator) {
185      // Upper and lower case are functions built into Java.
186      code = text + operator;
187    } else {
188      Blockly.FtcJava.generateImport_('JavaUtil');
189      code = 'JavaUtil.toTitleCase(' + text + ')';
190    }
191    return [code, Blockly.FtcJava.ORDER_FUNCTION_CALL];
192  };

194  Blockly.FtcJava['text_trim'] = function(block) {
195    // Trim spaces.
196    var mode = block.getFieldValue('MODE');
197    var textOrder = (mode == 'BOTH') ? Blockly.FtcJava.ORDER_MEMBER :
198        Blockly.FtcJava.ORDER_NONE;
199    var text = Blockly.FtcJava.valueToCode(block, 'TEXT',
200        textOrder) || '""';
201    var code;
202    if (mode == 'BOTH') {
203      code = text + '.trim()';
204    } else {
205      Blockly.FtcJava.generateImport_('JavaUtil');
206      code = 'JavaUtil.textTrim(' + text + ', JavaUtil.TrimMode.' + mode + ')';
207    }
208    return [code, Blockly.FtcJava.ORDER_FUNCTION_CALL];
209  };
