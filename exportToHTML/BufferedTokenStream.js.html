7    // This implementation of {@link TokenStream} loads tokens from a
8    // {@link TokenSource} on-demand, and places the tokens in a buffer to provide
9    // access to any previous token by index.
10   //
11   // <p>
12   // This token stream ignores the value of {@link Token//getChannel}. If your
13   // parser requires the token stream filter tokens to only those on a particular
14   // channel, such as {@link Token//DEFAULT_CHANNEL} or
15   // {@link Token//HIDDEN_CHANNEL}, use a filtering token stream such a
16   // {@link CommonTokenStream}.</p>
17   
18   var Token = require('./Token').Token;
19   var Lexer = require('./Lexer').Lexer;
20   var Interval = require('./IntervalSet').Interval;
21   
22   // this is just to keep meaningful parameter types to Parser
23   function TokenStream() {
24   	return this;
25   }
26   
27   function BufferedTokenStream(tokenSource) {
28   
29   	TokenStream.call(this);
30   	// The {@link TokenSource} from which tokens for this stream are fetched.
31   	this.tokenSource = tokenSource;
32   
33   	// A collection of all tokens fetched from the token source. The list is
34   	// considered a complete view of the input once {@link //fetchedEOF} is set
35   	// to {@code true}.
36   	this.tokens = [];
37   
38   	// The index into {@link //tokens} of the current token (next token to
39   	// {@link //consume}). {@link //tokens}{@code [}{@link //p}{@code ]} should
40   	// be
41   	// {@link //LT LT(1)}.
42   	//
43   	// <p>This field is set to -1 when the stream is first constructed or when
44   	// {@link //setTokenSource} is called, indicating that the first token has
45   	// not yet been fetched from the token source. For additional information,
46   	// see the documentation of {@link IntStream} for a description of
47   	// Initializing Methods.</p>
48   	this.index = -1;
49   
50   	// Indicates whether the {@link Token//EOF} token has been fetched from
51   	// {@link //tokenSource} and added to {@link //tokens}. This field improves
52   	// performance for the following cases:
53   	//
54   	// <ul>
55   	// <li>{@link //consume}: The lookahead check in {@link //consume} to
56   	// prevent
57   	// consuming the EOF symbol is optimized by checking the values of
58   	// {@link //fetchedEOF} and {@link //p} instead of calling {@link
59   	// //LA}.</li>
60   	// <li>{@link //fetch}: The check to prevent adding multiple EOF symbols
61   	// into
62   	// {@link //tokens} is trivial with this field.</li>
63   	// <ul>
64   	this.fetchedEOF = false;
65   	return this;
66   }

76   	// no resources to release

83   BufferedTokenStream.prototype.seek = function(index) {
84   	this.lazyInit();
85   	this.index = this.adjustSeekIndex(index);

88   BufferedTokenStream.prototype.get = function(index) {
89   	this.lazyInit();
90   	return this.tokens[index];
91   };

93   BufferedTokenStream.prototype.consume = function() {
94   	var skipEofCheck = false;
95   	if (this.index >= 0) {
96   		if (this.fetchedEOF) {
97   			// the last token in tokens is EOF. skip check if p indexes any
98   			// fetched token except the last.
99   			skipEofCheck = this.index < this.tokens.length - 1;
100  		} else {
101  			// no EOF token in tokens. skip check if p indexes a fetched token.
102  			skipEofCheck = this.index < this.tokens.length;
103  		}
104  	} else {
105  		// not yet initialized
106  		skipEofCheck = false;
107  	}
108  	if (!skipEofCheck && this.LA(1) === Token.EOF) {
109  		throw "cannot consume EOF";
110  	}
111  	if (this.sync(this.index + 1)) {
112  		this.index = this.adjustSeekIndex(this.index + 1);
113  	}
114  };
115  
116  // Make sure index {@code i} in tokens has a token.
117  //
118  // @return {@code true} if a token is located at index {@code i}, otherwise
119  // {@code false}.
120  // @see //get(int i)
121  // /
122  BufferedTokenStream.prototype.sync = function(i) {
123  	var n = i - this.tokens.length + 1; // how many more elements we need?
124  	if (n > 0) {
125  		var fetched = this.fetch(n);
126  		return fetched >= n;
127  	}
128  	return true;
129  };
130  
131  // Add {@code n} elements to buffer.
132  //
133  // @return The actual number of elements added to the buffer.
134  // /
135  BufferedTokenStream.prototype.fetch = function(n) {
136  	if (this.fetchedEOF) {
137  		return 0;
138  	}
139  	for (var i = 0; i < n; i++) {
140  		var t = this.tokenSource.nextToken();
141  		t.tokenIndex = this.tokens.length;
142  		this.tokens.push(t);
143  		if (t.type === Token.EOF) {
144  			this.fetchedEOF = true;
145  			return i + 1;
146  		}
147  	}
148  	return n;
149  };

151  // Get all tokens from start..stop inclusively///
152  BufferedTokenStream.prototype.getTokens = function(start, stop, types) {
153  	if (types === undefined) {
154  		types = null;
155  	}
156  	if (start < 0 || stop < 0) {
157  		return null;
158  	}
159  	this.lazyInit();
160  	var subset = [];
161  	if (stop >= this.tokens.length) {
162  		stop = this.tokens.length - 1;
163  	}
164  	for (var i = start; i < stop; i++) {
165  		var t = this.tokens[i];
166  		if (t.type === Token.EOF) {
167  			break;
168  		}
169  		if (types === null || types.contains(t.type)) {
170  			subset.push(t);
171  		}
172  	}
173  	return subset;
174  };
175  
176  BufferedTokenStream.prototype.LA = function(i) {
177  	return this.LT(i).type;
178  };
179  
180  BufferedTokenStream.prototype.LB = function(k) {
181  	if (this.index - k < 0) {
182  		return null;
183  	}
184  	return this.tokens[this.index - k];
185  };
186  
187  BufferedTokenStream.prototype.LT = function(k) {
188  	this.lazyInit();
189  	if (k === 0) {
190  		return null;
191  	}
192  	if (k < 0) {
193  		return this.LB(-k);
194  	}
195  	var i = this.index + k - 1;
196  	this.sync(i);
197  	if (i >= this.tokens.length) { // return EOF token
198  		// EOF must be last token
199  		return this.tokens[this.tokens.length - 1];
200  	}
201  	return this.tokens[i];
202  };

204  // Allowed derived classes to modify the behavior of operations which change
205  // the current stream position by adjusting the target token index of a seek
206  // operation. The default implementation simply returns {@code i}. If an
207  // exception is thrown in this method, the current stream index should not be
208  // changed.
209  //
210  // <p>For example, {@link CommonTokenStream} overrides this method to ensure
211  // that
212  // the seek target is always an on-channel token.</p>
213  //
214  // @param i The target token index.
215  // @return The adjusted target token index.
216  
217  BufferedTokenStream.prototype.adjustSeekIndex = function(i) {
218  	return i;
219  };
220  
221  BufferedTokenStream.prototype.lazyInit = function() {
222  	if (this.index === -1) {
223  		this.setup();
224  	}
225  };
226  
227  BufferedTokenStream.prototype.setup = function() {
228  	this.sync(0);
229  	this.index = this.adjustSeekIndex(0);
230  };

232  // Reset this token stream by setting its token source.///
233  BufferedTokenStream.prototype.setTokenSource = function(tokenSource) {
234  	this.tokenSource = tokenSource;
235  	this.tokens = [];
236  	this.index = -1;
237  	this.fetchedEOF = false;
238  };

241  // Given a starting index, return the index of the next token on channel.
242  // Return i if tokens[i] is on channel. Return -1 if there are no tokens
243  // on channel between i and EOF.
244  // /
245  BufferedTokenStream.prototype.nextTokenOnChannel = function(i, channel) {
246  	this.sync(i);
247  	if (i >= this.tokens.length) {
248  		return -1;
249  	}
250  	var token = this.tokens[i];
251  	while (token.channel !== this.channel) {
252  		if (token.type === Token.EOF) {
253  			return -1;
254  		}
255  		i += 1;
256  		this.sync(i);
257  		token = this.tokens[i];
258  	}
259  	return i;
260  };
261  
262  // Given a starting index, return the index of the previous token on channel.
263  // Return i if tokens[i] is on channel. Return -1 if there are no tokens
264  // on channel between i and 0.
265  BufferedTokenStream.prototype.previousTokenOnChannel = function(i, channel) {
266  	while (i >= 0 && this.tokens[i].channel !== channel) {
267  		i -= 1;
268  	}
269  	return i;
270  };

272  // Collect all tokens on specified channel to the right of
273  // the current token up until we see a token on DEFAULT_TOKEN_CHANNEL or
274  // EOF. If channel is -1, find any non default channel token.
275  BufferedTokenStream.prototype.getHiddenTokensToRight = function(tokenIndex,
276  		channel) {
277  	if (channel === undefined) {
278  		channel = -1;
279  	}
280  	this.lazyInit();
281  	if (tokenIndex < 0 || tokenIndex >= this.tokens.length) {
282  		throw "" + tokenIndex + " not in 0.." + this.tokens.length - 1;
283  	}
284  	var nextOnChannel = this.nextTokenOnChannel(tokenIndex + 1, Lexer.DEFAULT_TOKEN_CHANNEL);
285  	var from_ = tokenIndex + 1;
286  	// if none onchannel to right, nextOnChannel=-1 so set to = last token
287  	var to = nextOnChannel === -1 ? this.tokens.length - 1 : nextOnChannel;
288  	return this.filterForChannel(from_, to, channel);
289  };
290  
291  // Collect all tokens on specified channel to the left of
292  // the current token up until we see a token on DEFAULT_TOKEN_CHANNEL.
293  // If channel is -1, find any non default channel token.
294  BufferedTokenStream.prototype.getHiddenTokensToLeft = function(tokenIndex,
295  		channel) {
296  	if (channel === undefined) {
297  		channel = -1;
298  	}
299  	this.lazyInit();
300  	if (tokenIndex < 0 || tokenIndex >= this.tokens.length) {
301  		throw "" + tokenIndex + " not in 0.." + this.tokens.length - 1;
302  	}
303  	var prevOnChannel = this.previousTokenOnChannel(tokenIndex - 1, Lexer.DEFAULT_TOKEN_CHANNEL);
304  	if (prevOnChannel === tokenIndex - 1) {
305  		return null;
306  	}
307  	// if none on channel to left, prevOnChannel=-1 then from=0
308  	var from_ = prevOnChannel + 1;
309  	var to = tokenIndex - 1;
310  	return this.filterForChannel(from_, to, channel);
311  };
312  
313  BufferedTokenStream.prototype.filterForChannel = function(left, right, channel) {
314  	var hidden = [];
315  	for (var i = left; i < right + 1; i++) {
316  		var t = this.tokens[i];
317  		if (channel === -1) {
318  			if (t.channel !== Lexer.DEFAULT_TOKEN_CHANNEL) {
319  				hidden.push(t);
320  			}
321  		} else if (t.channel === channel) {
322  			hidden.push(t);
323  		}
324  	}
325  	if (hidden.length === 0) {
326  		return null;
327  	}
328  	return hidden;
329  };

335  // Get the text of all tokens in this buffer.///
336  BufferedTokenStream.prototype.getText = function(interval) {
337  	this.lazyInit();
338  	this.fill();
339  	if (interval === undefined || interval === null) {
340  		interval = new Interval(0, this.tokens.length - 1);
341  	}
342  	var start = interval.start;
343  	if (start instanceof Token) {
344  		start = start.tokenIndex;
345  	}
346  	var stop = interval.stop;
347  	if (stop instanceof Token) {
348  		stop = stop.tokenIndex;
349  	}
350  	if (start === null || stop === null || start < 0 || stop < 0) {
351  		return "";
352  	}
353  	if (stop >= this.tokens.length) {
354  		stop = this.tokens.length - 1;
355  	}
356  	var s = "";
357  	for (var i = start; i < stop + 1; i++) {
358  		var t = this.tokens[i];
359  		if (t.type === Token.EOF) {
360  			break;
361  		}
362  		s = s + t.text;
363  	}
364  	return s;
365  };
366  
367  // Get all tokens from lexer until EOF///
368  BufferedTokenStream.prototype.fill = function() {
369  	this.lazyInit();
370  	while (this.fetch(1000) === 1000) {
371  		continue;
372  	}
373  };
