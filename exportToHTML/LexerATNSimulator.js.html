8    // When we hit an accept state in either the DFA or the ATN, we
9    //  have to notify the character stream to start buffering characters
10   //  via {@link IntStream//mark} and record the current state. The current sim state
11   //  includes the current index into the input, the current line,
12   //  and current character position in that line. Note that the Lexer is
13   //  tracking the starting line and characterization of the token. These
14   //  variables track the "state" of the simulator when it hits an accept state.
15   //
16   //  <p>We track these variables separately for the DFA and ATN simulation
17   //  because the DFA simulation often has to fail over to the ATN
18   //  simulation. If the ATN simulation fails, we need the DFA to fall
19   //  back to its previously accepted state, if any. If the ATN succeeds,
20   //  then the ATN does the accept and the DFA simulator that invoked it
21   //  can simply return the predicted token type.</p>
22   ///
23

39   function resetSimState(sim) {
40   	sim.index = -1;
41   	sim.line = 0;
42   	sim.column = -1;
43   	sim.dfaState = null;
44   }
45   
46   function SimState() {
47   	resetSimState(this);
48   	return this;
49   }

51   SimState.prototype.reset = function() {
52   	resetSimState(this);

55   function LexerATNSimulator(recog, atn, decisionToDFA, sharedContextCache) {
56   	ATNSimulator.call(this, atn, sharedContextCache);
57   	this.decisionToDFA = decisionToDFA;
58   	this.recog = recog;
59   	// The current token's starting index into the character stream.
60   	// Shared across DFA to ATN simulation in case the ATN fails and the
61   	// DFA did not have a previous accept state. In this case, we use the
62   	// ATN-generated exception object.
63   	this.startIndex = -1;
64   	// line number 1..n within the input///
65   	this.line = 1;
66   	// The index of the character relative to the beginning of the line
67   	// 0..n-1///
68   	this.column = 0;
69   	this.mode = Lexer.DEFAULT_MODE;
70   	// Used during DFA/ATN exec to record the most recent accept configuration
71   	// info
72   	this.prevAccept = new SimState();
73   	// done
74   	return this;
75   }

95   LexerATNSimulator.prototype.match = function(input, mode) {
96   	this.match_calls += 1;
97   	this.mode = mode;
98   	var mark = input.mark();
99   	try {
100  		this.startIndex = input.index;
101  		this.prevAccept.reset();
102  		var dfa = this.decisionToDFA[mode];
103  		if (dfa.s0 === null) {
104  			return this.matchATN(input);
105  		} else {
106  			return this.execATN(input, dfa.s0);
107  		}
108  	} finally {
109  		input.release(mark);
110  	}
111  };

113  LexerATNSimulator.prototype.reset = function() {
114  	this.prevAccept.reset();
115  	this.startIndex = -1;
116  	this.line = 1;
117  	this.column = 0;
118  	this.mode = Lexer.DEFAULT_MODE;
119  };

121  LexerATNSimulator.prototype.matchATN = function(input) {
122  	var startState = this.atn.modeToStartState[this.mode];
123  
124  	if (LexerATNSimulator.debug) {
125  		console.log("matchATN mode " + this.mode + " start: " + startState);
126  	}
127  	var old_mode = this.mode;
128  	var s0_closure = this.computeStartState(input, startState);
129  	var suppressEdge = s0_closure.hasSemanticContext;
130  	s0_closure.hasSemanticContext = false;
131  
132  	var next = this.addDFAState(s0_closure);
133  	if (!suppressEdge) {
134  		this.decisionToDFA[this.mode].s0 = next;
135  	}
136  
137  	var predict = this.execATN(input, next);
138  
139  	if (LexerATNSimulator.debug) {
140  		console.log("DFA after matchATN: " + this.decisionToDFA[old_mode].toLexerString());
141  	}
142  	return predict;
143  };

145  LexerATNSimulator.prototype.execATN = function(input, ds0) {
146  	if (LexerATNSimulator.debug) {
147  		console.log("start state closure=" + ds0.configs);
148  	}
149  	if (ds0.isAcceptState) {
150  		// allow zero-length tokens
151  		this.captureSimState(this.prevAccept, input, ds0);
152  	}
153  	var t = input.LA(1);
154  	var s = ds0; // s is current/from DFA state
155  
156  	while (true) { // while more work
157  		if (LexerATNSimulator.debug) {
158  			console.log("execATN loop starting closure: " + s.configs);
159  		}
160  
161  		// As we move src->trg, src->trg, we keep track of the previous trg to
162  		// avoid looking up the DFA state again, which is expensive.
163  		// If the previous target was already part of the DFA, we might
164  		// be able to avoid doing a reach operation upon t. If s!=null,
165  		// it means that semantic predicates didn't prevent us from
166  		// creating a DFA state. Once we know s!=null, we check to see if
167  		// the DFA state has an edge already for t. If so, we can just reuse
168  		// it's configuration set; there's no point in re-computing it.
169  		// This is kind of like doing DFA simulation within the ATN
170  		// simulation because DFA simulation is really just a way to avoid
171  		// computing reach/closure sets. Technically, once we know that
172  		// we have a previously added DFA state, we could jump over to
173  		// the DFA simulator. But, that would mean popping back and forth
174  		// a lot and making things more complicated algorithmically.
175  		// This optimization makes a lot of sense for loops within DFA.
176  		// A character will take us back to an existing DFA state
177  		// that already has lots of edges out of it. e.g., .* in comments.
178  		// print("Target for:" + str(s) + " and:" + str(t))
179  		var target = this.getExistingTargetState(s, t);
180  		// print("Existing:" + str(target))
181  		if (target === null) {
182  			target = this.computeTargetState(input, s, t);
183  			// print("Computed:" + str(target))
184  		}
185  		if (target === ATNSimulator.ERROR) {
186  			break;
187  		}
188  		// If this is a consumable input element, make sure to consume before
189  		// capturing the accept state so the input index, line, and char
190  		// position accurately reflect the state of the interpreter at the
191  		// end of the token.
192  		if (t !== Token.EOF) {
193  			this.consume(input);
194  		}
195  		if (target.isAcceptState) {
196  			this.captureSimState(this.prevAccept, input, target);
197  			if (t === Token.EOF) {
198  				break;
199  			}
200  		}
201  		t = input.LA(1);
202  		s = target; // flip; current DFA target becomes new src/from state
203  	}
204  	return this.failOrAccept(this.prevAccept, input, s.configs, t);
205  };

207  // Get an existing target state for an edge in the DFA. If the target state
208  // for the edge has not yet been computed or is otherwise not available,
209  // this method returns {@code null}.
210  //
211  // @param s The current DFA state
212  // @param t The next input symbol
213  // @return The existing target DFA state for the given input symbol
214  // {@code t}, or {@code null} if the target state for this edge is not
215  // already cached
216  LexerATNSimulator.prototype.getExistingTargetState = function(s, t) {
217  	if (s.edges === null || t < LexerATNSimulator.MIN_DFA_EDGE || t > LexerATNSimulator.MAX_DFA_EDGE) {
218  		return null;
219  	}
220  
221  	var target = s.edges[t - LexerATNSimulator.MIN_DFA_EDGE];
222  	if(target===undefined) {
223  		target = null;
224  	}
225  	if (LexerATNSimulator.debug && target !== null) {
226  		console.log("reuse state " + s.stateNumber + " edge to " + target.stateNumber);
227  	}
228  	return target;
229  };

231  // Compute a target state for an edge in the DFA, and attempt to add the
232  // computed state and corresponding edge to the DFA.
233  //
234  // @param input The input stream
235  // @param s The current DFA state
236  // @param t The next input symbol
237  //
238  // @return The computed target DFA state for the given input symbol
239  // {@code t}. If {@code t} does not lead to a valid DFA state, this method
240  // returns {@link //ERROR}.
241  LexerATNSimulator.prototype.computeTargetState = function(input, s, t) {
242  	var reach = new OrderedATNConfigSet();
243  	// if we don't find an existing DFA state
244  	// Fill reach starting from closure, following t transitions
245  	this.getReachableConfigSet(input, s.configs, reach, t);
246  
247  	if (reach.items.length === 0) { // we got nowhere on t from s
248  		if (!reach.hasSemanticContext) {
249  			// we got nowhere on t, don't throw out this knowledge; it'd
250  			// cause a failover from DFA later.
251  			this.addDFAEdge(s, t, ATNSimulator.ERROR);
252  		}
253  		// stop when we can't match any more char
254  		return ATNSimulator.ERROR;
255  	}
256  	// Add an edge from s to target DFA found/created for reach
257  	return this.addDFAEdge(s, t, null, reach);
258  };

260  LexerATNSimulator.prototype.failOrAccept = function(prevAccept, input, reach, t) {
261  	if (this.prevAccept.dfaState !== null) {
262  		var lexerActionExecutor = prevAccept.dfaState.lexerActionExecutor;
263  		this.accept(input, lexerActionExecutor, this.startIndex,
264  				prevAccept.index, prevAccept.line, prevAccept.column);
265  		return prevAccept.dfaState.prediction;
266  	} else {
267  		// if no accept and EOF is first char, return EOF
268  		if (t === Token.EOF && input.index === this.startIndex) {
269  			return Token.EOF;
270  		}
271  		throw new LexerNoViableAltException(this.recog, input, this.startIndex, reach);
272  	}
273  };

275  // Given a starting configuration set, figure out all ATN configurations
276  // we can reach upon input {@code t}. Parameter {@code reach} is a return
277  // parameter.
278  LexerATNSimulator.prototype.getReachableConfigSet = function(input, closure,
279  		reach, t) {
280  	// this is used to skip processing for configs which have a lower priority
281  	// than a config that already reached an accept state for the same rule
282  	var skipAlt = ATN.INVALID_ALT_NUMBER;
283  	for (var i = 0; i < closure.items.length; i++) {
284  		var cfg = closure.items[i];
285  		var currentAltReachedAcceptState = (cfg.alt === skipAlt);
286  		if (currentAltReachedAcceptState && cfg.passedThroughNonGreedyDecision) {
287  			continue;
288  		}
289  		if (LexerATNSimulator.debug) {
290  			console.log("testing %s at %s\n", this.getTokenName(t), cfg
291  					.toString(this.recog, true));
292  		}
293  		for (var j = 0; j < cfg.state.transitions.length; j++) {
294  			var trans = cfg.state.transitions[j]; // for each transition
295  			var target = this.getReachableTarget(trans, t);
296  			if (target !== null) {
297  				var lexerActionExecutor = cfg.lexerActionExecutor;
298  				if (lexerActionExecutor !== null) {
299  					lexerActionExecutor = lexerActionExecutor.fixOffsetBeforeMatch(input.index - this.startIndex);
300  				}
301  				var treatEofAsEpsilon = (t === Token.EOF);
302  				var config = new LexerATNConfig({state:target, lexerActionExecutor:lexerActionExecutor}, cfg);
303  				if (this.closure(input, config, reach,
304  						currentAltReachedAcceptState, true, treatEofAsEpsilon)) {
305  					// any remaining configs for this alt have a lower priority
306  					// than the one that just reached an accept state.
307  					skipAlt = cfg.alt;
308  				}
309  			}
310  		}
311  	}
312  };

314  LexerATNSimulator.prototype.accept = function(input, lexerActionExecutor,
315  		startIndex, index, line, charPos) {
316  	if (LexerATNSimulator.debug) {
317  		console.log("ACTION %s\n", lexerActionExecutor);
318  	}
319  	// seek to after last char in token
320  	input.seek(index);
321  	this.line = line;
322  	this.column = charPos;
323  	if (lexerActionExecutor !== null && this.recog !== null) {
324  		lexerActionExecutor.execute(this.recog, input, startIndex);
325  	}
326  };

328  LexerATNSimulator.prototype.getReachableTarget = function(trans, t) {
329  	if (trans.matches(t, 0, Lexer.MAX_CHAR_VALUE)) {
330  		return trans.target;
331  	} else {
332  		return null;
333  	}
334  };

336  LexerATNSimulator.prototype.computeStartState = function(input, p) {
337  	var initialContext = PredictionContext.EMPTY;
338  	var configs = new OrderedATNConfigSet();
339  	for (var i = 0; i < p.transitions.length; i++) {
340  		var target = p.transitions[i].target;
341          var cfg = new LexerATNConfig({state:target, alt:i+1, context:initialContext}, null);
342  		this.closure(input, cfg, configs, false, false, false);
343  	}
344  	return configs;
345  };

347  // Since the alternatives within any lexer decision are ordered by
348  // preference, this method stops pursuing the closure as soon as an accept
349  // state is reached. After the first accept state is reached by depth-first
350  // search from {@code config}, all other (potentially reachable) states for
351  // this rule would have a lower priority.
352  //
353  // @return {@code true} if an accept state is reached, otherwise
354  // {@code false}.
355  LexerATNSimulator.prototype.closure = function(input, config, configs,
356  		currentAltReachedAcceptState, speculative, treatEofAsEpsilon) {
357  	var cfg = null;
358  	if (LexerATNSimulator.debug) {
359  		console.log("closure(" + config.toString(this.recog, true) + ")");
360  	}
361  	if (config.state instanceof RuleStopState) {
362  		if (LexerATNSimulator.debug) {
363  			if (this.recog !== null) {
364  				console.log("closure at %s rule stop %s\n", this.recog.ruleNames[config.state.ruleIndex], config);
365  			} else {
366  				console.log("closure at rule stop %s\n", config);
367  			}
368  		}
369  		if (config.context === null || config.context.hasEmptyPath()) {
370  			if (config.context === null || config.context.isEmpty()) {
371  				configs.add(config);
372  				return true;
373  			} else {
374  				configs.add(new LexerATNConfig({ state:config.state, context:PredictionContext.EMPTY}, config));
375  				currentAltReachedAcceptState = true;
376  			}
377  		}
378  		if (config.context !== null && !config.context.isEmpty()) {
379  			for (var i = 0; i < config.context.length; i++) {
380  				if (config.context.getReturnState(i) !== PredictionContext.EMPTY_RETURN_STATE) {
381  					var newContext = config.context.getParent(i); // "pop" return state
382  					var returnState = this.atn.states[config.context.getReturnState(i)];
383  					cfg = new LexerATNConfig({ state:returnState, context:newContext }, config);
384  					currentAltReachedAcceptState = this.closure(input, cfg,
385  							configs, currentAltReachedAcceptState, speculative,
386  							treatEofAsEpsilon);
387  				}
388  			}
389  		}
390  		return currentAltReachedAcceptState;
391  	}
392  	// optimization
393  	if (!config.state.epsilonOnlyTransitions) {
394  		if (!currentAltReachedAcceptState || !config.passedThroughNonGreedyDecision) {
395  			configs.add(config);
396  		}
397  	}
398  	for (var j = 0; j < config.state.transitions.length; j++) {
399  		var trans = config.state.transitions[j];
400  		cfg = this.getEpsilonTarget(input, config, trans, configs, speculative, treatEofAsEpsilon);
401  		if (cfg !== null) {
402  			currentAltReachedAcceptState = this.closure(input, cfg, configs,
403  					currentAltReachedAcceptState, speculative, treatEofAsEpsilon);
404  		}
405  	}
406  	return currentAltReachedAcceptState;
407  };

409  // side-effect: can alter configs.hasSemanticContext
410  LexerATNSimulator.prototype.getEpsilonTarget = function(input, config, trans,
411  		configs, speculative, treatEofAsEpsilon) {
412  	var cfg = null;
413  	if (trans.serializationType === Transition.RULE) {
414  		var newContext = SingletonPredictionContext.create(config.context, trans.followState.stateNumber);
415  		cfg = new LexerATNConfig( { state:trans.target, context:newContext}, config);
416  	} else if (trans.serializationType === Transition.PRECEDENCE) {
417  		throw "Precedence predicates are not supported in lexers.";
418  	} else if (trans.serializationType === Transition.PREDICATE) {
419  		// Track traversing semantic predicates. If we traverse,
420  		// we cannot add a DFA state for this "reach" computation
421  		// because the DFA would not test the predicate again in the
422  		// future. Rather than creating collections of semantic predicates
423  		// like v3 and testing them on prediction, v4 will test them on the
424  		// fly all the time using the ATN not the DFA. This is slower but
425  		// semantically it's not used that often. One of the key elements to
426  		// this predicate mechanism is not adding DFA states that see
427  		// predicates immediately afterwards in the ATN. For example,
428  
429  		// a : ID {p1}? | ID {p2}? ;
430  
431  		// should create the start state for rule 'a' (to save start state
432  		// competition), but should not create target of ID state. The
433  		// collection of ATN states the following ID references includes
434  		// states reached by traversing predicates. Since this is when we
435  		// test them, we cannot cash the DFA state target of ID.
436  
437  		if (LexerATNSimulator.debug) {
438  			console.log("EVAL rule " + trans.ruleIndex + ":" + trans.predIndex);
439  		}
440  		configs.hasSemanticContext = true;
441  		if (this.evaluatePredicate(input, trans.ruleIndex, trans.predIndex, speculative)) {
442  			cfg = new LexerATNConfig({ state:trans.target}, config);
443  		}
444  	} else if (trans.serializationType === Transition.ACTION) {
445  		if (config.context === null || config.context.hasEmptyPath()) {
446  			// execute actions anywhere in the start rule for a token.
447  			//
448  			// TODO: if the entry rule is invoked recursively, some
449  			// actions may be executed during the recursive call. The
450  			// problem can appear when hasEmptyPath() is true but
451  			// isEmpty() is false. In this case, the config needs to be
452  			// split into two contexts - one with just the empty path
453  			// and another with everything but the empty path.
454  			// Unfortunately, the current algorithm does not allow
455  			// getEpsilonTarget to return two configurations, so
456  			// additional modifications are needed before we can support
457  			// the split operation.
458  			var lexerActionExecutor = LexerActionExecutor.append(config.lexerActionExecutor,
459  					this.atn.lexerActions[trans.actionIndex]);
460  			cfg = new LexerATNConfig({ state:trans.target, lexerActionExecutor:lexerActionExecutor }, config);
461  		} else {
462  			// ignore actions in referenced rules
463  			cfg = new LexerATNConfig( { state:trans.target}, config);
464  		}
465  	} else if (trans.serializationType === Transition.EPSILON) {
466  		cfg = new LexerATNConfig({ state:trans.target}, config);
467  	} else if (trans.serializationType === Transition.ATOM ||
468  				trans.serializationType === Transition.RANGE ||
469  				trans.serializationType === Transition.SET) {
470  		if (treatEofAsEpsilon) {
471  			if (trans.matches(Token.EOF, 0, Lexer.MAX_CHAR_VALUE)) {
472  				cfg = new LexerATNConfig( { state:trans.target }, config);
473  			}
474  		}
475  	}
476  	return cfg;
477  };

479  // Evaluate a predicate specified in the lexer.
480  //
481  // <p>If {@code speculative} is {@code true}, this method was called before
482  // {@link //consume} for the matched character. This method should call
483  // {@link //consume} before evaluating the predicate to ensure position
484  // sensitive values, including {@link Lexer//getText}, {@link Lexer//getLine},
485  // and {@link Lexer//getcolumn}, properly reflect the current
486  // lexer state. This method should restore {@code input} and the simulator
487  // to the original state before returning (i.e. undo the actions made by the
488  // call to {@link //consume}.</p>
489  //
490  // @param input The input stream.
491  // @param ruleIndex The rule containing the predicate.
492  // @param predIndex The index of the predicate within the rule.
493  // @param speculative {@code true} if the current index in {@code input} is
494  // one character before the predicate's location.
495  //
496  // @return {@code true} if the specified predicate evaluates to
497  // {@code true}.
498  // /
499  LexerATNSimulator.prototype.evaluatePredicate = function(input, ruleIndex,
500  		predIndex, speculative) {
501  	// assume true if no recognizer was provided
502  	if (this.recog === null) {
503  		return true;
504  	}
505  	if (!speculative) {
506  		return this.recog.sempred(null, ruleIndex, predIndex);
507  	}
508  	var savedcolumn = this.column;
509  	var savedLine = this.line;
510  	var index = input.index;
511  	var marker = input.mark();
512  	try {
513  		this.consume(input);
514  		return this.recog.sempred(null, ruleIndex, predIndex);
515  	} finally {
516  		this.column = savedcolumn;
517  		this.line = savedLine;
518  		input.seek(index);
519  		input.release(marker);
520  	}
521  };

523  LexerATNSimulator.prototype.captureSimState = function(settings, input, dfaState) {
524  	settings.index = input.index;
525  	settings.line = this.line;
526  	settings.column = this.column;
527  	settings.dfaState = dfaState;
528  };

530  LexerATNSimulator.prototype.addDFAEdge = function(from_, tk, to, cfgs) {
531  	if (to === undefined) {
532  		to = null;
533  	}
534  	if (cfgs === undefined) {
535  		cfgs = null;
536  	}
537  	if (to === null && cfgs !== null) {
538  		// leading to this call, ATNConfigSet.hasSemanticContext is used as a
539  		// marker indicating dynamic predicate evaluation makes this edge
540  		// dependent on the specific input sequence, so the static edge in the
541  		// DFA should be omitted. The target DFAState is still created since
542  		// execATN has the ability to resynchronize with the DFA state cache
543  		// following the predicate evaluation step.
544  		//
545  		// TJP notes: next time through the DFA, we see a pred again and eval.
546  		// If that gets us to a previously created (but dangling) DFA
547  		// state, we can continue in pure DFA mode from there.
548  		// /
549  		var suppressEdge = cfgs.hasSemanticContext;
550  		cfgs.hasSemanticContext = false;
551  
552  		to = this.addDFAState(cfgs);
553  
554  		if (suppressEdge) {
555  			return to;
556  		}
557  	}
558  	// add the edge
559  	if (tk < LexerATNSimulator.MIN_DFA_EDGE || tk > LexerATNSimulator.MAX_DFA_EDGE) {
560  		// Only track edges within the DFA bounds
561  		return to;
562  	}
563  	if (LexerATNSimulator.debug) {
564  		console.log("EDGE " + from_ + " -> " + to + " upon " + tk);
565  	}
566  	if (from_.edges === null) {
567  		// make room for tokens 1..n and -1 masquerading as index 0
568  		from_.edges = [];
569  	}
570  	from_.edges[tk - LexerATNSimulator.MIN_DFA_EDGE] = to; // connect
571  
572  	return to;
573  };
574  
575  // Add a new DFA state if there isn't one with this set of
576  // configurations already. This method also detects the first
577  // configuration containing an ATN rule stop state. Later, when
578  // traversing the DFA, we will know which rule to accept.
579  LexerATNSimulator.prototype.addDFAState = function(configs) {
580  	var proposed = new DFAState(null, configs);
581  	var firstConfigWithRuleStopState = null;
582  	for (var i = 0; i < configs.items.length; i++) {
583  		var cfg = configs.items[i];
584  		if (cfg.state instanceof RuleStopState) {
585  			firstConfigWithRuleStopState = cfg;
586  			break;
587  		}
588  	}
589  	if (firstConfigWithRuleStopState !== null) {
590  		proposed.isAcceptState = true;
591  		proposed.lexerActionExecutor = firstConfigWithRuleStopState.lexerActionExecutor;
592  		proposed.prediction = this.atn.ruleToTokenType[firstConfigWithRuleStopState.state.ruleIndex];
593  	}
594  	var dfa = this.decisionToDFA[this.mode];
595  	var existing = dfa.states.get(proposed);
596  	if (existing!==null) {
597  		return existing;
598  	}
599  	var newState = proposed;
600  	newState.stateNumber = dfa.states.length;
601  	configs.setReadonly(true);
602  	newState.configs = configs;
603  	dfa.states.add(newState);
604  	return newState;
605  };
606  
607  LexerATNSimulator.prototype.getDFA = function(mode) {
608  	return this.decisionToDFA[mode];
609  };
610  
611  // Get the text matched so far for the current token.
612  LexerATNSimulator.prototype.getText = function(input) {
613  	// index is first lookahead char, don't include.
614  	return input.getText(this.startIndex, input.index - 1);
615  };
616  
617  LexerATNSimulator.prototype.consume = function(input) {
618  	var curChar = input.LA(1);
619  	if (curChar === "\n".charCodeAt(0)) {
620  		this.line += 1;
621  		this.column = 0;
622  	} else {
623  		this.column += 1;
624  	}
625  	input.consume();
626  };
627  
628  LexerATNSimulator.prototype.getTokenName = function(tt) {
629  	if (tt === -1) {
630  		return "EOF";
631  	} else {
632  		return "'" + String.fromCharCode(tt) + "'";
633  	}
634  };
