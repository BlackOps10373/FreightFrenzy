38   var addClass = function(element, classToAdd) {
39       element.className += (" " + classToAdd);
40   }
41   
42   var hasClass = function(element, classToTest) {
43       var expr = '(?:^|\\s)' + classToTest + '(?!\\S)';
44       return element.className.match(new RegExp(expr, ''));
45   }
46   
47   var ensureClass = function(element, classToAdd) {
48       if (!hasClass(element, classToAdd)) {
49           addClass(element, classToAdd);
50       }
51   }
52   
53   var removeClass = function(element, classToRemove) {
54       // https://stackoverflow.com/questions/195951/change-an-elements-class-with-javascript
55       var expr = '(?:^|\\s)' + classToRemove + '(?!\\S)';
56       element.className = element.className.replace(new RegExp(expr, 'g'), '');
57   }

63   // Scales the font-size of an element over what it otherwise would be. Is idempotent.
64   var scaleFontSize = function(element, fraction) {
65       // originalSize will always be in 'px'. I think.
66       var originalSize = element.getAttribute("data-orig-font-size");
67       if (!originalSize) originalSize = window.getComputedStyle(element, null).getPropertyValue("font-size");
68   
69       var value = parseFloat(originalSize);
70       value = value * fraction;
71       value = value + "px";
72   
73       // console.log("originalSize=" + originalSize + "new size=" + value);
74       element.style.fontSize = value;
75       element.setAttribute("data-orig-font-size", originalSize);
76   }

78   var scaleTextInDocumentBody = function (rcInfo) {
79       // The scaling adaptation is experimental, can likely use polish. You might think we'd
80       // scale the whole view etc, but that seems to complication button layout etc. As I said,
81       // this is kind of experimental.
82   
83       var uaIsAndroid = rcInfo.ftcUserAgentCategory !== "OTHER";
84   
85       var width = window.screen.width;
86       var height = window.screen.height;
87   
88       // These are the values observed given the scaling WebView does
89       var widthZTESpeed = 320;            var heightZTESpeed = 570;       // actual: 540x960
90       var widthAmazonFire7 = 600;         var heightAmazonFire7 = 1024;   // actual: 800x1280
91   
92       var minWidth = widthZTESpeed;       var minScale = 1.0;
93       var maxWidth = widthAmazonFire7;    var maxScale = 1.2;
94   
95       var dw = maxWidth - minWidth;
96       var ds = maxScale - minScale;
97   
98       var scale = 1.0;
99       if (uaIsAndroid) {
100          scale = minScale + (width - minWidth) / dw * ds;
101          if (scale < minScale) scale = minScale;
102          if (scale > maxScale) scale = maxScale;
103      }
104  
105      console.log("width=" + width + " height=" + height + " scale=" + scale);
106      scaleFontSize(document.getElementsByTagName("BODY")[0], scale);
107  }

144  // synchronously loads an array of scripts by name, calling the continuation when all are complete
145  var syncLoadScripts = function (scripts, continuation) {
146      var scriptToLoad = scripts.shift();
147      var script = document.createElement('script');
148      script.type = 'text/javascript';
149      script.src = scriptToLoad;
150      script.onload = function () {
151          if (0 == scripts.length) {
152              console.log("all sync scripts loaded");
153              continuation();
154          } else {
155              syncLoadScripts(scripts, continuation);
156          }
157      };
158      document.head.appendChild(script);
159  }
160  
161  // asynchronously loads an array of scripts by name, calling the continuation when all are complete
162  var asyncLoadScripts = function (scripts, continuation) {
163      var scriptsLoaded = 0;
164      for (var i = 0; i < scripts.length; i++) {
165          var scriptToLoad = scripts[i];
166          var script = document.createElement('script');
167          script.type = 'text/javascript';
168          script.src = scriptToLoad;
169          script.onload = function () {
170              scriptsLoaded++;
171              if (scriptsLoaded == scripts.length) {
172                  console.log("all async scripts loaded");
173                  continuation();
174              }
175          };
176          document.head.appendChild(script);
177      }
178  }

184  var simpleGet = function(uri, callback) {
185      var xhr = new XMLHttpRequest();
186      xhr.onreadystatechange = function () {
187          if (xhr.readyState === 4) {
188              if (xhr.status === 200) {
189                  // console.log("uri=" + uri + " json=" + xhr.responseText);
190                  if(callback) callback(xhr.responseText);
191              }
192              else {
193                  if(callback) callback(null);
194              }
195          }
196      };
197      xhr.open("GET", uri);
198      xhr.send();
199  }

201  var loadJsonObject = function(uri, callback) {
202      simpleGet(uri, function (responseText) {
203          if (responseText) {
204              callback(JSON.parse(responseText));
205          } else {
206              callback(null);
207          }
208      });
209  };

228  var uploadFile = function (localFile /*a File*/, url, force, success, failure) {
229      var xhr = new XMLHttpRequest();
230      var lastDotIndex = localFile.name.lastIndexOf(".");
231      var base = localFile.name.substring(0, lastDotIndex);
232      var ext = localFile.name.substring(lastDotIndex, localFile.name.length);
233      var fileName = makeAndroidSafeFileName(base) + ext;
234      var uploadData = new FormData();
235  
236      xhr.open("POST", url);
237  
238      xhr.onreadystatechange = function () {
239          if (xhr.readyState === 4) {
240              if (xhr.status === 200) {
241                  success(xhr);
242              } else {
243                  failure(xhr);
244                  // Make sure the response is text and not json before we show it to the user.
245                  if ((xhr.responseType == '' || xhr.responseType == 'text') &&
246                          xhr.responseText && !xhr.responseText.startsWith('{')) {
247                      alert("Failed to upload file: " + localFile.name + "\n\n" + xhr.responseText);
248                  } else {
249                      alert("Failed to upload file: " + localFile.name);
250                  }
251              }
252          }
253      }
254  
255      var uploadAbandonmentEventHandler = function(event) { event.preventDefault(); event.returnValue = "Upload in progress"; }
256  
257      // On Chrome, give the wait cursor a chance to show up, dang it.
258      addClass(document.documentElement, "wait");
259      window.addEventListener('beforeunload', uploadAbandonmentEventHandler);
260      xhr.upload.addEventListener("loadend", function(e) {
261          window.removeEventListener('beforeunload', uploadAbandonmentEventHandler);
262          removeClass(document.documentElement, "wait");
263      });
264      window.setTimeout(function() {
265          uploadData.append("file", localFile, fileName);
266          xhr.send(uploadData);
267      }, 1000);
268  }

288  // The endsWith() method is not supported in IE 11 (and earlier versions). So we roll our own.
289  var endsWith = function (str, suffix) {
290      var cch = suffix.length;
291      if (str.length >= cch) {
292          var test = str.substring(str.length - cch, str.length);
293          return test === suffix;
294      } else {
295          return false;
296      }
297  };
298  
299  // Polyfill the String prototype with the required methods
300  (function () {
301      if (!String.prototype.endsWith) {
302  
303          String.prototype.endsWith = function (suffix) {
304              var cch = suffix.length;
305              if (this.length >= cch) {
306                  var test = this.substring(this.length - cch, this.length);
307                  return test === suffix;
308              } else {
309                  return false;
310              }
311          };
312      }
313  }).call(String.prototype);
