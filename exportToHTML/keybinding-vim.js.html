295    this.$updateMarkers = function(delta) {
296      var isInsert = delta.action == "insert";
297      var start = delta.start;
298      var end = delta.end;
299      var rowShift = (end.row - start.row) * (isInsert ? 1 : -1);
300      var colShift = (end.column - start.column) * (isInsert ? 1 : -1);
301      if (isInsert) end = start;
302  
303      for (var i in this.marks) {
304        var point = this.marks[i];
305        var cmp = Range.comparePoints(point, start);
306        if (cmp < 0) {
307          continue; // delta starts after the range
308        }
309        if (cmp === 0) {
310          if (isInsert) {
311            if (point.bias == 1) {
312              cmp = 1;
313            } else {
314              point.bias = -1;
315              continue;
316            }
317          }
318        }
319        var cmp2 = isInsert ? cmp : Range.comparePoints(point, end);
320        if (cmp2 > 0) {
321          point.row += rowShift;
322          point.column += point.row == end.row ? colShift : 0;
323          continue;
324        }
325        if (!isInsert && cmp2 <= 0) {
326          point.row = start.row;
327          point.column = start.column;
328          if (cmp2 === 0)
329            point.bias = 1;
330        }
331      }
