21       var serverStatusNotOK = function() {
22           var serverStatus = document.getElementById("id_server_status");
23           serverStatus.innerHTML = "Server not OK";
24           removeClass(serverStatus, "text-ok");
25           addClass(serverStatus, "text-error");
26       }
27   
28       var hideWhenAndroid = function(hide) {
29           hideElement(document.getElementById("id_when_android"), hide);
30       }
31   
32       var refreshServerStatus = function() {
33           loadRcInfo(function (rcInfo) {
34               if (rcInfo) {
35                   console.log("refreshSeverStatus(connected)");
36   
37                   scaleTextInDocumentBody(rcInfo);
38   
39                   var uaIsAndroid = rcInfo.ftcUserAgentCategory !== "OTHER";
40   
41                   document.getElementById("id_network_name").innerHTML = rcInfo.networkName;
42                   document.getElementById("id_passphrase").innerHTML = rcInfo.passphrase;
43                   document.getElementById("id_server_url").innerHTML = rcInfo.serverUrl;
44   
45                   hideElement(document.getElementById("id_span_connected"), false);
46                   hideWhenAndroid(!uaIsAndroid);
47   
48                   if (!rcInfo.serverIsAlive) {
49                       serverStatusNotOK();
50                   } else {
51                       var serverStatus = document.getElementById("id_server_status");
52                       var formattedDate = rcInfo.timeServerStarted;
53                       var status = "Server OK (Running since " + formattedDate + ")";
54                       serverStatus.innerHTML = status;
55                       removeClass(serverStatus, "text-error");
56                       addClass(serverStatus, "text-ok");
57                   }
58               } else {
59                   console.log("refreshSeverStatus(disconnected)");
60                   hideElement(document.getElementById("id_span_connected"), true);
61                   hideWhenAndroid(true);
62                   serverStatusNotOK();
63               }
64           });
65       }
66   
67       var isConnected = false;
68   
69       document.onPingResult = function (pingResponse) {
70           var activeConnections = document.getElementById("id_active_connections");
71           if (pingResponse && pingResponse.connectedDevices.length > 0) {
72               let connectedDevices = pingResponse.connectedDevices;
73   
74               if (!isConnected) refreshServerStatus();
75               isConnected = true;
76   
77               // remove all children
78               while (activeConnections.childNodes.length > 0) {
79                   activeConnections.removeChild(activeConnections.childNodes[0]);
80               }
81               var table = document.createElement("TABLE");
82               addClass(table, "active_connections_table");
83               var tbody = document.createElement("TBODY");
84               table.appendChild(tbody);
85               var i;
86               for (i = 0; i < connectedDevices.length; i++) {
87                   var connectedDevice = connectedDevices[i];
88                   var row = document.createElement("TR");
89                   addClass(row, "active_connections_row");
90                   //
91                   var machine = document.createElement("TD");
92                   addClass(machine, "active_connections_machine");
93                   machine.innerHTML = connectedDevice.machineName;
94                   //
95                   var name = document.createElement("TD");
96                   addClass(name, "active_connections_page");
97                   name.innerHTML = connectedDevice.currentPage;
98                   //
99                   row.appendChild(machine);
100                  row.appendChild(name);
101                  tbody.appendChild(row);
102              }
103              activeConnections.appendChild(table);
104          } else {
105              console.log("onPingResult: disconnected");
106  
107              if (isConnected) refreshServerStatus();
108              isConnected = false;
109  
110              activeConnections.innerHTML = "<p class=\"active_connections_none\">&lt;none&gt;</p>";
111          }
112      }
113  
114      refreshServerStatus();
115
