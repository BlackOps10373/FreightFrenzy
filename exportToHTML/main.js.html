38           this.each(function () {
39               var attrs = {};
40               $.each(this.attributes, function (idx, attribute) {
41                   var nodeName = attribute.nodeName;
42                   if (nodeName === 'href' && newType !== 'a') return;
43   
44                   attrs[nodeName] = attribute.value;
45               });
46   
47               $(this).replaceWith($('<' + newType + ' />', attrs).append($(this).contents()));
48           });

58       env.getParameterByName = function getParameterByName(name) {
59           const url = window.location.href;
60           name = name.replace(/[\[\]]/g, "\\$&");
61           const regex = new RegExp("[?&]" + name + "(=([^&#]*)|&|#|$)");
62           var results = regex.exec(url);
63           if (!results) return null;
64           if (!results[2]) return '';
65           return decodeURIComponent(results[2].replace(/\+/g, " "));
66       };

178                  let restToMatch = file;
179                  let currNode = null;
180                  for (let node of tree) {

182                      if (restToMatch.startsWith(node.parentFile + node.file + "/")) {

191                  }

193                  let foundNode = true;

195                      foundNode = false;
196                      for (let node of currNode.nodes) {
197                          if (node.folder && restToMatch.startsWith(node.file + "/")) {
198                              if (restToMatch === node.file + "/") {
199                                  return node;

206                          // this only happens since we sometimes skip directories

208                          } else if (file.startsWith(node.parentFile + node.file)) {
209                              if (file === node.parentFile + node.file) {
210                                  return node;
211                              }
212                              restToMatch = file.substr(node.parentFile.length + node.file.length + 1);
213                              console.log(restToMatch);
214                              currNode = node;
215                              foundNode = true;
216                              break;
217                          } else if (!node.folder && restToMatch.startsWith(node.file + "/")) {
218                              return env.errors.FILE_NOT_DIRECTORY;
219                          } else if (!node.folder && restToMatch === node.file) {
220                              return node;

264                  function stage2Parser(prop, href, fileNamespace, parentNode) {
265                      const resultsFiles = [];
266                      const resultsFolder = [];
267  
268                      function hasExtension(ext) {
269                          return attr.indexOf(ext, attr.length - ext.length) !== -1;
270                      }
271  
272                      var iconForFile = function () {
273                          if (hasExtension('.java')) {
274                              return 'fa fa-file-code-o';
275                          } else if (hasExtension('.zip')
276                              || hasExtension('.jar') || hasExtension('.aar')) {
277                              return 'fa fa-archive';
278                          } else if (hasExtension('.txt') || hasExtension('.md')) {
279                              return 'fa fa-file-text';
280                          } else {
281                              return 'fa fa-question';
282                          }
283                      };
284  
285                      for (var attr in prop) {
286                          if (!prop.hasOwnProperty(attr)) continue; // skip if not a property we should worry about
287  
288                          if (attr === '' || hasExtension('.tmp')) continue;
289  
290                          const location = fileNamespace + '/';
291                          var items = {
292                              text: attr,
293                              file: attr,
294                              folder: false,
295                              icon: iconForFile(),
296                              parentFile: location,
297                              parent: JSON.stringify(parentNode)
298                          };
299  
300                          if (!(Object.keys(prop[attr]).length === 0 && prop[attr].constructor === Object)) { // detect folders
301                              const children = stage2Parser(prop[attr], href, location + attr, items);
302                              // We don't want the default '#' behavior, so literally do nothing
303                              items.href = 'javaScript:void(0);';
304                              items.file = items.text;
305                              // Change item to have a folder icon
306                              items.icon = 'fa fa-folder';
307                              items.folder = true;
308                              items.nodes = children;
309  
310                              // Check if nodes has only one child, if so rename to package layout
311                              if (items.nodes.length === 1) {
312                                  const nodeToMerge = items.nodes[0];
313                                  if (nodeToMerge.hasOwnProperty('nodes')) { // don't merge if nodeToMerge isn't also a parent
314                                      items = {
315                                          text: items.text + '.' + nodeToMerge.text,
316                                          nodes: nodeToMerge.nodes,
317                                          href: nodeToMerge.href,
318                                          parentFile: nodeToMerge.parentFile,
319                                          file: nodeToMerge.file,
320                                          icon: 'fa fa-folder',
321                                          parentNode: nodeToMerge.parentNode,
322                                          folder: true,
323                                          parent: nodeToMerge.parent
324                                      };
325                                  }
326                              }
327  
328                              resultsFolder.push(items);
329                          } else { // file
330                              validFiles.push('/' + location + attr);
331                              if (attr.endsWith('.jar') || attr.endsWith('.aar')) {
332                                items.href = 'javaScript:void(0);';
333                              } else {
334                                items.href = href + '/' + location + attr;
335                              }
336                              resultsFiles.push(items);
337                          }
338                      }
339  
340                      const nodeSortFn = function (a, b) {
341                          return a.text.localeCompare(b.text);
342                      };
343  
344                      resultsFolder.sort(nodeSortFn);
345                      resultsFiles.sort(nodeSortFn);
346                      return resultsFolder.concat(resultsFiles);
347  
348                  }
349  
350                  const fileGetUri = env.javaUrlRoot + '/editor.html?';
351                  var ret = stage2Parser(stage1Output, fileGetUri, fileNamespace, null);
352                  console.log(validFiles);
353                  return ret;
354              }

591              const touchHandler = function touchHandlerF(event) {
592                  const touches = event.changedTouches,
593                      first = touches[0];
594                  var type = "";
595                  switch (event.type) {
596                      case "touchstart":
597                          type = "mousedown";
598                          break;
599                      case "touchmove":
600                          type = "mousemove";
601                          break;
602                      case "touchend":
603                          type = "mouseup";
604                          break;
605                      default:
606                          return;
607                  }
608  
609                  const mouseEvent = new MouseEvent(type, {
610                      'screenX': first.screenX,
611                      'screenY': first.screenY,
612                      'clientX': first.clientX,
613                      'clientY': first.clientY,
614                      'buttons': 0
615                  });
616  
617                  if (type === 'mousemove' || type === 'mouseup') {
618                      document.body.dispatchEvent(mouseEvent);
619                  } else {
620                      first.target.dispatchEvent(mouseEvent);
621                  }
622                  event.preventDefault();
623              };
