267      var handleUpdaterNotification = function(webSocketMessage) {
268          var payload = JSON.parse(webSocketMessage.payload);
269          console.log(payload);
270          var message = payload.message;
271  
272          if (payload.presentationType === "STATUS") {
273              message = message.replace(/\n/g, "<br>");
274              showUploadBlocker(message);
275          } else {
276              hideUploadBlocker();
277              removeClass(document.documentElement, "wait");
278              window.removeEventListener('beforeunload', updateAbandonmentEventHandler);
279  
280              setTimeout(function() {
281                  // We put this functionality in a 10 ms timeout, because we want the hiding to finish before we display the alert/confirmation
282                  if (payload.presentationType === "PROMPT") {
283                      var confirmationResult = confirm(message);
284                      console.log("Confirmation result: " + confirmationResult);
285                      if (confirmationResult === true) {
286                          var confirmationMessage = new WEBSOCKET_LIB.WebSocketMessage(CH_UPDATER_NAMESPACE, "confirmDangerousAction", mostRecentUploadIdentifier);
287                          WEBSOCKET_LIB.webSocketManager.sendMessage(confirmationMessage);
288                      } else {
289                          // If we aren't going to install the uploaded file, we should delete it.
290                          var deleteUploadedFileMessage = new WEBSOCKET_LIB.WebSocketMessage(CH_UPDATER_NAMESPACE, "deleteUploadedFile", mostRecentUploadIdentifier);
291                          WEBSOCKET_LIB.webSocketManager.sendMessage(deleteUploadedFileMessage);
292                      }
293                  } else {
294                      // For now we treat successes and failures the same way. In the future, we can style the messages differently if desired.
295                      alert(message);
296                  }
297              }, 10);
298          }
299      };

401      var performNextFirmwareUpdate = function() {
402          var module = remainingModulesForFwUpdate.shift();
403          if (!module) return;
404          console.log("performNextFirmwareUpdate() called, module=" + JSON.stringify(module));
405          var fwUpdateRequest = new XMLHttpRequest();
406          fwUpdateRequest.onreadystatechange = function() {
407              if (fwUpdateRequest.readyState === 4) {
408                  if (fwUpdateRequest.status === 200) {
409                      var result = JSON.parse(fwUpdateRequest.responseText);
410                      console.log("Firmware update result: " + fwUpdateRequest.responseText);
411                      if (result.success === true) {
412                          if (remainingModulesForFwUpdate.length > 0) {
413                              performNextFirmwareUpdate();
414                          } else {
415                              alert("Firmware update completed successfully")
416                          }
417                          refreshRCInfo();
418                      } else {
419                          var deviceString;
420                          if (module.serialNumber === "(embedded)") {
421                              deviceString = "Control Hub";
422                          } else {
423                              deviceString = "Expansion Hub " + module.serialNumber;
424                          }
425                          var failureMessage = "Firmware update of " + deviceString + " failed";
426                          if (result.errorMessage) {
427                              failureMessage += " because " + result.errorMessage;
428                          }
429                          failureMessage += ". Please try again.";
430                          console.log(failureMessage);
431                          alert(failureMessage);
432                      }
433                  } else {
434                      console.log("Received code " + fwUpdateRequest.status + " when updating firmware");
435                  }
436              }
437          };
438          fwUpdateRequest.open("POST", URI_PERFORM_REV_FIRMWARE_UPDATE);
439          fwUpdateRequest.setRequestHeader("Content-Type", "application/x-www-form-urlencoded");
440          var params = "serialNumber=" + module.serialNumber;
441          if (fwUpdateFilename) {
442              params += "&filename=" + fwUpdateFilename;
443          }
444          fwUpdateRequest.send(params);
445      }

613      var onSubmitNetworkSettings = function () {
614          console.log("onSubmitNetworkSettings()");
615          var newName = document.getElementById("id_rc_name").value.trim();
616          var newPw = document.getElementById("id_ap_new_password").value;
617          var confirmPw = document.getElementById("id_ap_confirmed_password").value;
618          var newChannelName = document.getElementById("id_ap_channel").value;
619          const newChannelIs5Ghz = document.getElementById("id_wifi_band_5").checked;
620          if (newName.length < 2 || newName.length > 32) {
621              alert("RC Name must be between 2 and 32 chars");
622          } else if (newName.match(/\s/) != null) {
623              alert("RC Name must not include whitespace (space, tab, return, etc..)");
624          } else if (newPw && (newPw.length < 8 || newPw.length > 63)) {
625              alert("Password must be between 8 and 63 chars");
626          } else if (newPw && newPw.localeCompare(confirmPw) !== 0) {
627              alert("New password and the confirmation password are different");
628          } else {
629              console.log("newName=" + newName + " newPw=" + newPw + " newChannelName=" + newChannelName);
630              var xhr = new XMLHttpRequest();
631              var params = PARAM_NAME + '=' + encodeURIComponent(newName);
632              if (newPw) {
633                  params += "&" + PARAM_AP_PASSWORD + '=' + encodeURIComponent(newPw);
634              }
635              if (newChannelName) {
636                  params += "&" + PARAM_CHANNEL_NAME + '=' + encodeURIComponent(newChannelName);
637              }
638              xhr.open('POST', URI_CHANGE_NETWORK_SETTINGS, true /*async*/);
639              xhr.setRequestHeader('Content-type', 'application/x-www-form-urlencoded');
640              xhr.onreadystatechange = function () {
641                  if (xhr.readyState === 4) {
642                      if (xhr.status === 200 || xhr.status === 0) {
643                          // Either the settings application succeeded, or we were disconnected from the network before we got a response.
644                          currentChannel = newChannelName;
645                          currentlyOn5Ghz = newChannelIs5Ghz;
646                          currentlyOn2_4Ghz = !newChannelIs5Ghz;
647                      } else {
648                          console.log("Network settings failed to apply. Error code: " + xhr.status + " Error text: " + xhr.responseText);
649                          var error = "The network settings failed to apply";
650                          if (xhr.responseText) {
651                              error += ". " + xhr.responseText;
652                          }
653                          alert(error);
654                      }
655                  }
656              };
657              xhr.send(params);
658  
659              if (isControlHub) {
660                  let nameChanged = originalName !== newName;
661                  let passwordChanged = originalPassword !== newPw;
662                  if (nameChanged && passwordChanged) {
663                      alert("The Control Hub's name and password have been changed. If you are connected to the Control Hub via Wi-Fi, you have been disconnected. You will need to connect to the new network.");
664                  } else if (nameChanged) {
665                      alert("The Control Hub's name has been changed. If you are connected to the Control Hub via Wi-Fi, you have been disconnected. You will need to connect to the new network.");
666                  } else if (passwordChanged) {
667                      // If the password changed but not the name, the user needs to "forget" the old network before reconnecting.
668                      alert("The Control Hub's password has been changed. If you are connected to the Control Hub via Wi-Fi, you have been disconnected. You will need to \"forget\" the network, and then connect to it again.");
669                  }
670              }
671          }
672      }

719      var updateChannelList = function (reason) {
720          var show5GhzChannels = document.getElementById("id_wifi_band_5").checked;
721          var showOverlappingChannels = document.getElementById("id_show_overlapping_channels_checkbox").checked;
722          var channelDropdown = document.getElementById("id_ap_channel");
723          const previousChannelSelection = channelDropdown.value;
724          hideElement(document.getElementById("id_show_overlapping_channels"), show5GhzChannels);
725          channelDropdown.innerText = '';
726          allAvailableChannels.forEach(function(channel) {
727              var fiveGhzChannel = (channel.band === "BAND_5_GHZ");
728              var showChannel = (show5GhzChannels && fiveGhzChannel) || (!show5GhzChannels && !fiveGhzChannel);
729              if (!showOverlappingChannels && channel.overlapsWithOtherChannels) {
730                  showChannel = false;
731              }
732              if (showChannel) {
733                  var channelOption = document.createElement("option");
734                  channelOption.value = channel.name;
735                  channelOption.textContent = channel.displayName;
736                  channelDropdown.appendChild(channelOption);
737              }
738          });
739  
740          if (reason === 'overlappingChannelsCheckbox') {
741              // When the user checks the overlapping channels checkbox, we want to restore the
742              // previous selection (if possible), not the current channel we're broadcasting on.
743              channelDropdown.value = previousChannelSelection;
744          } else {
745              if ((show5GhzChannels && currentlyOn5Ghz) || (!show5GhzChannels && currentlyOn2_4Ghz)) {
746                  channelDropdown.value = currentChannel;
747              } else {
748                  channelDropdown.value = '';
749              }
750          }
751          updateApSettingsButton();
752      }
