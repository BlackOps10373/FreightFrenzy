11   Blockly.Blocks['distanceSensor_getDistance'] = {
12     init: function() {
13       this.setOutput(true, 'Number');
14       this.appendDummyInput()
15           .appendField('call')
16           .appendField(createDistanceSensorDropdown(), 'IDENTIFIER')
17           .appendField('.')
18           .appendField(createNonEditableField('getDistance'));
19       this.appendValueInput('DISTANCE_UNIT').setCheck('DistanceUnit')
20           .appendField('distanceUnit')
21           .setAlign(Blockly.ALIGN_RIGHT);
22       this.setColour(functionColor);

24     }

27   Blockly.JavaScript['distanceSensor_getDistance'] = function(block) {
28     var identifier = block.getFieldValue('IDENTIFIER');
29     var distanceUnit = Blockly.JavaScript.valueToCode(
30         block, 'DISTANCE_UNIT', Blockly.JavaScript.ORDER_NONE);
31     var code = identifier + '.getDistance(' + distanceUnit + ')';
32     return [code, Blockly.JavaScript.ORDER_FUNCTION_CALL];
33   };
